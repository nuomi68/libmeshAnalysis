531  OldSolutionBase(const libMesh::System & sys_in,

533  last_elem(nullptr),
534  sys(sys_in),
535  old_context(sys_in, vars, /* allocate_local_matrices= */ false)
536  {
537  // We'll be queried for components but we'll typically be looking
538  // up data by variables, and those indices don't always match
539  auto make_lookup = [this](unsigned int v)
540  {
541  const unsigned int vcomp = sys.variable_scalar_number(v,0);
542  if (vcomp >= component_to_var.size())
543  component_to_var.resize(vcomp+1, static_cast<unsigned int>(-1));
544  component_to_var[vcomp] = v;
545  };
546 
547  if (vars)
548  for (auto v : *vars)
549  make_lookup(v);
550  else
551  for (auto v : make_range(sys.n_vars()))
552  make_lookup(v);
553  }

556  last_elem(nullptr),
557  sys(in.sys),

559  component_to_var(in.component_to_var)
560  {

563  static void get_shape_outputs(FEAbstract & fe);
564 
565  // Integrating on new mesh elements, we won't yet have an up to date
566  // current_local_solution.
567  void init_context (FEMContext & c)
568  {
569  c.set_algebraic_type(FEMContext::DOFS_ONLY);
570 
571  const std::set<unsigned char> & elem_dims =
572  old_context.elem_dimensions();
573 
574  // Loop over variables and dimensions, to prerequest
575  for (const auto & dim : elem_dims)
576  {
577  FEAbstract * fe = nullptr;
578  if (old_context.active_vars())
579  for (auto var : *old_context.active_vars())
580  {
581  old_context.get_element_fe(var, fe, dim);
582  get_shape_outputs(*fe);
583  }
584  else
585  for (auto var : make_range(sys.n_vars()))
586  {
587  old_context.get_element_fe(var, fe, dim);
588  get_shape_outputs(*fe);
589  }
590  }
591  }
592 
593  bool is_grid_projection() { return true; }

595 protected:
596  void check_old_context (const FEMContext & c)
597  {
598  LOG_SCOPE ("check_old_context(c)", "OldSolutionBase");
599  const Elem & elem = c.get_elem();
600  if (last_elem != &elem)
601  {
602  if (elem.refinement_flag() == Elem::JUST_REFINED)
603  {
604  old_context.pre_fe_reinit(sys, elem.parent());
605  }
606  else if (elem.refinement_flag() == Elem::JUST_COARSENED)

613  {
614  libmesh_error();
615  }

628 
629  bool check_old_context (const FEMContext & c, const Point & p)
630  {
631  LOG_SCOPE ("check_old_context(c,p)", "OldSolutionBase");
632  const Elem & elem = c.get_elem();
633  if (last_elem != &elem)
634  {
635  if (elem.refinement_flag() == Elem::JUST_REFINED)
636  {
637  old_context.pre_fe_reinit(sys, elem.parent());
638  }
639  else if (elem.refinement_flag() == Elem::JUST_COARSENED)
640  {
641  // Find the child with this point. Use out_of_elem_tol
642  // (in physical space, which may correspond to a large
643  // tolerance in master space!) to allow for out-of-element
644  // finite differencing of mixed gradient terms. Pray we
645  // have no quadrature locations which are within 1e-5 of
646  // the element subdivision boundary but are not exactly on
647  // that boundary.
648  const Real master_tol = out_of_elem_tol / elem.hmax() * 2;
649 
650  for (auto & child : elem.child_ref_range())
651  if (child.close_to_point(p, master_tol))
652  {
653  old_context.pre_fe_reinit(sys, &child);
654  break;
655  }
656 
657  libmesh_assert
658  (old_context.get_elem().close_to_point(p, master_tol));
659  }
660  else
661  {
662  if (!elem.get_old_dof_object())
663  return false;
664 
665  old_context.pre_fe_reinit(sys, &elem);
666  }
667 
668  last_elem = &elem;
669  }
670  else
671  {
672  libmesh_assert(old_context.has_elem());
673 
674  const Real master_tol = out_of_elem_tol / elem.hmax() * 2;
675 
676  if (!old_context.get_elem().close_to_point(p, master_tol))
677  {
678  libmesh_assert_equal_to
679  (elem.refinement_flag(), Elem::JUST_COARSENED);
680 
681  for (auto & child : elem.child_ref_range())
682  if (child.close_to_point(p, master_tol))
683  {
684  old_context.pre_fe_reinit(sys, &child);
685  break;
686  }
687 
688  libmesh_assert
689  (old_context.get_elem().close_to_point(p, master_tol));
690  }
691  }
692 
693  return true;
694  }

698  const System & sys;
699  FEMContext old_context;

928  libmesh_assert_not_equal_to (d_new, DofObject::invalid_id);

1465 template <typename FFunctor, typename GFunctor,
1466  typename FValue, typename ProjectionAction>
1467 void GenericProjector<FFunctor, GFunctor, FValue, ProjectionAction>::SortAndCopy::operator()
1468  (const ConstElemRange & range)
1469 {
1470  LOG_SCOPE ("SortAndCopy::operator()","GenericProjector");
1471 
1472  // Look at all the elements in the range. Directly copy values from
1473  // unchanged elements. For other elements, determine sets of
1474  // vertices, edge nodes, and side nodes to project.
1475  //
1476  // As per our other weird nomenclature, "sides" means faces in 3D
1477  // and edges in 2D, and "edges" gets skipped in 2D
1478  //
1479  // This gets tricky in the case of subdomain-restricted
1480  // variables, for which we might need to do the same projection
1481  // from different elements when evaluating different variables.
1482  // We'll keep track of which variables can be projected from which
1483  // elements.
1484 
1485  // If we copy DoFs on a node, keep track of it so we don't bother
1486  // with any redundant interpolations or projections later.
1487  //
1488  // We still need to keep track of *which* variables get copied,
1489  // since we may be able to copy elements which lack some
1490  // subdomain-restricted variables.
1491  //
1492  // For extra_hanging_dofs FE types, we'll keep track of which
1493  // variables were copied from vertices, and which from edges/sides,
1494  // because those types need copies made from *both* on hanging
1495  // nodes.
1496  std::unordered_map<const Node *, std::pair<var_set, var_set>> copied_nodes;
1497 
1498  const unsigned int sys_num = system.number();
1499 
1500  // At hanging nodes for variables with extra hanging dofs we'll need
1501  // to do projections *separately* from vertex elements and side/edge
1502  // elements.
1503  std::vector<unsigned short> extra_hanging_dofs;
1504  bool all_extra_hanging_dofs = true;
1505  for (auto v_num : this->projector.variables)
1506  {
1507  if (extra_hanging_dofs.size() <= v_num)
1508  extra_hanging_dofs.resize(v_num+1, false);
1509  extra_hanging_dofs[v_num] =
1510  FEInterface::extra_hanging_dofs(system.variable(v_num).type());
1511 
1512  if (!extra_hanging_dofs[v_num])
1513  all_extra_hanging_dofs = false;
1514  }
1515 
1516  for (const auto & elem : range)
1517  {
1518  // If we're doing AMR, we might be able to copy more DoFs than
1519  // we interpolate or project.
1520  bool copy_this_elem = false;
1521 
1522 #ifdef LIBMESH_ENABLE_AMR
1523  // If we're projecting from an old grid
1524  if (f.is_grid_projection())
1525  {
1526  // If this element doesn't have an old_dof_object, but it
1527  // wasn't just refined or just coarsened into activity, then
1528  // it must be newly added, so the user is responsible for
1529  // setting the new dofs on it during a grid projection.
1530  const DofObject * old_dof_object = elem->get_old_dof_object();
1531  const Elem::RefinementState h_flag = elem->refinement_flag();
1532  const Elem::RefinementState p_flag = elem->p_refinement_flag();
1533  if (!old_dof_object &&
1534  h_flag != Elem::JUST_REFINED &&
1535  h_flag != Elem::JUST_COARSENED)
1536  continue;
1537 
1538  // If this is an unchanged element, just copy everything
1539  if (h_flag != Elem::JUST_REFINED &&
1540  h_flag != Elem::JUST_COARSENED &&
1541  p_flag != Elem::JUST_REFINED &&
1542  p_flag != Elem::JUST_COARSENED)
1543  copy_this_elem = true;
1544  else
1545  {
1546  bool reinitted = false;
1547 
1548  const unsigned int p_level = elem->p_level();
1549 
1550  // If this element has a low order monomial which has
1551  // merely been h refined, copy it.
1552  const bool copy_possible =
1553  p_level == 0 &&
1554  h_flag != Elem::JUST_COARSENED &&
1555  p_flag != Elem::JUST_COARSENED;
1556 
1557  std::vector<typename FFunctor::ValuePushType> Ue(1);
1558  std::vector<dof_id_type> elem_dof_ids(1);
1559 
1560  for (auto v_num : this->projector.variables)
1561  {
1562  const Variable & var = system.variable(v_num);
1563  if (!var.active_on_subdomain(elem->subdomain_id()))
1564  continue;
1565  const FEType fe_type = var.type();
1566 
1567  if (fe_type.family == MONOMIAL &&
1568  fe_type.order == CONSTANT &&
1569  copy_possible)
1570  {
1571  if (!reinitted)
1572  {
1573  reinitted = true;
1574  context.pre_fe_reinit(system, elem);
1575  }
1576 
1577  f.eval_old_dofs(*elem, fe_type, sys_num, v_num,
1578  elem_dof_ids, Ue);
1579 
1580  action.insert(elem_dof_ids[0], Ue[0]);
1581  }
1582  }
1583  }
1584  }
1585 #endif // LIBMESH_ENABLE_AMR
1586 
1587  const int dim = elem->dim();
1588 
1589  const unsigned int n_vertices = elem->n_vertices();
1590  const unsigned int n_edges = elem->n_edges();
1591  const unsigned int n_nodes = elem->n_nodes();
1592 
1593  // In 1-D we already handle our sides as vertices
1594  const unsigned int n_sides = (dim > 1) * elem->n_sides();
1595 
1596  // What variables are supported on each kind of node on this elem?
1597  var_set vertex_vars, edge_vars, side_vars;
1598 
1599  // If we have non-vertex nodes, the first is an edge node, but
1600  // if we're in 2D we'll call that a side node
1601  const bool has_edge_nodes = (n_nodes > n_vertices && dim > 2);
1602 
1603  // If we have even more nodes, the next is a side node.
1604  const bool has_side_nodes =
1605  (n_nodes > n_vertices + ((dim > 2) * n_edges));
1606 
1607  // We may be out of nodes at this point or we have interior
1608  // nodes which may have DoFs to project too
1609  const bool has_interior_nodes =
1610  (n_nodes > n_vertices + ((dim > 2) * n_edges) + n_sides);
1611 
1612  for (auto v_num : this->projector.variables)
1613  {
1614  const Variable & var = this->projector.system.variable(v_num);
1615  if (!var.active_on_subdomain(elem->subdomain_id()))
1616  continue;
1617  const FEType fe_type = var.type();
1618  const auto & dof_map = this->projector.system.get_dof_map();
1619  const auto vg = dof_map.var_group_from_var_number(v_num);
1620  const bool add_p_level = dof_map.should_p_refine(vg);
1621 
1622  // If we're trying to do projections on an isogeometric
1623  // analysis mesh, only the finite element nodes constrained
1624  // by those spline nodes truly have delta function degrees
1625  // of freedom. We'll have to back out the spline degrees of
1626  // freedom indirectly later.
1627  if (fe_type.family == RATIONAL_BERNSTEIN &&
1628  elem->type() == NODEELEM)
1629  continue;
1630 
1631  if (FEInterface::n_dofs_at_node(fe_type, elem, 0, add_p_level))
1632  vertex_vars.insert(vertex_vars.end(), v_num);
1633 
1634  // The first non-vertex node is always an edge node if those
1635  // exist. All edge nodes have the same number of DoFs
1636  if (has_edge_nodes)
1637  if (FEInterface::n_dofs_at_node(fe_type, elem, n_vertices, add_p_level))
1638  edge_vars.insert(edge_vars.end(), v_num);
1639 
1640  if (has_side_nodes)
1641  {
1642  if (dim != 3)
1643  {
1644  if (FEInterface::n_dofs_at_node(fe_type, elem, n_vertices, add_p_level))
1645  side_vars.insert(side_vars.end(), v_num);
1646  }
1647  else
1648  // In 3D, not all face nodes always have the same number of
1649  // DoFs! We'll loop over all sides to be safe.
1650  for (unsigned int n = 0; n != n_nodes; ++n)
1651  if (elem->is_face(n))
1652  if (FEInterface::n_dofs_at_node(fe_type, elem, n, add_p_level))
1653  {
1654  side_vars.insert(side_vars.end(), v_num);
1655  break;
1656  }
1657  }
1658 
1659  if (FEInterface::n_dofs_per_elem(fe_type, elem, add_p_level) ||
1660  (has_interior_nodes &&
1661  FEInterface::n_dofs_at_node(fe_type, elem, n_nodes-1, add_p_level)))
1662  {
1663 #ifdef LIBMESH_ENABLE_AMR
1664  // We may have already just copied constant monomials,
1665  // or we may be about to copy the whole element
1666  if ((f.is_grid_projection() &&
1667  fe_type.family == MONOMIAL &&
1668  fe_type.order == CONSTANT &&
1669  elem->p_level() == 0 &&
1670  elem->refinement_flag() != Elem::JUST_COARSENED &&
1671  elem->p_refinement_flag() != Elem::JUST_COARSENED)
1672  || copy_this_elem
1673  )
1674  continue;
1675 #endif // LIBMESH_ENABLE_AMR
1676 
1677  // We need to project any other variables
1678  if (interiors.empty() || interiors.back() != elem)
1679  interiors.push_back(elem);
1680  }
1681  }
1682 
1683  // We'll use a greedy algorithm in most cases: if another
1684  // element has already claimed some of our DoFs, we'll let it do
1685  // the work.
1686  auto erase_covered_vars = []
1687  (const Node * node, var_set & remaining, const ves_multimap & covered)
1688  {
1689  auto covered_range = covered.equal_range(node);
1690  for (const auto & v_ent : as_range(covered_range))
1691  for (const unsigned int var_covered :
1692  std::get<2>(v_ent.second))
1693  remaining.erase(var_covered);
1694  };
1695 
1696  auto erase_nonhanging_vars = [&extra_hanging_dofs]
1697  (const Node * node, var_set & remaining, const ves_multimap & covered)
1698  {
1699  auto covered_range = covered.equal_range(node);
1700  for (const auto & v_ent : as_range(covered_range))
1701  for (const unsigned int var_covered :
1702  std::get<2>(v_ent.second))
1703  if (!extra_hanging_dofs[var_covered])
1704  remaining.erase(var_covered);
1705  };
1706 
1707  auto erase_copied_vars = [&copied_nodes, &extra_hanging_dofs]
1708  (const Node * node, bool is_vertex, var_set & remaining)
1709  {
1710  auto copying_range = copied_nodes.equal_range(node);
1711  for (const auto & v_ent : as_range(copying_range))
1712  {
1713  for (const unsigned int var_covered :
1714  v_ent.second.first)
1715  if (is_vertex || !extra_hanging_dofs[var_covered])
1716  remaining.erase(var_covered);
1717 
1718  for (const unsigned int var_covered :
1719  v_ent.second.second)
1720  if (!is_vertex || !extra_hanging_dofs[var_covered])
1721  remaining.erase(var_covered);
1722  }
1723  };
1724 
1725  for (unsigned int v=0; v != n_vertices; ++v)
1726  {
1727  const Node * node = elem->node_ptr(v);
1728 
1729  auto remaining_vars = vertex_vars;
1730 
1731  erase_covered_vars(node, remaining_vars, vertices);
1732 
1733  if (remaining_vars.empty())
1734  continue;
1735 
1736  if (!all_extra_hanging_dofs)
1737  {
1738  erase_nonhanging_vars(node, remaining_vars, edges);
1739  if (remaining_vars.empty())
1740  continue;
1741 
1742  erase_nonhanging_vars(node, remaining_vars, sides);
1743  if (remaining_vars.empty())
1744  continue;
1745  }
1746 
1747  erase_copied_vars(node, true, remaining_vars);
1748  if (remaining_vars.empty())
1749  continue;
1750 
1751  if (copy_this_elem)
1752  {
1753  std::vector<dof_id_type> node_dof_ids;
1754  std::vector<typename FFunctor::ValuePushType> values;
1755 
1756  for (auto var : remaining_vars)
1757  {
1758  f.eval_old_dofs(*elem, v, var, node_dof_ids, values);
1759  insert_ids(node_dof_ids, values, node->processor_id());
1760  }
1761  copied_nodes[node].first.insert(remaining_vars.begin(),
1762  remaining_vars.end());
1763  this->find_dofs_to_send(*node, *elem, v, remaining_vars);
1764  }
1765  else
1766  vertices.emplace
1767  (node, std::make_tuple(elem, v, std::move(remaining_vars)));
1768  }
1769 
1770  if (has_edge_nodes)
1771  {
1772  for (unsigned int e=0; e != n_edges; ++e)
1773  {
1774  const Node * node = elem->node_ptr(n_vertices+e);
1775 
1776  auto remaining_vars = edge_vars;
1777 
1778  erase_covered_vars(node, remaining_vars, edges);
1779  if (remaining_vars.empty())
1780  continue;
1781 
1782  erase_covered_vars(node, remaining_vars, sides);
1783  if (remaining_vars.empty())
1784  continue;
1785 
1786  if (!all_extra_hanging_dofs)
1787  {
1788  erase_nonhanging_vars(node, remaining_vars, vertices);
1789  if (remaining_vars.empty())
1790  continue;
1791  }
1792 
1793  erase_copied_vars(node, false, remaining_vars);
1794  if (remaining_vars.empty())
1795  continue;
1796 
1797  if (copy_this_elem)
1798  {
1799  std::vector<dof_id_type> edge_dof_ids;
1800  std::vector<typename FFunctor::ValuePushType> values;
1801 
1802  for (auto var : remaining_vars)
1803  {
1804  f.eval_old_dofs(*elem, n_vertices+e, var, edge_dof_ids, values);
1805  insert_ids(edge_dof_ids, values, node->processor_id());
1806  }
1807 
1808  copied_nodes[node].second.insert(remaining_vars.begin(),
1809  remaining_vars.end());
1810  this->find_dofs_to_send(*node, *elem, n_vertices+e, remaining_vars);
1811  }
1812  else
1813  edges.emplace
1814  (node, std::make_tuple(elem, e, std::move(remaining_vars)));
1815  }
1816  }
1817 
1818  if (has_side_nodes)
1819  {
1820  for (unsigned int side=0; side != n_sides; ++side)
1821  {
1822  const Node * node = nullptr;
1823  unsigned short node_num = n_vertices+(dim>2)*n_edges+side;
1824  if (dim != 3)
1825  node = elem->node_ptr(node_num);
1826  else
1827  {
1828  // In 3D only some sides may have nodes
1829  for (unsigned int n = 0; n != n_nodes; ++n)

1832  continue;

1850  continue;
1851 
1852  erase_covered_vars(node, remaining_vars, sides);
1853  if (remaining_vars.empty())

1918 void GenericProjector<FFunctor, GFunctor, FValue, ProjectionAction>::SortAndCopy::join
1919  (const GenericProjector<FFunctor, GFunctor, FValue, ProjectionAction>::SortAndCopy & other)
1920 {
1921  auto merge_multimaps = [](ves_multimap & self, const ves_multimap & other_mm)
1922  {
1923  for (const auto & pair : other_mm)
1924  {
1925  const Node * node = pair.first;
1926  auto remaining_vars = std::get<2>(pair.second);
1927 
1928  auto my_range = self.equal_range(node);
1929  for (const auto & v_ent : as_range(my_range))
1930  for (const unsigned int var_covered :
1931  std::get<2>(v_ent.second))
1932  remaining_vars.erase(var_covered);
1933 
1934  if (!remaining_vars.empty())
1935  self.emplace
1936  (node, std::make_tuple(std::get<0>(pair.second),
1937  std::get<1>(pair.second),
1938  std::move(remaining_vars)));
1939  }
1940  };
1941 
1942  merge_multimaps(vertices, other.vertices);
1943  merge_multimaps(edges, other.edges);
1944  merge_multimaps(sides, other.sides);
1945 
1946  std::sort(interiors.begin(), interiors.end());
1947  std::vector<const Elem *> other_interiors = other.interiors;
1948  std::sort(other_interiors.begin(), other_interiors.end());
1949  std::vector<const Elem *> merged_interiors;
1950  std::set_union(interiors.begin(), interiors.end(),
1951  other_interiors.begin(), other_interiors.end(),
1952  std::back_inserter(merged_interiors));
1953  interiors.swap(merged_interiors);
1954 
1955  SubFunctor::join(other);
