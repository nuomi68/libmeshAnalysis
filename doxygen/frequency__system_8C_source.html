59 void FrequencySystem::clear ()
60 {
61  LinearImplicitSystem::clear();
62 
63  _finished_set_frequencies = false;
64  _keep_solution_duplicates = true;
65  _finished_init = false;
66  _finished_assemble = false;
67 
68  // We have to distinguish between the
69  // simple straightforward "clear()"

82 {
83  this->clear ();

85  EquationSystems & es = this->get_equation_systems();

87  // clear frequencies in the parameters section of the
88  // EquationSystems object

102 {

107  LOG_SCOPE("init()", "FrequencySystem");

126  _finished_set_frequencies = true;
127 
128  this->set_current_frequency(0.);

161 void FrequencySystem::set_frequencies_by_steps (const Number base_freq,
162  const Number freq_step,
163  const unsigned int n_freq,
164  const bool allocate_solution_duplicates)
165 {
166  this->_keep_solution_duplicates = allocate_solution_duplicates;
167 
168  // sanity check
169  libmesh_error_msg_if(_finished_set_frequencies, "ERROR: frequencies already initialized.");
170 
171  EquationSystems & es =
172  this->get_equation_systems();
173 
174  // store number of frequencies as parameter
175  es.parameters.set<unsigned int>("n_frequencies") = n_freq;
176 
177  for (unsigned int n=0; n<n_freq; n++)
178  {
179  // remember frequencies as parameters
180  es.parameters.set<Number>(this->form_freq_param_name(n)) =
181  base_freq + Number(n) * freq_step;
182 
183  // build storage for solution vector, if wanted
184  if (this->_keep_solution_duplicates)
185  this->add_vector(this->form_solu_vec_name(n));
186  }
187 
188  _finished_set_frequencies = true;
189 
190  // set the current frequency
191  this->set_current_frequency(0);
192 }

196 void FrequencySystem::set_frequencies_by_range (const Number min_freq,
197  const Number max_freq,
198  const unsigned int n_freq,
199  const bool allocate_solution_duplicates)
200 {
201  this->_keep_solution_duplicates = allocate_solution_duplicates;
202 
203  // sanity checks. Only look at the real part.
204  libmesh_assert_greater_equal (std::real(max_freq), std::real(min_freq));
205  libmesh_assert_greater (n_freq, 0);
206  libmesh_error_msg_if(_finished_set_frequencies, "ERROR: frequencies already initialized.");
207 
208  EquationSystems & es =
209  this->get_equation_systems();
210 
211  // store number of frequencies as parameter
212  es.parameters.set<unsigned int>("n_frequencies") = n_freq;
213 
214  // set frequencies, build solution storage
215  for (unsigned int n=0; n<n_freq; n++)
216  {
217  // remember frequencies as parameters
218  es.parameters.set<Number>(this->form_freq_param_name(n)) =
219  min_freq + static_cast<Number>(n)*(max_freq-min_freq)/static_cast<Number>(n_freq-1);
220 
221  // build storage for solution vector, if wanted
222  if (this->_keep_solution_duplicates)
223  System::add_vector(this->form_solu_vec_name(n));
224  }
225 
226  _finished_set_frequencies = true;
227 
228  // set the current frequency
229  this->set_current_frequency(0);
230 }

322 void FrequencySystem::solve (const unsigned int n_start,
323  const unsigned int n_stop)
324 {
325  // Assemble the linear system, if not already done
326  if (!_finished_assemble)
327  this->assemble ();
328 
329  // the user-supplied solve method _has_ to be provided by the user
330  libmesh_assert(solve_system);
331 
332  // existence & range checks
333  libmesh_assert_greater (this->n_frequencies(), 0);
334  libmesh_assert_less (n_stop, this->n_frequencies());
335 
336  EquationSystems & es =
337  this->get_equation_systems();
338 
339  // Get the user-specified linear solver tolerance,
340  // the user-specified maximum # of linear solver iterations,
341  // the user-specified wave speed
342  const Real tol =
343  es.parameters.get<Real>("linear solver tolerance");
344  const unsigned int maxits =
345  es.parameters.get<unsigned int>("linear solver maximum iterations");
346 
347  // start solver loop
348  for (unsigned int n=n_start; n<= n_stop; n++)
349  {
350  // set the current frequency
351  this->set_current_frequency(n);
352 
353  // Call the user-supplied pre-solve method
354  LOG_CALL("user_pre_solve()", "FrequencySystem", this->solve_system(es, this->name()));
355 
356  // Solve the linear system for this specific frequency
357  const std::pair<unsigned int, Real> rval =
358  linear_solver->solve (*matrix, *solution, *rhs, tol, maxits);
359 
360  std::tie(_n_linear_iterations, _final_linear_residual) = rval;
361  vec_rval.push_back(rval);
362 
366  if (this->_keep_solution_duplicates)
367  this->get_vector(this->form_solu_vec_name(n)) = *solution;
368  }
369 
370  // sanity check
371  //libmesh_assert_equal_to (vec_rval.size(), (n_stop-n_start+1));
372 }
