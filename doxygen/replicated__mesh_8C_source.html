269 Elem * ReplicatedMesh::add_elem (Elem * e)
270 {
271  libmesh_assert(e);
272 
273  // We no longer merely append elements with ReplicatedMesh
274 
275  // If the user requests a valid id that doesn't correspond to an
276  // existing element, let's give them that id, resizing the elements
277  // container if necessary.
278  if (!e->valid_id())
279  e->set_id (cast_int<dof_id_type>(_elements.size()));
280 
281 #ifdef LIBMESH_ENABLE_UNIQUE_ID
282  if (!e->valid_unique_id())
283  e->set_unique_id(_next_unique_id++);
284  else
285  _next_unique_id = std::max(_next_unique_id, e->unique_id()+1);
286 #endif
287 
288  const dof_id_type id = e->id();
289 
290  if (id < _elements.size())
291  {
292  // This should *almost* never happen, but we rely on it when
293  // using allgather to replicate a not-yet-actually-replicated
294  // ReplicatedMesh under construction in parallel.
295  if (e == _elements[id])
296  return e;
297 
298  // Overwriting existing elements is still probably a mistake.
299  libmesh_assert(!_elements[id]);
300  }
301  else
302  {
303  _elements.resize(id+1, nullptr);
304  }
305 
306  ++_n_elem;
307  _elements[id] = e;
308 
309  // Make sure any new element is given space for any extra integers
310  // we've requested
311  e->add_extra_integers(_elem_integer_names.size(),
312  _elem_integer_default_values);
313 
314  // And set mapping type and data on any new element
315  e->set_mapping_type(this->default_mapping_type());
316  e->set_mapping_data(this->default_mapping_data());
317 
318  return e;
319 }

375 void ReplicatedMesh::delete_elem(Elem * e)
376 {
377  libmesh_assert(e);
378 
379  // Initialize an iterator to eventually point to the element we want to delete
380  std::vector<Elem *>::iterator pos = _elements.end();
381 
382  // In many cases, e->id() gives us a clue as to where e
383  // is located in the _elements vector. Try that first
384  // before trying the O(n_elem) search.
385  libmesh_assert_less (e->id(), _elements.size());
386 
387  if (_elements[e->id()] == e)
388  {
389  // We found it!
390  pos = _elements.begin();
391  std::advance(pos, e->id());
392  }
393 
394  else
395  {
396  // This search is O(n_elem)
397  pos = std::find (_elements.begin(),
398  _elements.end(),
399  e);
400  }
401 
402  // Huh? Element not in the vector?
403  libmesh_assert (pos != _elements.end());
404 
405  // Remove the element from the BoundaryInfo object
406  this->get_boundary_info().remove(e);
407 
408  // delete the element
409  --_n_elem;
410  delete e;
411 
412  // explicitly zero the pointer
413  *pos = nullptr;
414 }

433 Node * ReplicatedMesh::add_point (const Point & p,
434  const dof_id_type id,
435  const processor_id_type proc_id)
436 {
437  Node * n = nullptr;
438 
439  // If the user requests a valid id, either
440  // provide the existing node or resize the container
441  // to fit the new node.
442  if (id != DofObject::invalid_id)
443  if (id < _nodes.size())
444  n = _nodes[id];
445  else
446  _nodes.resize(id+1);
447  else
448  _nodes.push_back (static_cast<Node *>(nullptr));
449 
450  // if the node already exists, then assign new (x,y,z) values
451  if (n)
452  *n = p;
453  // otherwise build a new node, put it in the right spot, and return
454  // a valid pointer.
455  else
456  {
457  n = Node::build(p, (id == DofObject::invalid_id) ?
458  cast_int<dof_id_type>(_nodes.size()-1) : id).release();
459  n->processor_id() = proc_id;
460 
461  n->add_extra_integers(_node_integer_names.size(),
462  _node_integer_default_values);
463 
464 #ifdef LIBMESH_ENABLE_UNIQUE_ID
465  if (!n->valid_unique_id())
466  n->set_unique_id(_next_unique_id++);
467  else
468  _next_unique_id = std::max(_next_unique_id, n->unique_id()+1);
469 #endif
470 
471  ++_n_nodes;
472  if (id == DofObject::invalid_id)
473  _nodes.back() = n;
474  else
475  _nodes[id] = n;
476  }
477 
478  // better not pass back a nullptr.
479  libmesh_assert (n);
480 
481  return n;
482 }

533 Node * ReplicatedMesh::insert_node(Node * n)
534 {
535  libmesh_deprecated();
536  libmesh_error_msg_if(!n, "Error, attempting to insert nullptr node.");
537  libmesh_error_msg_if(n->id() == DofObject::invalid_id, "Error, cannot insert node with invalid id.");
538 
539  if (n->id() < _nodes.size())
540  {
541  // Trying to add an existing node is a no-op. This lets us use
542  // a straightforward MeshCommunication::allgather() to fix a
543  // not-actually-replicated-yet ReplicatedMesh, as occurs when
544  // reading Nemesis files.
545  if (n->valid_id() && _nodes[n->id()] == n)
546  return n;
547 
548  // Don't allow anything else to replace an existing Node.
549  libmesh_error_msg_if(_nodes[ n->id() ] != nullptr,
550  "Error, cannot insert new node on top of existing node.");
551  }
552  else
553  {
554  // Allocate just enough space to store the new node. This will
555  // cause highly non-ideal memory allocation behavior if called
556  // repeatedly...
557  _nodes.resize(n->id() + 1);
558  }
559 
560 #ifdef LIBMESH_ENABLE_UNIQUE_ID
561  if (!n->valid_unique_id())
562  n->set_unique_id(_next_unique_id++);
563  else
564  _next_unique_id = std::max(_next_unique_id, n->unique_id()+1);
565 #endif
566 
567  n->add_extra_integers(_node_integer_names.size(),
568  _node_integer_default_values);
569 
570  // We have enough space and this spot isn't already occupied by
571  // another node, so go ahead and add it.
572  ++_n_nodes;
573  _nodes[ n->id() ] = n;
574 
575  // If we made it this far, we just inserted the node the user handed
576  // us, so we can give it right back.
577  return n;
578 }

708 void ReplicatedMesh::renumber_nodes_and_elements ()
709 {
710  LOG_SCOPE("renumber_nodes_and_elem()", "Mesh");
711 
712  // node and element id counters
713  dof_id_type next_free_elem = 0;
714  dof_id_type next_free_node = 0;
715 
716  // Will hold the set of nodes that are currently connected to elements
717  std::unordered_set<Node *> connected_nodes;
718 
719  // Loop over the elements. Note that there may
720  // be nullptrs in the _elements vector from the coarsening
721  // process. Pack the elements in to a contiguous array
722  // and then trim any excess.
723  {
724  std::vector<Elem *>::iterator in = _elements.begin();
725  std::vector<Elem *>::iterator out_iter = _elements.begin();
726  const std::vector<Elem *>::iterator end = _elements.end();
727 
728  for (; in != end; ++in)
729  if (*in != nullptr)
730  {
731  Elem * el = *in;
732 
733  *out_iter = *in;
734  ++out_iter;
735 
736  // Increment the element counter
737  el->set_id (next_free_elem++);
738 
739  if (_skip_renumber_nodes_and_elements)
740  {
741  // Add this elements nodes to the connected list
742  for (auto & n : el->node_ref_range())
743  connected_nodes.insert(&n);
744  }
745  else // We DO want node renumbering
746  {
747  // Loop over this element's nodes. Number them,
748  // if they have not been numbered already. Also,
749  // position them in the _nodes vector so that they
750  // are packed contiguously from the beginning.
751  for (auto & n : el->node_ref_range())
752  if (n.id() == next_free_node) // don't need to process
753  next_free_node++; // [(src == dst) below]
754 
755  else if (n.id() > next_free_node) // need to process
756  {
757  // The source and destination indices
758  // for this node
759  const dof_id_type src_idx = n.id();
760  const dof_id_type dst_idx = next_free_node++;
761 
762  // ensure we want to swap a valid nodes
763  libmesh_assert(_nodes[src_idx]);
764 
765  // Swap the source and destination nodes
766  std::swap(_nodes[src_idx],
767  _nodes[dst_idx] );
768 
769  // Set proper indices where that makes sense
770  if (_nodes[src_idx] != nullptr)
771  _nodes[src_idx]->set_id (src_idx);
772  _nodes[dst_idx]->set_id (dst_idx);
773  }
774  }
775  }
776 
777  // Erase any additional storage. These elements have been
778  // copied into nullptr voids by the procedure above, and are
779  // thus repeated and unnecessary.
780  _elements.erase (out_iter, end);
781  }
782 
783 
784  if (_skip_renumber_nodes_and_elements)
785  {
786  // Loop over the nodes. Note that there may
787  // be nullptrs in the _nodes vector from the coarsening
788  // process. Pack the nodes in to a contiguous array
789  // and then trim any excess.
790 
791  std::vector<Node *>::iterator in = _nodes.begin();
792  std::vector<Node *>::iterator out_iter = _nodes.begin();
793  const std::vector<Node *>::iterator end = _nodes.end();
794 
795  for (; in != end; ++in)
796  if (*in != nullptr)
797  {
798  // This is a reference so that if we change the pointer it will change in the vector
799  Node * & nd = *in;
800 
801  // If this node is still connected to an elem, put it in the list
802  if (connected_nodes.find(nd) != connected_nodes.end())
803  {
804  *out_iter = nd;
805  ++out_iter;
806 
807  // Increment the node counter
808  nd->set_id (next_free_node++);
809  }
810  else // This node is orphaned, delete it!
811  {
812  this->get_boundary_info().remove (nd);
813  _constraint_rows.erase(nd);
814 
815  // delete the node
816  --_n_nodes;
817  delete nd;
818  nd = nullptr;
819  }
820  }
821 
822  // Erase any additional storage. Whatever was
823  _nodes.erase (out_iter, end);
824  }
825  else // We really DO want node renumbering
826  {
827  // Any nodes in the vector >= _nodes[next_free_node]
828  // are not connected to any elements and may be deleted
829  // if desired.
830 
831  // Now, delete the unused nodes
832  {
833  std::vector<Node *>::iterator nd = _nodes.begin();
834  const std::vector<Node *>::iterator end = _nodes.end();
835 
836  std::advance (nd, next_free_node);
837 
838  for (auto & node : as_range(nd, end))
839  {
840  // Mesh modification code might have already deleted some
841  // nodes
842  if (node == nullptr)
843  continue;
844 
845  // remove any boundary information associated with
846  // this node
847  this->get_boundary_info().remove (node);
848  _constraint_rows.erase(node);
849 
850  // delete the node
851  --_n_nodes;
852  delete node;
853  node = nullptr;
854  }
855 
856  _nodes.erase (nd, end);
857  }
858  }
859 
860  libmesh_assert_equal_to (next_free_elem, _elements.size());
861  libmesh_assert_equal_to (next_free_node, _nodes.size());
862 
863  this->update_parallel_id_counts();
864 }

888 std::vector<dof_id_type>
889 ReplicatedMesh::get_disconnected_subdomains(std::vector<subdomain_id_type> * subdomain_ids) const
890 {
891  // find number of disconnected subdomains
892  std::vector<dof_id_type> representative_elem_ids;
893 
894  // use subdomain_ids as markers for all elements to indicate if the elements
895  // have been visited. Note: here subdomain ID is unrelated with element
896  // subdomain_id().
897  std::vector<subdomain_id_type> subdomains;
898  if (!subdomain_ids)
899  subdomain_ids = &subdomains;
900  subdomain_ids->clear();
901  subdomain_ids->resize(max_elem_id() + 1, Elem::invalid_subdomain_id);
902 
903  // counter of disconnected subdomains
904  subdomain_id_type subdomain_counter = 0;
905 
906  // a stack for visiting elements, make its capacity sufficiently large to avoid
907  // memory allocation and deallocation when the vector size changes
908  std::vector<const Elem *> list;
909  list.reserve(n_elem());
910 
911  // counter of visited elements
912  dof_id_type visited = 0;
913  dof_id_type n_active = n_active_elem();
914  do
915  {
916  for (const auto & elem : active_element_ptr_range())
917  if ((*subdomain_ids)[elem->id()] == Elem::invalid_subdomain_id)
918  {
919  list.push_back(elem);
920  (*subdomain_ids)[elem->id()] = subdomain_counter;
921  break;
922  }
923  // we should be able to find a seed here
924  libmesh_assert(list.size() > 0);
925 
926  dof_id_type min_id = std::numeric_limits<dof_id_type>::max();
927  while (list.size() > 0)
928  {
929  // pop up an element
930  const Elem * elem = list.back(); list.pop_back(); ++visited;
931 
932  min_id = std::min(elem->id(), min_id);
933 
934  for (auto s : elem->side_index_range())
935  {
936  const Elem * neighbor = elem->neighbor_ptr(s);
937  if (neighbor != nullptr && (*subdomain_ids)[neighbor->id()] == Elem::invalid_subdomain_id)
938  {
939  // neighbor must be active
940  libmesh_assert(neighbor->active());
941  list.push_back(neighbor);
942  (*subdomain_ids)[neighbor->id()] = subdomain_counter;
943  }
944  }
945  }
946 
947  representative_elem_ids.push_back(min_id);
948  subdomain_counter++;
949  }
950  while (visited != n_active);
951 
952  return representative_elem_ids;
953 }

955 std::unordered_map<dof_id_type, std::vector<std::vector<Point>>>
956 ReplicatedMesh::get_boundary_points() const
957 {
958  libmesh_error_msg_if(mesh_dimension() != 2,
959  "Error: get_boundary_points only works for 2D now");
960 
961  // find number of disconnected subdomains
962  // subdomains will hold the IDs of disconnected subdomains for all elements.
963  std::vector<subdomain_id_type> subdomains;
964  std::vector<dof_id_type> elem_ids = get_disconnected_subdomains(&subdomains);
965 
966  std::unordered_map<dof_id_type, std::vector<std::vector<Point>>> boundary_points;
967 
968  // get all boundary sides that are to be erased later during visiting
969  // use a comparison functor to avoid run-time randomness due to pointers
970  struct boundary_side_compare
971  {
972  bool operator()(const std::pair<const Elem *, unsigned int> & lhs,
973  const std::pair<const Elem *, unsigned int> & rhs) const
974  {
975  if (lhs.first->id() < rhs.first->id())
976  return true;
977  else if (lhs.first->id() == rhs.first->id())
978  {
979  if (lhs.second < rhs.second)
980  return true;
981  }
982  return false;
983  }
984  };
985  std::set<std::pair<const Elem *, unsigned int>, boundary_side_compare> boundary_elements;
986  for (const auto & elem : active_element_ptr_range())
987  for (auto s : elem->side_index_range())
988  if (elem->neighbor_ptr(s) == nullptr)
989  boundary_elements.insert(std::pair<const Elem *, unsigned int>(elem, s));
990 
991  while (!boundary_elements.empty())
992  {
993  // get the first entry as the seed
994  const Elem * eseed = boundary_elements.begin()->first;
995  unsigned int sseed = boundary_elements.begin()->second;
996 
997  // get the subdomain ID that these boundary sides attached to
998  subdomain_id_type subdomain_id = subdomains[eseed->id()];
999 
1000  // start visiting the mesh to find all boundary nodes with the seed
1001  std::vector<Point> bpoints;
1002  const Elem * elem = eseed;
1003  unsigned int s = sseed;
1004  std::vector<unsigned int> local_side_nodes = elem->nodes_on_side(s);
1005  while (true)
1006  {
1007  std::pair<const Elem *, unsigned int> side(elem, s);
1008  libmesh_assert(boundary_elements.find(side) != boundary_elements.end());
1009  boundary_elements.erase(side);
1010 
1011  // push all nodes on the side except the node on the other end of the side (index 1)
1012  for (auto i : index_range(local_side_nodes))
1013  if (i != 1)
1014  bpoints.push_back(*static_cast<const Point *>(elem->node_ptr(local_side_nodes[i])));
1015 
1016  // use the last node to find next element and side
1017  const Node * node = elem->node_ptr(local_side_nodes[1]);
1018  std::set<const Elem *> neighbors;
1019  elem->find_point_neighbors(*node, neighbors);
1020 
1021  // if only one neighbor is found (itself), this node is a cornor node on boundary
1022  if (neighbors.size() != 1)
1023  neighbors.erase(elem);
1024 
1025  // find the connecting side
1026  bool found = false;
1027  for (const auto & neighbor : neighbors)
1028  {
1029  for (auto ss : neighbor->side_index_range())
1030  if (neighbor->neighbor_ptr(ss) == nullptr && !(elem == neighbor && s == ss))
1031  {
1032  local_side_nodes = neighbor->nodes_on_side(ss);
1033  // we expect the starting point of the side to be the same as the end of the previous side
1034  if (neighbor->node_ptr(local_side_nodes[0]) == node)
1035  {
1036  elem = neighbor;
1037  s = ss;
1038  found = true;
1039  break;
1040  }
1041  else if (neighbor->node_ptr(local_side_nodes[1]) == node)
1042  {
1043  elem = neighbor;
1044  s = ss;
1045  found = true;
1046  // flip nodes in local_side_nodes because the side is in an opposite direction
1047  auto temp(local_side_nodes);
1048  local_side_nodes[0] = temp[1];
1049  local_side_nodes[1] = temp[0];
1050  for (unsigned int i = 2; i < temp.size(); ++i)
1051  local_side_nodes[temp.size() + 1 - i] = temp[i];
1052  break;
1053  }
1054  }
1055  if (found)
1056  break;
1057  }
1058 
1059  libmesh_error_msg_if(!found, "ERROR: mesh topology error on visiting boundary sides");
1060 
1061  // exit if we reach the starting point
1062  if (elem == eseed && s == sseed)
1063  break;
1064  }
1065  boundary_points[elem_ids[subdomain_id]].push_back(bpoints);
1066  }
1067 
1068  return boundary_points;
1069 }
