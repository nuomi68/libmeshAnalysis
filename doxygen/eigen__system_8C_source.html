85 void EigenSystem::set_eigenproblem_type (EigenProblemType ept)
86 {
87  if (!can_add_matrices())
88  libmesh_error_msg("ERROR: Cannot change eigen problem type after system initialization");
89 
90  _eigen_problem_type = ept;
91 
92  eigen_solver->set_eigenproblem_type(ept);
93 
94  // libMesh::out << "The Problem type is set to be: " << std::endl;
95 
96  switch (_eigen_problem_type)
97  {
98  case HEP:
99  // libMesh::out << "Hermitian" << std::endl;
100  break;
101 
102  case NHEP:
103  // libMesh::out << "Non-Hermitian" << std::endl;
104  break;
105 
106  case GHEP:
107  // libMesh::out << "Generalized Hermitian" << std::endl;
108  break;
109 
110  case GNHEP:
111  // libMesh::out << "Generalized Non-Hermitian" << std::endl;
112  break;
113 
114  case GHIEP:
115  // libMesh::out << "Generalized indefinite Hermitian" << std::endl;
116  break;
117 
118  default:
119  // libMesh::out << "not properly specified" << std::endl;
120  libmesh_error_msg("Unrecognized _eigen_problem_type = " << _eigen_problem_type);
121  }

126 void EigenSystem::add_matrices ()
127 {
128  Parent::add_matrices();
129 
130  if (_use_shell_matrices)
131  {
132  if (!shell_matrix_A)
133  shell_matrix_A = ShellMatrix<Number>::build(this->comm());
134 
135  if (generalized() && !shell_matrix_B)
136  shell_matrix_B = ShellMatrix<Number>::build(this->comm());
137 
138  if (_use_shell_precond_matrix)
139  {
140  if (!shell_precond_matrix)
141  shell_precond_matrix = ShellMatrix<Number>::build(this->comm());
142  }
143  else if (!precond_matrix)
144  precond_matrix = &(this->add_matrix("Eigen Preconditioner"));
145  }
146  else
147  {
148  if (!matrix_A)
149  matrix_A = &(this->add_matrix("Eigen Matrix A"));
150 
151  if (generalized() && !matrix_B)
152  matrix_B = &(this->add_matrix("Eigen Matrix B"));
153  }
154 }

209 void EigenSystem::solve ()
210 {
211  if (get_dof_map().n_constrained_dofs())
212  libmesh_warning("EigenSystem does not have first-class support for constrained degrees of "
213  "freedom. You may see spurious effects of the constrained degrees of freedom "
214  "in a given eigenvector. If you wish to perform a reliable solve on a system "
215  "with constraints, please use the CondensedEigenSystem class instead");
216 
217  // A reference to the EquationSystems
218  EquationSystems & es = this->get_equation_systems();
219 
220  // check that necessary parameters have been set
221  libmesh_assert (es.parameters.have_parameter<unsigned int>("eigenpairs"));
222  libmesh_assert (es.parameters.have_parameter<unsigned int>("basis vectors"));
223 
224  if (this->assemble_before_solve)
225  // Assemble the linear system
226  this->assemble ();
227 
228  // Get the tolerance for the solver and the maximum
229  // number of iterations. Here, we simply adopt the linear solver
230  // specific parameters.
231  const double tol =
232  double(es.parameters.get<Real>("linear solver tolerance"));
233 
234  const unsigned int maxits =
235  es.parameters.get<unsigned int>("linear solver maximum iterations");
236 
237  const unsigned int nev =
238  es.parameters.get<unsigned int>("eigenpairs");
239 
240  const unsigned int ncv =
241  es.parameters.get<unsigned int>("basis vectors");
242 
243  std::pair<unsigned int, unsigned int> solve_data;
244 
245  // call the solver depending on the type of eigenproblem
246 
247  if (_use_shell_matrices)
248  {
249  // Generalized eigenproblem
250  if (generalized())
251  // Shell preconditioning matrix
252  if (_use_shell_precond_matrix)
253  solve_data = eigen_solver->solve_generalized (*shell_matrix_A, *shell_matrix_B,*shell_precond_matrix, nev, ncv, tol, maxits);
254  else
255  solve_data = eigen_solver->solve_generalized (*shell_matrix_A, *shell_matrix_B,*precond_matrix, nev, ncv, tol, maxits);
256 
257  // Standard eigenproblem
258  else
259  {
260  libmesh_assert (!shell_matrix_B);
261  // Shell preconditioning matrix
262  if (_use_shell_precond_matrix)
263  solve_data = eigen_solver->solve_standard (*shell_matrix_A,*shell_precond_matrix, nev, ncv, tol, maxits);
264  else
265  solve_data = eigen_solver->solve_standard (*shell_matrix_A,*precond_matrix, nev, ncv, tol, maxits);
266  }
267  }
268  else
269  {
270  // Generalized eigenproblem
271  if (generalized())
272  solve_data = eigen_solver->solve_generalized (*matrix_A, *matrix_B, nev, ncv, tol, maxits);
273 
274  // Standard eigenproblem
275  else
276  {
277  libmesh_assert (!matrix_B);
278  solve_data = eigen_solver->solve_standard (*matrix_A, nev, ncv, tol, maxits);
279  }
280  }
281 
282  this->_n_converged_eigenpairs = solve_data.first;
283  this->_n_iterations = solve_data.second;
284 }
