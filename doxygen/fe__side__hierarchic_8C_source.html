36 void side_hierarchic_nodal_soln(const Elem * elem,
37  const Order /* order */,
38  const std::vector<Number> & /* elem_soln */,
39  std::vector<Number> & nodal_soln,
40  const bool /*add_p_level*/)
41 {
42  const unsigned int n_nodes = elem->n_nodes();
43 
44  std::fill(nodal_soln.begin(), nodal_soln.end(), 0);
45  nodal_soln.resize(n_nodes, 0);

47  // We request nodal solutions when plotting, for consistency with
48  // other elements; plotting 0 on non-sides makes sense in that
49  // context.
50  // libmesh_warning("Nodal solution requested for a side element; this makes no sense.");
51 } // side_hierarchic_nodal_soln()

54 void side_hierarchic_side_nodal_soln
55  (const Elem * elem, const Order o,
56  const unsigned int side,
57  const std::vector<Number> & elem_soln,
58  std::vector<Number> & nodal_soln_on_side,
59  const bool /*add_p_level*/)
60 {
61  // Cheat here for now: perturb vertices toward the side center so as
62  // to make the values well-defined.
63  const std::vector<unsigned int> side_nodes =
64  elem->nodes_on_side(side);
65  const std::size_t n_side_nodes = side_nodes.size();
66 
67  const FEType fe_type{o, SIDE_HIERARCHIC};
68 
69  Point side_center;
70  for (auto n : side_nodes)
71  side_center += elem->master_point(n);
72  side_center /= n_side_nodes;
73 
74  nodal_soln_on_side.resize(n_side_nodes);
75  for (auto i : make_range(n_side_nodes))
76  {
77  const auto n = side_nodes[i];
78  Point master_p = elem->master_point(n);
79  master_p += TOLERANCE*TOLERANCE*(side_center-master_p);
80 
81  const unsigned int n_sf =
82  FEInterface::n_shape_functions(fe_type, elem);
83 
84  nodal_soln_on_side[i] = 0;
85  for (auto j : make_range(n_sf))
86  nodal_soln_on_side[i] += elem_soln[j] *
87  FEInterface::shape(fe_type, elem, j, master_p);
88  }
89 }

92 unsigned int side_hierarchic_n_dofs_at_node(const ElemType t,
93  const Order o,
94  const unsigned int n)
95 {
96  switch (t)
97  {
98  case EDGE2:
99  case EDGE3:
100  case EDGE4:
101  if (n < 2)
102  return 1; // One per side
103  else
104  return 0;
105  case QUAD8:
106  case QUADSHELL8:
107  case QUAD9:
108  if (n > 3 && n < 8)
109  return o+1;
110  else
111  return 0;
112  case HEX27:
113  if (n > 19 && n < 26)
114  return (o+1)*(o+1); // (o+1)^2 per side
115  else
116  return 0;
117  case TRI6:
118  case TRI7:
119  if (n > 2 && n < 6)
120  return o+1;
121  else
122  return 0;
123  case TET14:
124  if (n > 9)
125  return (o+1)*(o+2)/2;
126  else
127  return 0;
128  case PRISM20:
129  case PRISM21:
130  if (n > 19)
131  return 0;
132  if (n > 17)
133  return (o+1)*(o+2)/2;
134  if (n > 14)
135  return (o+1)*(o+1);
136  return 0;
137  case INVALID_ELEM:
138  return 0;
139  // Without side nodes on all sides we can't support side elements
140  default:
141  libmesh_error_msg("ERROR: Invalid ElemType " << Utility::enum_to_string(t) << " selected for SIDE_HIERARCHIC FE family!");
142  }
143 } // side_hierarchic_n_dofs()

147 unsigned int side_hierarchic_n_dofs(const ElemType t, const Order o)
148 {
149  switch (t)
150  {
151  case EDGE2:
152  case EDGE3:
153  case EDGE4:
154  return 2; // One per side
155  case QUAD8:
156  case QUADSHELL8:
157  case QUAD9:
158  return ((o+1)*4); // o+1 per side
159  case HEX27:
160  return ((o+1)*(o+1)*6); // (o+1)^2 per side
161  case TRI6:
162  case TRI7:
163  return ((o+1)*3); // o+1 per side
164  case TET14:
165  return (o+1)*(o+2)*2; // 4 sides, each (o+1)(o+2)/2
166  case PRISM20:
167  case PRISM21:
168  return (o+1)*(o+1)*3+(o+1)*(o+2); // 2 tris, (o+1)(o+2)/2; 3 quads
169  case INVALID_ELEM:
170  return 0;
171  // Without side nodes on all sides we can't support side elements
172  default:
173  libmesh_error_msg("ERROR: Invalid ElemType " << Utility::enum_to_string(t) << " selected for HIERARCHIC FE family!");
174  }
175 } // side_hierarchic_n_dofs()
