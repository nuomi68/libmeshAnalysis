27 void QSimpson::init_3D(const ElemType, unsigned int)
28 {
29 #if LIBMESH_DIM == 3
30 
31  //-----------------------------------------------------------------------
32  // 3D quadrature rules
33  switch (_type)
34  {
35  //---------------------------------------------
36  // Hex quadrature rules
37  case HEX8:
38  case HEX20:
39  case HEX27:
40  {
41  // We compute the 3D quadrature rule as a tensor
42  // product of the 1D quadrature rule.
43  QSimpson q1D(1);
44  q1D.init(EDGE2);
45 
46  tensor_product_hex( q1D );
47 
48  return;
49  }
50 
51 
52 
53  //---------------------------------------------
54  // Tetrahedral quadrature rules
55  case TET4:
56  case TET10:
57  case TET14:
58  {
59  // This rule is created by combining 8 subtets
60  // which use the trapezoidal rule. The weights
61  // may seem a bit odd, but they are correct,
62  // and should add up to 1/6, the volume of the
63  // reference tet. The points of this rule are
64  // at the nodal points of the TET10, allowing
65  // you to generate diagonal element stiffness
66  // matrices when using quadratic elements.
67  // It should be able to integrate something
68  // better than linears, but I'm not sure how
69  // high.
70 
71  _points.resize(10);
72  _weights.resize(10);
73 
74  _points[0](0) = 0.; _points[5](0) = .5;
75  _points[0](1) = 0.; _points[5](1) = .5;
76  _points[0](2) = 0.; _points[5](2) = 0.;
77 
78  _points[1](0) = 1.; _points[6](0) = 0.;
79  _points[1](1) = 0.; _points[6](1) = .5;
80  _points[1](2) = 0.; _points[6](2) = 0.;
81 
82  _points[2](0) = 0.; _points[7](0) = 0.;
83  _points[2](1) = 1.; _points[7](1) = 0.;
84  _points[2](2) = 0.; _points[7](2) = .5;
85 
86  _points[3](0) = 0.; _points[8](0) = .5;
87  _points[3](1) = 0.; _points[8](1) = 0.;
88  _points[3](2) = 1.; _points[8](2) = .5;
89 
90  _points[4](0) = .5; _points[9](0) = 0.;
91  _points[4](1) = 0.; _points[9](1) = .5;
92  _points[4](2) = 0.; _points[9](2) = .5;
93 
94 
95  _weights[0] = Real(1)/192;
96  _weights[1] = _weights[0];
97  _weights[2] = _weights[0];
98  _weights[3] = _weights[0];
99 
100  _weights[4] = Real(14)/576;
101  _weights[5] = _weights[4];
102  _weights[6] = _weights[4];
103  _weights[7] = _weights[4];
104  _weights[8] = _weights[4];
105  _weights[9] = _weights[4];
106 
107  return;
108  }
109 
110 
111 
112  //---------------------------------------------
113  // Prism quadrature rules
114  case PRISM6:
115  case PRISM15:
116  case PRISM18:
117  case PRISM20:
118  case PRISM21:
119  {
120  // We compute the 3D quadrature rule as a tensor
121  // product of the 1D quadrature rule and a 2D
122  // triangle quadrature rule
123 
124  QSimpson q1D(1);
125  QSimpson q2D(2);
126 
127  // Initialize
128  q1D.init(EDGE2);
129  q2D.init(TRI3);
130 
131  tensor_product_prism(q1D, q2D);
132 
133  return;
134  }
135 
136 
137  //---------------------------------------------
138  // Pyramid quadrature rules
139  case PYRAMID5:
140  case PYRAMID13:
141  case PYRAMID14:
142  case PYRAMID18:
143  {
144  libmesh_error_msg_if(!allow_nodal_pyramid_quadrature,
145  "Nodal quadrature on Pyramid elements is not allowed by default since\n"
146  "the Jacobian of the inverse element map is not well-defined at the Pyramid apex.\n"
147  "Set the QBase::allow_nodal_pyramid_quadrature flag to true to ignore skip this check.");
148 
149  _points.resize(14);
150  _weights.resize(14);
151 
152  _points[0](0) = -1.;
153  _points[0](1) = -1.;
154  _points[0](2) = 0.;
155 
156  _points[1](0) = 1.;
157  _points[1](1) = -1.;
158  _points[1](2) = 0.;
159 
160  _points[2](0) = 1.;
161  _points[2](1) = 1.;
162  _points[2](2) = 0.;
163

165  _points[3](1) = 1.;
166  _points[3](2) = 0.;
167 
168  _points[4](0) = 0.;
169  _points[4](1) = 0.;
170  _points[4](2) = 1.;
171 
172  _points[5](0) = 0.;
173  _points[5](1) = -1.;
174  _points[5](2) = 0.;
175 
176  _points[6](0) = 1.;
177  _points[6](1) = 0.;
178  _points[6](2) = 0.;
179 
180  _points[7](0) = 0.;
181  _points[7](1) = 1.;
182  _points[7](2) = 0.;
183 
184  _points[8](0) = -1.;
185  _points[8](1) = 0.;
186  _points[8](2) = 0.;
187 
188  _points[9](0) = 0.;
189  _points[9](1) = -0.5;
190  _points[9](2) = 0.5;
191 
192  _points[10](0) = 0.5;
193  _points[10](1) = 0.;
194  _points[10](2) = 0.5;
195 
196  _points[11](0) = 0.;
197  _points[11](1) = 0.5;
198  _points[11](2) = 0.5;
199 
200  _points[12](0) = -0.5;
201  _points[12](1) = 0.;
202  _points[12](2) = 0.5;
203 
204  _points[13](0) = 0.;
205  _points[13](1) = 0.;
206  _points[13](2) = 0.;
207 
208  // These are of dubious value since we can't integrate on the
209  // vertex where the mapping Jacobian is ill-defined, and even
210  // if we could it looks like we'd need negative weight at that
211  // vertex to give us exact integrals of both z and z^2. So I
212  // punt and just use QTrap weights.
213  _weights[0] = 1/Real(4);
214  _weights[1] = _weights[0];
215  _weights[2] = _weights[0];
216  _weights[3] = _weights[0];
217  _weights[4] = 1/Real(3);
218  _weights[5] = 0;
219  _weights[6] = 0;
220  _weights[7] = 0;
221  _weights[8] = 0;
222  _weights[9] = 0;
223  _weights[10] = 0;
224  _weights[11] = 0;
225  _weights[12] = 0;
226  _weights[13] = 0;
227 
228  return;
229  }
