343  template <class DataMap>
344  Real get_max_abs_value(const DataMap & v) const
345  {
346  Real max_value = 0.;
347 
348  for (const auto & pr : v)
349  {
350  const auto & v_comp_and_qp = pr.second;
351 
352  for (const auto & comp : index_range(v_comp_and_qp))
353  {
354  // If scale_components_in_enrichment() returns true then we
355  // apply a scaling to give an approximately uniform scaling
356  // for all components.
357  Real comp_scaling = 1.;
358  if (get_rb_eim_evaluation().scale_components_in_enrichment())
359  {
360  // Make sure that _component_scaling_in_training_set is initialized
361  libmesh_error_msg_if(comp >= _component_scaling_in_training_set.size(),
362  "Invalid vector index");
363  comp_scaling = _component_scaling_in_training_set[comp];
364  }
365 
366  const std::vector<Number> & v_qp = v_comp_and_qp[comp];
367  for (Number value : v_qp)
368  max_value = std::max(max_value, std::abs(value * comp_scaling));
369  }
370  }
371 
372  comm().max(max_value);
373  return max_value;
374  }
