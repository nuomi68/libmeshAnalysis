112 void dofobjectkey_max_op (libMesh::Parallel::DofObjectKey * in,
113  libMesh::Parallel::DofObjectKey * inout,
114  int * len, void *)
115 {
116  // When (*in <= *inout), then inout already contains max(*in,*inout)
117  // Otherwise we need to copy from in.
118  for (int i=0; i<*len; i++, in++, inout++)
119  if (*inout < *in)
120  *inout = *in;
121 }
