222 bool InfPrism::contains_point (const Point & p, Real tol) const
223 {
224  // For infinite elements with linear base interpolation:
225  // make use of the fact that infinite elements do not
226  // live inside the envelope. Use a fast scheme to
227  // check whether point \p p is inside or outside
228  // our relevant part of the envelope. Note that
229  // this is not exclusive: only when the distance is less,
230  // we are safe. Otherwise, we cannot say anything. The
231  // envelope may be non-spherical, the physical point may lie
232  // inside the envelope, outside the envelope, or even inside
233  // this infinite element. Therefore if this fails,
234  // fall back to the inverse_map()
235  const Point my_origin (this->origin());
236 
237  // determine the minimal distance of the base from the origin
238  // use norm_sq(), it is faster than norm() and produces
239  // the same behavior. Shift my_origin to center first:
240  Point pt0_o(this->point(0) - my_origin);
241  Point pt1_o(this->point(1) - my_origin);
242  Point pt2_o(this->point(2) - my_origin);
243  const Real tmp_min_distance_sq = std::min(pt0_o.norm_sq(),
244  std::min(pt1_o.norm_sq(),
245  pt2_o.norm_sq()));
246 
247  // For a coarse grid, it is important to account for the fact
248  // that the sides are not spherical, thus the closest point
249  // can be closer than all edges.
250  // This is an estimator using Pythagoras:
251  const Real min_distance_sq = tmp_min_distance_sq
252  - .5*std::max((point(0)-point(1)).norm_sq(),
253  std::max((point(0)-point(2)).norm_sq(),
254  (point(1)-point(2)).norm_sq()));
255 
256  // work with 1% allowable deviation. We can still fall
257  // back to the InfFE::inverse_map()
258  const Real conservative_p_dist_sq = 1.01 * (Point(p - my_origin).norm_sq());
259 
260  if (conservative_p_dist_sq < min_distance_sq)
261  {
262  // the physical point is definitely not contained in the element:
263  return false;
264  }
265 
266  // this captures the case that the point is not (almost) in the direction of the element.:
267  // first, project the problem onto the unit sphere:
268  Point p_o(p - my_origin);
269  pt0_o /= pt0_o.norm();
270  pt1_o /= pt1_o.norm();
271  pt2_o /= pt2_o.norm();
272  p_o /= p_o.norm();
273 
274  // now, check if it is in the projected face; by comparing the distance of
275  // any point in the element to \p p with the largest distance between this point
276  // to any other point in the element.
277  if ((p_o - pt0_o).norm_sq() > std::max((pt0_o - pt1_o).norm_sq(), (pt0_o - pt2_o).norm_sq()) ||
278  (p_o - pt1_o).norm_sq() > std::max((pt1_o - pt2_o).norm_sq(), (pt1_o - pt0_o).norm_sq()) ||
279  (p_o - pt2_o).norm_sq() > std::max((pt2_o - pt0_o).norm_sq(), (pt2_o - pt1_o).norm_sq()) )
280  {
281  // the physical point is definitely not contained in the element
282  return false;
283  }
284 
285 
286  // It seems that \p p is at least close to this element.
287  //
288  // Declare a basic FEType. Will use default in the base,
289  // and something else (not important) in radial direction.
290  FEType fe_type(default_order());
291 
292  const Point mapped_point = InfFEMap::inverse_map(dim(), this, p,
293  tol, false);
294 
295  return FEInterface::on_reference_element(mapped_point, this->type(), tol);
296 }
