39 // OpFunction<> specializations to return an MPI_Op version of the
40 // reduction operations on FETypes
41 //
42 // We use static variables to minimize the number of MPI datatype
43 // construction calls executed over the course of the program.
44 //
45 // We use a singleton pattern because a global variable would
46 // have tried to call MPI functions before MPI got initialized.
47 //
48 // Doing this to the FEFamily part of a type doesn't make much sense,
49 // *except* that it's how we verify that a requested family matches
50 // across multiple processors.
51 
52 template <>
53 class OpFunction<libMesh::FEType>
54 {
55 public:
56 #ifdef LIBMESH_HAVE_MPI
57  static void fetype_max (void * intype, void * inouttype, int * len, MPI_Datatype *)
58  {
59  libMesh::FEType *in = static_cast<libMesh::FEType *>(intype);
60  libMesh::FEType *inout = static_cast<libMesh::FEType *>(inouttype);
61  for (int i=0; i != *len; ++i)
62  {
63  inout[i].family = std::max(in[i].family, inout[i].family);
64  inout[i].order = std::max(in[i].order, inout[i].order);
65  }
66  }

73  {
74  inout[i].family = std::min(in[i].family, inout[i].family);

84  static MPI_Op max()
85  {
86  // _static_op never gets freed, but it only gets committed once,
87  // so it's not a *huge* memory leak...
88  static MPI_Op _static_op;

142 // StandardType<> specializations to return a derived MPI datatype
143 // to handle communication of FETypes.
144 template <>
145 class StandardType<libMesh::FEType> : public DataType
146 {
147 public:
148  explicit
149  StandardType(const libMesh::FEType * example=nullptr)
150  {
151 #ifdef LIBMESH_HAVE_MPI
152  using libMesh::FEType;
153 
154  // We need an example for MPI_Address to use
155  FEType * ex;
156  std::unique_ptr<FEType> temp;
157  if (example)
158  ex = const_cast<FEType *>(example);
159  else
160  {
161  temp = std::make_unique<FEType>();
162  ex = temp.get();
163  }
164 
165 #ifdef LIBMESH_ENABLE_INFINITE_ELEMENTS
166  constexpr std::size_t structsize = 5;
167 #else
168  constexpr std::size_t structsize = 2;
169 #endif
170 
171  // Our FEType enums pack into int
172  MPI_Datatype int_types[structsize];
173 
174  // We require MPI-2 here:
175  int blocklengths[structsize];
176  MPI_Aint start, displs[structsize];
177  MPI_Datatype tmptype;
178 
179  timpi_call_mpi
180  (MPI_Get_address (ex, &start));
181 #ifdef LIBMESH_ENABLE_INFINITE_ELEMENTS
182  timpi_call_mpi
183  (MPI_Get_address (&(ex->order), &displs[0]));
184  timpi_call_mpi
185  (MPI_Get_address (&(ex->radial_order), &displs[1]));
186  timpi_call_mpi
187  (MPI_Get_address (&(ex->family), &displs[2]));
188  timpi_call_mpi
189  (MPI_Get_address (&(ex->radial_family), &displs[3]));
190  timpi_call_mpi
191  (MPI_Get_address (&(ex->inf_map), &displs[4]));
192 #else // ifdef LIBMESH_ENABLE_INFINITE_ELEMENTS
193  timpi_call_mpi
194  (MPI_Get_address (&(ex->order), &displs[0]));
195  timpi_call_mpi
196  (MPI_Get_address (&(ex->family), &displs[1]));
197 #endif // ifdef LIBMESH_ENABLE_INFINITE_ELEMENTS
198 
199  // subtract off beginning of the structure to get offsets. We'll
200  // be creating a struct type later to account for any weird
201  // padding issues.
202  for (std::size_t i = 0; i != structsize; ++i)
203  {
204  displs[i] -= start;
205  int_types[i] = MPI_INT;
206  blocklengths[i] = 1;
207  }
208 
209  // create a prototype structure
210  timpi_call_mpi
211  (MPI_Type_create_struct (structsize, blocklengths, displs,
212  int_types, &tmptype));
213  timpi_call_mpi
214  (MPI_Type_commit (&tmptype));
215 
216  // resize the structure type to account for padding, if any
217  timpi_call_mpi
218  (MPI_Type_create_resized (tmptype, 0, sizeof(FEType),
219  &_datatype));
220 
221  timpi_call_mpi
222  (MPI_Type_commit (&_datatype));
223 
224  timpi_call_mpi
225  (MPI_Type_free (&tmptype));
226 #else
227  libmesh_ignore(example);
228 #endif // #ifdef LIBMESH_HAVE_MPI
229  }
230 
231  StandardType(const StandardType<libMesh::FEType> & timpi_mpi_var(t))
232  : DataType()
233  {
234  timpi_call_mpi (MPI_Type_dup (t._datatype, &_datatype));
235  }
236 
237  ~StandardType() { this->free(); }
238 
239  static const bool is_fixed_type = true;
240 };
