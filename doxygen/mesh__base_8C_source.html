90  boundary_info (new BoundaryInfo(*this)), // BoundaryInfo has protected ctor, can't use std::make_unique
91  _n_parts (other_mesh._n_parts),
92  _default_mapping_type(other_mesh._default_mapping_type),
93  _default_mapping_data(other_mesh._default_mapping_data),
94  _is_prepared (other_mesh._is_prepared),
95  _point_locator (),
96  _count_lower_dim_elems_in_point_locator(other_mesh._count_lower_dim_elems_in_point_locator),
97  _partitioner (),
98 #ifdef LIBMESH_ENABLE_UNIQUE_ID
99  _next_unique_id(other_mesh._next_unique_id),
100 #endif
101  _skip_noncritical_partitioning(false),
102  _skip_all_partitioning(libMesh::on_command_line("--skip-partitioning")),
103  _skip_renumber_nodes_and_elements(other_mesh._skip_renumber_nodes_and_elements),
104  _skip_find_neighbors(other_mesh._skip_find_neighbors),
105  _allow_remote_element_removal(other_mesh._allow_remote_element_removal),
106  _elem_dims(other_mesh._elem_dims),
107  _elemset_codes_inverse_map(other_mesh._elemset_codes_inverse_map),
108  _all_elemset_ids(other_mesh._all_elemset_ids),
109  _spatial_dimension(other_mesh._spatial_dimension),
110  _default_ghosting(std::make_unique<GhostPointNeighbors>(*this)),
111  _point_locator_close_to_point_tol(other_mesh._point_locator_close_to_point_tol)
112 {
113  const GhostingFunctor * const other_default_ghosting = other_mesh._default_ghosting.get();
114 
115  for (GhostingFunctor * const gf : other_mesh._ghosting_functors)
116  {
117  // If the other mesh is using default ghosting, then we will use our own
118  // default ghosting
119  if (gf == other_default_ghosting)
120  {
121  _ghosting_functors.insert(_default_ghosting.get());
122  continue;
123  }
124 
125  std::shared_ptr<GhostingFunctor> clone_gf = gf->clone();
126  // Some subclasses of GhostingFunctor might not override the
127  // clone function yet. If this is the case, GhostingFunctor will
128  // return nullptr by default. The clone function should be overridden
129  // in all derived classes. This following code ("else") is written
130  // for API upgrade. That will allow users gradually to update their code.
131  // Once the API upgrade is done, we will come back and delete "else."
132  if (clone_gf)
133  {
134  clone_gf->set_mesh(this);
135  add_ghosting_functor(clone_gf);
136  }
137  else
138  {
139  libmesh_deprecated();
140  add_ghosting_functor(*gf);
141  }
142  }
143 
144  if (other_mesh._partitioner.get())
145  _partitioner = other_mesh._partitioner->clone();
146 
147  // _elemset_codes stores pointers to entries in _elemset_codes_inverse_map,
148  // so it is not possible to simply copy it directly from other_mesh
149  for (const auto & [set, code] : _elemset_codes_inverse_map)
150  _elemset_codes.emplace(code, &set);
151 }

220 bool MeshBase::locally_equals (const MeshBase & other_mesh) const
221 {
222  // Check whether everything in the base is equal
223  if (_n_parts != other_mesh._n_parts ||
224  _default_mapping_type != other_mesh._default_mapping_type ||
225  _default_mapping_data != other_mesh._default_mapping_data ||
226  _is_prepared != other_mesh._is_prepared ||
227  _count_lower_dim_elems_in_point_locator !=
228  other_mesh._count_lower_dim_elems_in_point_locator ||
229  // We expect this to change in a DistributeMesh prepare_for_use();
230  // it's conceptually "mutable"...
231 /*
232 #ifdef LIBMESH_ENABLE_UNIQUE_ID
233  _next_unique_id != other_mesh._next_unique_id ||
234 #endif
235 */
236  _skip_noncritical_partitioning != other_mesh._skip_noncritical_partitioning ||
237  _skip_all_partitioning != other_mesh._skip_all_partitioning ||
238  _skip_renumber_nodes_and_elements != other_mesh._skip_renumber_nodes_and_elements ||
239  _skip_find_neighbors != other_mesh._skip_find_neighbors ||
240  _allow_remote_element_removal != other_mesh._allow_remote_element_removal ||
241  _spatial_dimension != other_mesh._spatial_dimension ||
242  _point_locator_close_to_point_tol != other_mesh._point_locator_close_to_point_tol ||
243  _block_id_to_name != other_mesh._block_id_to_name ||
244  _elem_dims != other_mesh._elem_dims ||
245  _mesh_subdomains != other_mesh._mesh_subdomains ||
246  _all_elemset_ids != other_mesh._all_elemset_ids ||
247  _elem_integer_names != other_mesh._elem_integer_names ||
248  _elem_integer_default_values != other_mesh._elem_integer_default_values ||
249  _node_integer_names != other_mesh._node_integer_names ||
250  _node_integer_default_values != other_mesh._node_integer_default_values ||
251  bool(_default_ghosting) != bool(other_mesh._default_ghosting) ||
252  bool(_partitioner) != bool(other_mesh._partitioner) ||
253  *boundary_info != *other_mesh.boundary_info)
254  return false;
255 
256  const constraint_rows_type & other_rows =
257  other_mesh.get_constraint_rows();
258  for (const auto & [node, row] : this->_constraint_rows)
259  {
260  const dof_id_type node_id = node->id();
261  const Node * other_node = other_mesh.query_node_ptr(node_id);
262  if (!other_node)
263  return false;
264 
265  auto it = other_rows.find(other_node);
266  if (it == other_rows.end())
267  return false;
268 
269  const auto & other_row = it->second;
270  if (row.size() != other_row.size())
271  return false;
272 
273  for (auto i : index_range(row))
274  {
275  const auto & [elem_pair, coef] = row[i];
276  const auto & [other_elem_pair, other_coef] = other_row[i];
277  libmesh_assert(elem_pair.first);
278  libmesh_assert(other_elem_pair.first);
279  if (elem_pair.first->id() !=
280  other_elem_pair.first->id() ||
281  elem_pair.second !=
282  other_elem_pair.second ||
283  coef != other_coef)
284  return false;
285  }
286  }
287 
288  for (const auto & [elemset_code, elemset_ptr] : this->_elemset_codes)
289  {
290  auto it = other_mesh._elemset_codes.find(elemset_code);
291  if (it == other_mesh._elemset_codes.end() ||
292  *elemset_ptr != *it->second)
293  return false;
294  }
295 
296  // FIXME: we have no good way to compare ghosting functors, since
297  // they're in a set sorted by pointer, and we have no way *at all*
298  // to compare ghosting functors, since they don't have operator==
299  // defined and we encourage users to subclass them. We can check if
300  // we have the same number, is all.
301  if (_ghosting_functors.size() !=
302  other_mesh._ghosting_functors.size())
303  return false;
304 
305  // Same deal for partitioners. We tested that we both have one or
306  // both don't, but are they equivalent? Let's guess "yes".
307 
308  // Now let the subclasses decide whether everything else is equal
309  return this->subclass_locally_equals(other_mesh);
310 }

348 void MeshBase::add_elemset_code(dof_id_type code, MeshBase::elemset_type id_set)
349 {
350  // Populate inverse map, stealing id_set's resources
351  auto [it1, inserted1] = _elemset_codes_inverse_map.emplace(std::move(id_set), code);
352 
353  // Reference to the newly inserted (or previously existing) id_set
354  const auto & inserted_id_set = it1->first;
355 
356  // Keep track of all elemset ids ever added for O(1) n_elemsets()
357  // performance. Only need to do this if we didn't know about this
358  // id_set before...
359  if (inserted1)
360  _all_elemset_ids.insert(inserted_id_set.begin(), inserted_id_set.end());
361 
362  // Take the address of the newly emplaced set to use in
363  // _elemset_codes, avoid duplicating std::set storage
364  auto [it2, inserted2] = _elemset_codes.emplace(code, &inserted_id_set);
365 
366  // Throw an error if this code already exists with a pointer to a
367  // different set of ids.
368  libmesh_error_msg_if(!inserted2 && it2->second != &inserted_id_set,
369  "The elemset code " << code << " already exists with a different id_set.");
370 }

404 void MeshBase::change_elemset_code(dof_id_type old_code, dof_id_type new_code)
405 {
406  // Look up elemset ids for old_code
407  auto it = _elemset_codes.find(old_code);
408 
409  // If we don't have the old_code, then do nothing. Alternatively, we
410  // could throw an error since trying to change an elemset code you
411  // don't have could indicate there's a problem...
412  if (it == _elemset_codes.end())
413  return;
414 
415  // Make copy of the set of elemset ids. We are not changing these,
416  // only updating the elemset code it corresponds to.
417  elemset_type id_set_copy = *(it->second);
418 
419  // Look up the corresponding entry in the inverse map. Note: we want
420  // the iterator because we are going to remove it.
421  auto inverse_it = _elemset_codes_inverse_map.find(id_set_copy);
422  libmesh_error_msg_if(inverse_it == _elemset_codes_inverse_map.end(),
423  "Expected _elemset_codes_inverse_map entry for elemset code " << old_code);
424 
425  // Erase entry from inverse map
426  _elemset_codes_inverse_map.erase(inverse_it);
427 
428  // Erase entry from forward map
429  _elemset_codes.erase(it);
430 
431  // Add new code with original set of ids.
432  this->add_elemset_code(new_code, id_set_copy);
433 
434  // We can't update any actual elemset codes if there is no extra integer defined for it.
435  if (!this->has_elem_integer("elemset_code"))
436  return;
437 
438  // Get index of elemset_code extra integer
439  unsigned int elemset_index = this->get_elem_integer_index("elemset_code");
440 
441  // Loop over all elems and update code
442  for (auto & elem : this->element_ptr_range())
443  {
444  dof_id_type elemset_code =
445  elem->get_extra_integer(elemset_index);
446 
447  if (elemset_code == old_code)
448  elem->set_extra_integer(elemset_index, new_code);
449  }
450 }

452 void MeshBase::change_elemset_id(elemset_id_type old_id, elemset_id_type new_id)
453 {
454  // Early return if we don't have old_id
455  if (!_all_elemset_ids.count(old_id))
456  return;
457 
458  // Throw an error if the new_id is already used
459  libmesh_error_msg_if(_all_elemset_ids.count(new_id),
460  "Cannot change elemset id " << old_id <<
461  " to " << new_id << ", " << new_id << " already exists.");
462 
463  // We will build up a new version of the inverse map so we can iterate over
464  // the current one without invalidating anything.
465  std::map<MeshBase::elemset_type, dof_id_type> new_elemset_codes_inverse_map;
466  for (const auto & [id_set, elemset_code] : _elemset_codes_inverse_map)
467  {
468  auto id_set_copy = id_set;
469  if (id_set_copy.count(old_id))
470  {
471  // Remove old_id, insert new_id
472  id_set_copy.erase(old_id);
473  id_set_copy.insert(new_id);
474  }
475 
476  // Store in new version of map
477  new_elemset_codes_inverse_map.emplace(id_set_copy, elemset_code);
478  }
479 
480  // Swap existing map with newly-built one
481  _elemset_codes_inverse_map.swap(new_elemset_codes_inverse_map);
482 
483  // Reconstruct _elemset_codes map
484  _elemset_codes.clear();
485  for (const auto & [id_set, elemset_code] : _elemset_codes_inverse_map)
486  _elemset_codes.emplace(elemset_code, &id_set);
487 
488  // Update _all_elemset_ids
489  _all_elemset_ids.erase(old_id);
490  _all_elemset_ids.insert(new_id);
491 }

533 std::vector<unsigned int> MeshBase::add_elem_integers(const std::vector<std::string> & names,
534  bool allocate_data,
535  const std::vector<dof_id_type> * default_values)
536 {
537  libmesh_assert(!default_values || default_values->size() == names.size());
538  libmesh_assert_equal_to(_elem_integer_names.size(), _elem_integer_default_values.size());
539 
540  std::unordered_map<std::string, std::size_t> name_indices;
541  for (auto i : index_range(_elem_integer_names))
542  name_indices[_elem_integer_names[i]] = i;
543 
544  std::vector<unsigned int> returnval(names.size());
545 
546  bool added_an_integer = false;
547  for (auto i : index_range(names))
548  {
549  const std::string & name = names[i];
550  auto it = name_indices.find(name);
551  if (it != name_indices.end())
552  {
553  returnval[i] = it->second;
554  _elem_integer_default_values[it->second] =
555  default_values ? (*default_values)[i] : DofObject::invalid_id;
556  }

740 void MeshBase::prepare_for_use ()
741 {
742  LOG_SCOPE("prepare_for_use()", "MeshBase");
743 
744  parallel_object_only();
745 
746  libmesh_assert(this->comm().verify(this->is_serial()));
747 
748  // A distributed mesh may have processors with no elements (or
749  // processors with no elements of higher dimension, if we ever
750  // support mixed-dimension meshes), but we want consistent
751  // mesh_dimension anyways.
752  //
753  // cache_elem_data() should get the elem_dimensions() and
754  // mesh_dimension() correct later, and we don't need it earlier.
755 
756 
757  // Renumber the nodes and elements so that they in contiguous
758  // blocks. By default, _skip_renumber_nodes_and_elements is false.
759  //
760  // Instances where you if prepare_for_use() should not renumber the nodes
761  // and elements include reading in e.g. an xda/r or gmv file. In
762  // this case, the ordering of the nodes may depend on an accompanying
763  // solution, and the node ordering cannot be changed.
764 
765 
766  // Mesh modification operations might not leave us with consistent
767  // id counts, or might leave us with orphaned nodes we're no longer
768  // using, but our partitioner might need that consistency and/or
769  // might be confused by orphaned nodes.
770  if (!_skip_renumber_nodes_and_elements)
771  this->renumber_nodes_and_elements();
772  else
773  {
774  this->remove_orphaned_nodes();
775  this->update_parallel_id_counts();
776  }
777 
778  // Let all the elements find their neighbors
779  if (!_skip_find_neighbors)
780  this->find_neighbors();
781 
782  // The user may have set boundary conditions. We require that the
783  // boundary conditions were set consistently. Because we examine
784  // neighbors when evaluating non-raw boundary condition IDs, this
785  // assert is only valid when our neighbor links are in place.
786 #ifdef DEBUG
787  MeshTools::libmesh_assert_valid_boundary_ids(*this);
788 #endif
789 
790  // Search the mesh for all the dimensions of the elements
791  // and cache them.
792  this->cache_elem_data();
793 
794  // Search the mesh for elements that have a neighboring element
795  // of dim+1 and set that element as the interior parent
796  this->detect_interior_parents();
797 
798  // Fix up node unique ids in case mesh generation code didn't take
799  // exceptional care to do so.
800  // MeshCommunication().make_node_unique_ids_parallel_consistent(*this);
801 
802  // We're going to still require that mesh generation code gets
803  // element unique ids consistent.
804 #if defined(DEBUG) && defined(LIBMESH_ENABLE_UNIQUE_ID)
805  MeshTools::libmesh_assert_valid_unique_ids(*this);
806 #endif
807 
808  // Reset our PointLocator. Any old locator is invalidated any time
809  // the elements in the underlying elements in the mesh have changed,
810  // so we clear it here.
811  this->clear_point_locator();
812 
813  // Allow our GhostingFunctor objects to reinit if necessary.
814  // Do this before partitioning and redistributing, and before
815  // deleting remote elements.
816  this->reinit_ghosting_functors();
817 
818  // Partition the mesh unless *all* partitioning is to be skipped.
819  // If only noncritical partitioning is to be skipped, the
820  // partition() call will still check for orphaned nodes.
821  if (!skip_partitioning())
822  this->partition();
823 
824  // If we're using DistributedMesh, we'll probably want it
825  // parallelized.
826  if (this->_allow_remote_element_removal)
827  this->delete_remote_elements();
828 
829  // Much of our boundary info may have been for now-remote parts of the mesh,
830  // in which case we don't want to keep local copies of data meant to be
831  // local. On the other hand we may have deleted, or the user may have added in
832  // a distributed fashion, boundary data that is meant to be global. So we
833  // handle both of those scenarios here
834  this->get_boundary_info().regenerate_id_sets();
835 
836  if (!_skip_renumber_nodes_and_elements)
837  this->renumber_nodes_and_elements();
838 
839  // The mesh is now prepared for use.
840  _is_prepared = true;
841 
842 #ifdef DEBUG
843  MeshTools::libmesh_assert_valid_boundary_ids(*this);
844 #ifdef LIBMESH_ENABLE_UNIQUE_ID
845  MeshTools::libmesh_assert_valid_unique_ids(*this);
846 #endif
847 #endif
848 }

1722 void MeshBase::detect_interior_parents()
1723 {
1724  // This requires an inspection on every processor
1725  parallel_object_only();
1726 
1727  // Check if the mesh contains mixed dimensions. If so, then set interior parents, otherwise return.
1728  if (this->elem_dimensions().size() == 1)
1729  return;
1730 
1731  //This map will be used to set interior parents
1732  std::unordered_map<dof_id_type, std::vector<dof_id_type>> node_to_elem;
1733 
1734  for (const auto & elem : this->active_element_ptr_range())
1735  {
1736  // Populating the node_to_elem map, same as MeshTools::build_nodes_to_elem_map
1737  for (auto n : make_range(elem->n_vertices()))
1738  {
1739  libmesh_assert_less (elem->id(), this->max_elem_id());
1740 
1741  node_to_elem[elem->node_id(n)].push_back(elem->id());
1742  }
1743  }
1744 
1745  // Automatically set interior parents
1746  for (const auto & element : this->element_ptr_range())
1747  {
1748  // Ignore an 3D element or an element that already has an interior parent
1749  if (element->dim()>=LIBMESH_DIM || element->interior_parent())
1750  continue;
1751 
1752  // Start by generating a SET of elements that are dim+1 to the current
1753  // element at each vertex of the current element, thus ignoring interior nodes.
1754  // If one of the SET of elements is empty, then we will not have an interior parent
1755  // since an interior parent must be connected to all vertices of the current element
1756  std::vector<std::set<dof_id_type>> neighbors( element->n_vertices() );
1757 
1758  bool found_interior_parents = false;
1759 
1760  for (auto n : make_range(element->n_vertices()))
1761  {
1762  std::vector<dof_id_type> & element_ids = node_to_elem[element->node_id(n)];
1763  for (const auto & eid : element_ids)
1764  if (this->elem_ref(eid).dim() == element->dim()+1)
1765  neighbors[n].insert(eid);
1766 
1767  if (neighbors[n].size()>0)
1768  {
1769  found_interior_parents = true;
1770  }
1771  else
1772  {
1773  // We have found an empty set, no reason to continue
1774  // Ensure we set this flag to false before the break since it could have
1775  // been set to true for previous vertex
1776  found_interior_parents = false;
1777  break;
1778  }
1779  }
1780 
1781  // If we have successfully generated a set of elements for each vertex, we will compare
1782  // the set for vertex 0 will the sets for the vertices until we find a id that exists in
1783  // all sets. If found, this is our an interior parent id. The interior parent id found
1784  // will be the lowest element id if there is potential for multiple interior parents.
1785  if (found_interior_parents)
1786  {
1787  std::set<dof_id_type> & neighbors_0 = neighbors[0];
1788  for (const auto & interior_parent_id : neighbors_0)
1789  {
1790  found_interior_parents = false;
1791  for (auto n : make_range(1u, element->n_vertices()))
1792  {
1793  if (neighbors[n].find(interior_parent_id)!=neighbors[n].end())
1794  {
1795  found_interior_parents=true;
1796  }
1797  else
1798  {
1799  found_interior_parents=false;
1800  break;
1801  }
1802  }
1803  if (found_interior_parents)
1804  {
1805  element->set_interior_parent(this->elem_ptr(interior_parent_id));
1806  break;
1807  }
1808  }
1809  }
1810  }
1811 }
