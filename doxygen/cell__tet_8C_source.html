128 bool Tet::is_child_on_side_helper(const unsigned int c,
129  const unsigned int s,
130  const unsigned int checked_nodes[][3]) const
131 {
132  libmesh_assert_less (c, this->n_children());
133  libmesh_assert_less (s, this->n_sides());
134 
135  // For the 4 vertices, child c touches vertex c, so we can return
136  // true if that vertex is on side s
137  for (unsigned int i = 0; i != 3; ++i)
138  if (Tet4::side_nodes_map[s][i] == c)
139  return true;
140 
141  // If we are a "vertex child" and we didn't already return true,
142  // we must not be on the side in question
143  if (c < 4)
144  return false;
145 
146  // For the 4 non-vertex children, the child ordering depends on the
147  // diagonal selection. We'll let the embedding matrix figure that
148  // out: if this child has three nodes that don't depend on the
149  // position of the node_facing_side[s], then we're on side s. Which
150  // three nodes those are depends on the subclass, so their responsibility
151  // is to call this function with the proper check_nodes array
152  const unsigned int node_facing_side[4] = {3, 2, 0, 1};
153  const unsigned int n = node_facing_side[s];
154 
155  // Add up the absolute values of the entries of the embedding matrix for the
156  // nodes opposite node n. If it is equal to zero, then the child in question is
157  // on side s, so return true.
158  Real embedding_sum = 0.;
159  for (unsigned i=0; i<3; ++i)
160  embedding_sum += std::abs(this->embedding_matrix(c, checked_nodes[n][i], n));
161 
162  return ( std::abs(embedding_sum) < 1.e-3 );
163 }

180 void Tet::choose_diagonal() const
181 {
182  // Check for uninitialized diagonal selection
183  if (this->_diagonal_selection==INVALID_DIAG)
184  {
185  Real diag_01_23 = (this->point(0) + this->point(1) - this->point(2) - this->point(3)).norm_sq();
186  Real diag_02_13 = (this->point(0) - this->point(1) + this->point(2) - this->point(3)).norm_sq();
187  Real diag_03_12 = (this->point(0) - this->point(1) - this->point(2) + this->point(3)).norm_sq();
188 
189  this->_diagonal_selection=DIAG_02_13;
190 
191  if (diag_01_23 < diag_02_13 || diag_03_12 < diag_02_13)
192  {
193  if (diag_01_23 < diag_03_12)
194  this->_diagonal_selection=DIAG_01_23;
195 
196  else
197  this->_diagonal_selection=DIAG_03_12;
198  }
199  }
200 }

241 std::pair<Real, Real> Tet::qual_bounds (const ElemQuality q) const
242 {
243  std::pair<Real, Real> bounds;
244 
245  switch (q)

249  case ASPECT_RATIO_GAMMA:
250  bounds.first = 1.;
251  bounds.second = 3.;
252  break;
253 
254  case SIZE:
255  case SHAPE:
256  bounds.first = 0.2;
257  bounds.second = 1.;
258  break;
259 
260  case CONDITION:
261  bounds.first = 1.;
262  bounds.second = 3.;
263  break;
264 
265  case DISTORTION:
266  bounds.first = 0.6;
267  bounds.second = 1.;
268  break;
269 
270  case JACOBIAN:
271  bounds.first = 0.5;
272  bounds.second = 1.414;
273  break;
274 
275  default:
276  libMesh::out << "Warning: Invalid quality measure chosen." << std::endl;
277  bounds.first = -1;
278  bounds.second = -1;
279  }
280 
281  return bounds;
