100 std::pair<bool, unsigned short>
101 can_delaunay_swap(const Elem & elem,
102  unsigned short side,
103  Real tol)
104 {
105  const Elem * neigh = elem.neighbor_ptr(side);
106  if (!neigh)
107  return {false, 0};
108 
109  unsigned short nn = 0;
110 
111  // What neighbor node does elem not share?
112  for (; nn < 3; ++nn)
113  {
114  const Node * neigh_node = neigh->node_ptr(nn);
115  if (neigh_node == elem.node_ptr(0) ||
116  neigh_node == elem.node_ptr(1) ||
117  neigh_node == elem.node_ptr(2))
118  continue;
119 
120  // Might we need to do a diagonal swap here? Avoid
121  // undoing a borderline swap.
122  if (in_circumcircle(elem, *neigh_node, tol))
123  break;
124  }
125 
126  if (nn == 3)
127  return {false, 0};
128 
129  const unsigned short n = (side+2)%3;
130  const RealVectorValue right =
131  (elem.point((n+1)%3)-elem.point(n)).unit();
132  const RealVectorValue mid =
133  (neigh->point(nn)-elem.point(n)).unit();
134  const RealVectorValue left =
135  (elem.point((n+2)%3)-elem.point(n)).unit();
136 
137  // If the "middle" vector isn't really in the middle, we can't do a
138  // swap without involving other triangles (or we can't at all if
139  // there's a domain boundary in the way)
140  if (mid*right < left*right ||
141  left*mid < left*right)
142  return {false, 0};
143 
144  return {true, nn};
145 }

181 // Restore a triangulation's Delaunay property, starting with a set of
182 // all triangles that might initially not be locally Delaunay with
183 // their neighbors.
184 template <typename Container>
185 inline
186 void restore_delaunay(Container & check_delaunay_on,
187  BoundaryInfo & boundary_info)
188 {
189  LOG_SCOPE("restore_delaunay()", "Poly2TriTriangulator");
190 
191  while (!check_delaunay_on.empty())
192  {
193  Elem & elem = **check_delaunay_on.begin();
194  check_delaunay_on.erase(&elem);
195  for (auto s : make_range(elem.n_sides()))
196  {
197  // Can we make a swap here? With what neighbor, with what
198  // far node? Use a negative tolerance to avoid swapping
199  // back and forth.
200  auto [can_swap, nn] =
201  can_delaunay_swap(elem, s, -TOLERANCE*TOLERANCE);
202  if (!can_swap)
203  continue;
204 
205  Elem * neigh = elem.neighbor_ptr(s);
206 
207  // If we made it here it's time to diagonal swap
208  const unsigned short n = (s+2)%3;
209 
210  const std::array<Node *,4> nodes {elem.node_ptr(n),
211  elem.node_ptr((n+1)%3), neigh->node_ptr(nn),
212  elem.node_ptr((n+2)%3)};
213 
214  // If we have to swap then either we or any of our neighbors
215  // might no longer be Delaunay
216  for (auto ds : make_range(3))
217  {
218  if (elem.neighbor_ptr(ds))
219  check_delaunay_on.insert(elem.neighbor_ptr(ds));
220  if (neigh->neighbor_ptr(ds))
221  check_delaunay_on.insert(neigh->neighbor_ptr(ds));
222  }
223 
224  // An interior boundary between two newly triangulated
225  // triangles shouldn't have any bcids
226  libmesh_assert(!boundary_info.n_boundary_ids(neigh, (nn+1)%3));
227  libmesh_assert(!boundary_info.n_boundary_ids(&elem, (n+1)%3));
228 
229  // The two changing boundary sides might have bcids
230  std::vector<boundary_id_type> bcids;
231  boundary_info.boundary_ids(&elem, (n+2)%3, bcids);
232  if (!bcids.empty())
233  {
234  boundary_info.remove_side(&elem, (n+2)%3);
235  boundary_info.add_side(neigh, (nn+1)%3, bcids);
236  }
237 
238  boundary_info.boundary_ids(neigh, (nn+2)%3, bcids);
239  if (!bcids.empty())
240  {
241  boundary_info.remove_side(neigh, (nn+2)%3);
242  boundary_info.add_side(&elem, (n+1)%3, bcids);
243  }
244 
245  elem.set_node((n+2)%3) = nodes[2];
246  neigh->set_node((nn+2)%3) = nodes[0];
247 
248  // No need for a temporary array to swap these around, if we
249  // do it in the right order.
250  //
251  // Watch me neigh->neigh
252  Elem * neighneigh = neigh->neighbor_ptr((nn+2)%3);
253  if (neighneigh)
254  {
255  unsigned int snn = neighneigh->which_neighbor_am_i(neigh);
256  neighneigh->set_neighbor(snn, &elem);
257  }
258 
259  Elem * elemoldneigh = elem.neighbor_ptr((n+2)%3);
260  if (elemoldneigh)
261  {
262  unsigned int seon = elemoldneigh->which_neighbor_am_i(&elem);
263  elemoldneigh->set_neighbor(seon, neigh);
264  }
265 
266  elem.set_neighbor((n+1)%3, neigh->neighbor_ptr((nn+2)%3));
267  neigh->set_neighbor((nn+1)%3, elem.neighbor_ptr((n+2)%3));
268  elem.set_neighbor((n+2)%3, neigh);
269  neigh->set_neighbor((nn+2)%3, &elem);
270 
271  // Start over after this much change, don't just loop to the
272  // next neighbor
273  break;
274  }
275  }

279 unsigned int segment_intersection(const Elem & elem,
280  Point & source,
281  const Point & target,
282  unsigned int source_side)
283 {
284  libmesh_assert_equal_to(elem.dim(), 2);
285 
286  const auto ns = elem.n_sides();
287 
288  for (auto s : make_range(ns))
289  {
290  // Don't go backwards just because some FP roundoff said to
291  if (s == source_side)
292  continue;
293 
294  const Point v0 = elem.point(s);
295  const Point v1 = elem.point((s+1)%ns);
296 
297  // Calculate intersection parameters (fractions of the distance
298  // along each segment)
299  const Real raydx = target(0)-source(0),
300  raydy = target(1)-source(1),
301  edgedx = v1(0)-v0(0),
302  edgedy = v1(1)-v0(1);
303  const Real denom = edgedx * raydy - edgedy * raydx;
304 
305  // divide-by-zero means the segments are parallel
306  if (denom == 0)
307  continue;
308 
309  const Real one_over_denom = 1 / denom;
310 
311  const Real targetsdx = v1(0)-target(0),
312  targetsdy = v1(1)-target(1);
313 
314  const Real t_num = targetsdx * raydy -
315  targetsdy * raydx;
316  const Real t = t_num * one_over_denom;
317 
318  if (t < -TOLERANCE*TOLERANCE || t > 1 + TOLERANCE*TOLERANCE)
319  continue;
320 
321  const Real u_num = targetsdx * edgedy - targetsdy * edgedx;
322  const Real u = u_num * one_over_denom;
323 
324  if (u < -TOLERANCE*TOLERANCE || u > 1 + TOLERANCE*TOLERANCE)
325  continue;
326 
327 /*
328  // Partial workaround for an old poly2tri bug (issue #39): if we
329  // end up with boundary points that are nearly-collinear but
330  // infinitesimally concave, p2t::CDT::Triangulate throws a "null
331  // triangle" exception. So let's try to be infinitesimally
332  // convex instead.
333  const Real ray_fraction = (1-u) * (1+TOLERANCE*TOLERANCE);
334 */
335  const Real ray_fraction = (1-u);
336 
337  source(0) += raydx * ray_fraction;
338  source(1) += raydy * ray_fraction;
339  return s;
340  }
341 
342  return libMesh::invalid_uint;
343 }

367 // Primary function responsible for performing the triangulation
368 void Poly2TriTriangulator::triangulate()
369 {
370  LOG_SCOPE("triangulate()", "Poly2TriTriangulator");
371 
372  // We only operate on serialized meshes. And it's not safe to
373  // serialize earlier, because it would then be possible for the user
374  // to re-parallelize the mesh in between there and here.
375  MeshSerializer serializer(_mesh);
376 
377  // We don't yet support every set of Triangulator options in the
378  // poly2tri implementation
379 
380  // We don't support convex hull triangulation, only triangulation of
381  // (implicitly defined, by node ordering) polygons (with holes if
382  // requested)
383  if (_triangulation_type != PSLG)
384  libmesh_not_implemented();
385 
386  // We currently don't handle region specifications
387  if (_regions)
388  libmesh_not_implemented();
389 
390  // We won't support quads any time soon, or 1D/3D in this interface
391  // ever.
392  if (_elem_type != TRI3 &&
393  _elem_type != TRI6 &&
394  _elem_type != TRI7)
395  libmesh_not_implemented();
396 
397  // If we have no explicit segments defined, we may get them from
398  // mesh elements
399  this->elems_to_segments();
400 
401  // If we *still* have no explicit segments defined, we get them from
402  // the order of nodes.
403  this->nodes_to_segments(_n_boundary_nodes);
404 
405  // Insert additional new points in between existing boundary points,
406  // if that is requested and reasonable
407  this->insert_any_extra_boundary_points();
408 
409  // Triangulate the points we have, then see if we need to add more;
410  // repeat until we don't need to add more.
411  //
412  // This is currently done redundantly in parallel; make sure no
413  // processor quits before the others.
414  do
415  {
416  libmesh_parallel_only(_mesh.comm());
417  this->triangulate_current_points();
418  }
419  while (this->insert_refinement_points());
420 
421  libmesh_parallel_only(_mesh.comm());
422 
423  // Okay, we really do need to support boundary ids soon, but we
424  // don't yet
425  if (_markers)
426  libmesh_not_implemented();
427 
428  _mesh.set_mesh_dimension(2);
429 
430  // To the naked eye, a few smoothing iterations usually looks better,
431  // so we do this by default unless the user says not to.
432  if (this->_smooth_after_generating)
433  LaplaceMeshSmoother(_mesh).smooth(2);
434 
435  // Prepare the mesh for use before returning. This ensures (among
436  // other things) that it is partitioned and therefore users can
437  // iterate over local elements, etc.
438  _mesh.prepare_for_use();
439 }

459  (const BoundaryInfo & boundary_info,
460  const Elem & elem,

462 {

467  boundary_info.boundary_ids(&elem, side, bcids);
468

471 
472  if (bcids[0] == 0)

482  return hole->refine_boundary_allowed();

486 void Poly2TriTriangulator::triangulate_current_points()
487 {
488  LOG_SCOPE("triangulate_current_points()", "Poly2TriTriangulator");
489 
490  // Will the triangulation have holes?
491  const std::size_t n_holes = _holes != nullptr ? _holes->size() : 0;
492 
493  // Mapping from Poly2Tri points to libMesh nodes, so we can get the
494  // connectivity translated back later.
495  std::map<const p2t::Point, Node *, P2TPointCompare> point_node_map;
496 
497  // Poly2Tri data structures
498  // Poly2Tri takes vectors of pointers-to-Point for some reason, but
499  // we'll just make those shims to vectors of Point rather than
500  // individually/manually heap allocating everything.
501  std::vector<p2t::Point> outer_boundary_points;
502  std::vector<std::vector<p2t::Point>> inner_hole_points(n_holes);
503 
504  dof_id_type nn = _mesh.max_node_id();
505  libmesh_error_msg_if
506  (!nn, "Poly2TriTriangulator cannot triangulate an empty mesh!");
507 
508  // Unless we're using an explicit segments list, we assume node ids
509  // are contiguous here.
510  if (this->segments.empty())
511  libmesh_error_msg_if
512  (_mesh.n_nodes() != nn,
513  "Poly2TriTriangulator needs contiguous node ids or explicit segments!");
514 
515  // And if we have more nodes than outer boundary points, the rest
516  // may be interior "Steiner points". We use a set here so we can
517  // cheaply search and erase from it later, when we're identifying
518  // hole points.
519  std::set<p2t::Point, P2TPointCompare> steiner_points;
520 
521  // If we were asked to use all mesh nodes as boundary nodes, now's
522  // the time to see how many that is.
523  if (_n_boundary_nodes == DofObject::invalid_id)
524  {
525  _n_boundary_nodes = _mesh.n_nodes();
526  libmesh_assert_equal_to(std::ptrdiff_t(_n_boundary_nodes),
527  std::distance(_mesh.nodes_begin(),
528  _mesh.nodes_end()));
529 
530  }
531  else
532  libmesh_assert_less_equal(_n_boundary_nodes,
533  _mesh.n_nodes());
534 
535  // Prepare poly2tri points for our nodes, sorted into outer boundary
536  // points and interior Steiner points.
537 
538  if (this->segments.empty())
539  {
540  // If we have no segments even after taking elems into account,
541  // the nodal id ordering defines our outer polyline ordering
542  for (auto & node : _mesh.node_ptr_range())
543  {
544  const p2t::Point pt = to_p2t(*node);
545 
546  // If we're out of boundary nodes, the rest are going to be
547  // Steiner points or hole points
548  if (node->id() < _n_boundary_nodes)
549  outer_boundary_points.push_back(pt);
550  else
551  steiner_points.insert(pt);
552 
553  // We're not going to support overlapping nodes on the boundary
554  if (point_node_map.count(pt))
555  libmesh_not_implemented();
556 
557  point_node_map.emplace(pt, node);
558  }
559  }
560  // If we have explicit segments defined, that's our outer polyline
561  // ordering:
562  else
563  {
564  // Let's make sure our segments are in order
565  dof_id_type last_id = DofObject::invalid_id;
566 
567  // Add nodes from every segment, in order, to the outer polyline
568  for (auto [segment_start, segment_end] : this->segments)
569  {
570  if (last_id != DofObject::invalid_id)
571  libmesh_error_msg_if(segment_start != last_id,
572  "Disconnected triangulator segments");
573  last_id = segment_end;
574 
575  Node * node = _mesh.query_node_ptr(segment_start);
576 
577  libmesh_error_msg_if(!node,
578  "Triangulator segments reference nonexistent node id " <<
579  segment_start);
580 
581  outer_boundary_points.emplace_back((*node)(0), (*node)(1));
582  p2t::Point * pt = &outer_boundary_points.back();
583 
584  // We're not going to support overlapping nodes on the boundary
585  if (point_node_map.count(*pt))
586  libmesh_not_implemented_msg
587  ("Triangulating overlapping boundary nodes is unsupported");
588 
589  point_node_map.emplace(*pt, node);
590  }
591 
592  libmesh_error_msg_if(last_id != this->segments[0].first,
593  "Non-closed-loop triangulator segments");
594 
595  // If we have points that aren't in any segments, those will be
596  // Steiner points
597  for (auto & node : _mesh.node_ptr_range())
598  {
599  const p2t::Point pt = to_p2t(*node);
600 
601  auto it = point_node_map.find(pt);
602 
603  if (it == point_node_map.end())
604  {
605  steiner_points.insert(pt);
606  point_node_map.emplace(pt, node);
607  }
608  else
609  libmesh_assert_equal_to(it->second, node);
610  }
611  }
612 
613  // If we have any elements from a previous triangulation, we're
614  // going to replace them with our own. If we have any elements that
615  // were used to create our segments, we're done creating and we no
616  // longer need them.
617  _mesh.clear_elems();
618 
619  // Keep track of what boundary ids we want to assign to each new
620  // triangle. We'll give the outer boundary BC 0, and give holes ids
621  // starting from 1. We've already got the point_node_map to find
622  // nodes, so we can just key on pairs of node ids to identify a side.
623  std::unordered_map<std::pair<dof_id_type,dof_id_type>,
624  boundary_id_type, libMesh::hash> side_boundary_id;
625 
626  const boundary_id_type outer_bcid = 0;
627  const std::size_t n_outer = outer_boundary_points.size();
628 
629  for (auto i : make_range(n_outer))
630  {
631  const Node * node1 =
632  libmesh_map_find(point_node_map, outer_boundary_points[i]),
633  * node2 =
634  libmesh_map_find(point_node_map, outer_boundary_points[(i+1)%n_outer]);
635 
636  side_boundary_id.emplace(std::make_pair(node1->id(),
637  node2->id()),
638  outer_bcid);
639  }
640 
641  // Create poly2tri triangulator with our mesh points
642  std::vector<p2t::Point *> outer_boundary_pointers(n_outer);
643  std::transform(outer_boundary_points.begin(),
644  outer_boundary_points.end(),
645  outer_boundary_pointers.begin(),
646  [](p2t::Point & p) { return &p; });
647 
648 
649  // Make sure shims for holes last as long as the CDT does; the
650  // poly2tri headers don't make clear whether or not they're hanging
651  // on to references to these vectors, and it would be reasonable for
652  // them to do so.
653  std::vector<std::vector<p2t::Point *>> inner_hole_pointers(n_holes);
654 
655  p2t::CDT cdt{outer_boundary_pointers};
656 
657  // Add any holes
658  for (auto h : make_range(n_holes))
659  {
660  const Hole * initial_hole = (*_holes)[h];
661  auto it = replaced_holes.find(initial_hole);
662  const Hole & our_hole =
663  (it == replaced_holes.end()) ?
664  *initial_hole : *it->second;
665  auto & poly2tri_hole = inner_hole_points[h];
666 
667  for (auto i : make_range(our_hole.n_points()))
668  {
669  Point p = our_hole.point(i);
670  poly2tri_hole.emplace_back(to_p2t(p));
671 
672  const auto & pt = poly2tri_hole.back();
673 
674  // This won't be a steiner point.
675  steiner_points.erase(pt);
676 
677  // If we see a hole point already in the mesh, we'll share
678  // that node. This might be a problem if it's a boundary
679  // node, but it might just be the same hole point already
680  // added during a previous triangulation refinement step.
681  if (point_node_map.count(pt))
682  {
683  libmesh_assert_equal_to
684  (point_node_map[pt],
685  _mesh.query_node_ptr(point_node_map[pt]->id()));
686  }
687  else
688  {
689  Node * node = _mesh.add_point(p, nn++);
690  point_node_map[pt] = node;
691  }
692  }
693 
694  const boundary_id_type inner_bcid = h+1;
695  const std::size_t n_inner = poly2tri_hole.size();
696 
697  for (auto i : make_range(n_inner))
698  {
699  const Node * node1 =
700  libmesh_map_find(point_node_map, poly2tri_hole[i]),
701  * node2 =
702  libmesh_map_find(point_node_map, poly2tri_hole[(i+1)%n_inner]);
703 
704  side_boundary_id.emplace(std::make_pair(node1->id(),
705  node2->id()),
706  inner_bcid);
707  }
708 
709  auto & poly2tri_ptrs = inner_hole_pointers[h];
710  poly2tri_ptrs.resize(n_inner);
711 
712  std::transform(poly2tri_hole.begin(),
713  poly2tri_hole.end(),
714  poly2tri_ptrs.begin(),
715  [](p2t::Point & p) { return &p; });
716 
717  cdt.AddHole(poly2tri_ptrs);
718  }
719 
720  // Add any steiner points. We had them in a set, but post-C++11
721  // that won't give us non-const element access (even if we
722  // pinky-promise not to change the elements in any way that affects
723  // our Comparator), and Poly2Tri wants non-const elements (to store
724  // edge data?), so we need to move them here.
725  std::vector<p2t::Point> steiner_vector(steiner_points.begin(), steiner_points.end());
726  steiner_points.clear();
727  for (auto & p : steiner_vector)
728  cdt.AddPoint(&p);
729 
730  // Triangulate!
731  cdt.Triangulate();
732 
733  // Get poly2tri triangles, turn them into libMesh triangles
734  std::vector<p2t::Triangle *> triangles = cdt.GetTriangles();
735 
736  // Do our own numbering, even on DistributedMesh
737  dof_id_type next_id = 0;
738 
739  BoundaryInfo & boundary_info = _mesh.get_boundary_info();
740  boundary_info.clear();
741 
742  // Add the triangles to our Mesh data structure.
743  for (auto ptri_ptr : triangles)
744  {
745  p2t::Triangle & ptri = *ptri_ptr;
746 
747  // We always use TRI3 here, since that's what we have nodes for;
748  // if we need a higher order we can convert at the end.
749  auto elem = Elem::build_with_id(TRI3, next_id++);
750  for (auto v : make_range(3))
751  {
752  const p2t::Point & vertex = *ptri.GetPoint(v);
753 
754  Node * node = libmesh_map_find(point_node_map, vertex);
755  libmesh_assert(node);
756  elem->set_node(v) = node;
757  }
758 
759  // We expect a consistent triangle orientation
760  libmesh_assert(!elem->is_flipped());
761 
762  Elem * added_elem = _mesh.add_elem(std::move(elem));
763 
764  for (auto v : make_range(3))
765  {
766  const Node & node1 = added_elem->node_ref(v),
767  & node2 = added_elem->node_ref((v+1)%3);
768 
769  auto it = side_boundary_id.find(std::make_pair(node1.id(), node2.id()));
770  if (it == side_boundary_id.end())
771  it = side_boundary_id.find(std::make_pair(node2.id(), node1.id()));
772  if (it != side_boundary_id.end())
773  boundary_info.add_side(added_elem, v, it->second);
774  }
775  }
776 }

780 bool Poly2TriTriangulator::insert_refinement_points()
781 {
782  LOG_SCOPE("insert_refinement_points()", "Poly2TriTriangulator");
783 
784  if (this->minimum_angle() != 0)
785  libmesh_not_implemented();
786 
787  // We need neighbor pointers for ray casting and cavity finding
788  UnstructuredMesh & mesh = dynamic_cast<UnstructuredMesh &>(this->_mesh);
789  mesh.find_neighbors();
790 
791  if (this->desired_area() == 0 &&
792  this->get_desired_area_function() == nullptr)
793  return false;
794 
795  BoundaryInfo & boundary_info = _mesh.get_boundary_info();
796 
797  // We won't immediately add these, lest we invalidate iterators on a
798  // ReplicatedMesh. They'll still be in the mesh neighbor topology
799  // for the purpose of doing Delaunay cavity stuff, so we need to
800  // manage memory here, but there's no point in adding them to the
801  // Mesh just to remove them again afterward when we hit up poly2tri.
802 
803  // We'll need to be able to remove new elems from new_elems, in
804  // cases where a later refinement insertion has a not-yet-added
805  // element in its cavity, so we'll use a map here to make searching
806  // possible.
807  //
808  // For parallel consistency, we can't order a container we plan to
809  // iterate through based on Elem * or a hash of it. We'll be doing
810  // Delaunay swaps so we can't iterate based on geometry. These are
811  // not-yet-added elements so we can't iterate based on proper
812  // element ids ... but we can set a temporary element id to use for
813  // the purpose.
814  struct cmp {
815  bool operator()(Elem * a, Elem * b) const {
816  libmesh_assert(a == b || a->id() != b->id());
817  return (a->id() < b->id());
818  }
819  } comp;
820 
821  std::map<Elem *, std::unique_ptr<Elem>, decltype(comp)> new_elems(comp);
822 
823  // We should already be Delaunay when we get here, otherwise we
824  // won't be able to stay Delaunay later. But we're *not* always
825  // Delaunay when we get here? What the hell, poly2tri? Fixing this
826  // is expensive!
827  {
828  // restore_delaunay should get to the same Delaunay triangulation up to
829  // isomorphism regardless of ordering ... but we actually care
830  // about the isomorphisms! If a triangle's nodes are permuted on
831  // one processor vs another that's an issue. So sort our input
832  // carefully.
833  std::set<Elem *, decltype(comp)> all_elems
834  { mesh.elements_begin(), mesh.elements_end(), comp };
835 
836  restore_delaunay(all_elems, boundary_info);
837 
838  libmesh_assert_delaunay(mesh, new_elems);
839  }
840 
841  // Map of which points follow which in the boundary polylines. If
842  // we have to add new boundary points, we'll use this to construct
843  // an updated this->segments to retriangulate with. If we have to
844  // add new hole points, we'll use this to insert points into an
845  // ArbitraryHole.
846  std::unordered_map<Point, Node *> next_boundary_node;
847 
848  // In cases where we've been working with contiguous node id ranges;
849  // let's keep it that way.
850  dof_id_type nn = _mesh.max_node_id();
851  dof_id_type ne = _mesh.max_elem_id();
852 
853  // We can't handle duplicated nodes. We shouldn't ever create one,
854  // but let's make sure of that.
855 #ifdef DEBUG
856  std::unordered_set<Point> mesh_points;
857  for (const Node * node : mesh.node_ptr_range())
858  {
859  libmesh_assert(!mesh_points.count(*node));
860  mesh_points.insert(*node);
861  }
862 #endif
863 
864  auto add_point = [&mesh,
865 #ifdef DEBUG
866  &mesh_points,
867 #endif
868  &nn](const Point & p)
869  {
870 #ifdef DEBUG
871  libmesh_assert(!mesh_points.count(p));
872  mesh_points.insert(p);
873 #endif
874  return mesh.add_point(p, nn++);
875  };
876 
877  for (auto & elem : mesh.element_ptr_range())
878  {
879  // element_ptr_range skips deleted elements ... right?
880  libmesh_assert(elem);
881  libmesh_assert(elem->valid_id());
882 
883  // We only handle triangles in our triangulation
884  libmesh_assert_equal_to(elem->level(), 0u);
885  libmesh_assert_equal_to(elem->type(), TRI3);
886 
887  // If this triangle is as small as we desire, move along
888  if (!should_refine_elem(*elem))
889  continue;
890 
891  // Otherwise add a Steiner point. We'd like to add the
892  // circumcenter ...
893  Point new_pt = elem->quasicircumcenter();
894 
895  // And to give it a node;
896  Node * new_node = nullptr;
897 
898  // But that might be outside our triangle, or even outside the
899  // boundary. We'll find a triangle that should contain our new
900  // point
901  Elem * cavity_elem = elem; // Start looking at elem anyway
902 
903  // We'll refine a boundary later if necessary.
904  auto boundary_refine = [this, &next_boundary_node,
905  &cavity_elem, &new_node]
906  (unsigned int side)
907  {
908  libmesh_ignore(this); // Only used in dbg/devel
909  libmesh_assert(new_node);
910  libmesh_assert(new_node->valid_id());
911 
912  Node * old_segment_start = cavity_elem->node_ptr(side),
913  * old_segment_end = cavity_elem->node_ptr((side+1)%3);
914  libmesh_assert(old_segment_start);
915  libmesh_assert(old_segment_start->valid_id());
916  libmesh_assert(old_segment_end);
917  libmesh_assert(old_segment_end->valid_id());
918 
919  auto it = next_boundary_node.find(*old_segment_start);
920  if (it != next_boundary_node.end())
921  {
922  libmesh_assert(it->second == old_segment_end);
923  it->second = new_node;
924  }
925  else
926  {
927  // This would be an O(N) sanity check if we already
928  // have a segments vector or any holes. :-P
929  libmesh_assert(!this->segments.empty() ||
930  (_holes && !_holes->empty()) ||
931  (old_segment_end->id() ==
932  old_segment_start->id() + 1));
933  next_boundary_node[*old_segment_start] = new_node;
934  }
935 
936  next_boundary_node[*new_node] = old_segment_end;
937  };
938 
939  // Let's find a triangle containing our new point, or at least
940  // containing the end of a ray leading from our current triangle
941  // to the new point.
942  Point ray_start = elem->vertex_average();
943 
944  // What side are we coming from, and what side are we going to?
945  unsigned int source_side = invalid_uint;
946  unsigned int side = invalid_uint;
947 
948  while (!cavity_elem->contains_point(new_pt))
949  {
950  side = segment_intersection(*cavity_elem, ray_start, new_pt, source_side);
951 
952  libmesh_assert_not_equal_to (side, invalid_uint);
953 
954  Elem * neigh = cavity_elem->neighbor_ptr(side);
955  // If we're on a boundary, stop there. Refine the boundary
956  // if we're allowed, the boundary element otherwise.
957  if (!neigh)
958  {
959  if (this->is_refine_boundary_allowed(boundary_info,
960  *cavity_elem,
961  side))
962  {
963  new_pt = ray_start;
964  new_node = add_point(new_pt);
965  boundary_refine(side);
966  }
967  else
968  {
969  // Should we just add the vertex average of the
970  // boundary element, to minimize the number of
971  // slivers created?
972  //
973  // new_pt = cavity_elem->vertex_average();
974  //
975  // That works for a while, but it
976  // seems to be able to "run away" and leave us with
977  // crazy slivers on boundaries if we push interior
978  // refinement too far while disabling boundary
979  // refinement.
980  //
981  // Let's go back to refining our original problem
982  // element.
983  cavity_elem = elem;
984  new_pt = cavity_elem->vertex_average();
985  new_node = add_point(new_pt);
986 
987  // This was going to be a side refinement but it's
988  // now an internal refinement
989  side = invalid_uint;
990  }
991 
992  break;
993  }
994 
995  source_side = neigh->which_neighbor_am_i(cavity_elem);
996  cavity_elem = neigh;
997  side = invalid_uint;
998  }
999 
1000  // If we're ready to create a new node and we're not on a
1001  // boundary ... should we be? We don't want to create any
1002  // sliver elements or confuse poly2tri or anything.
1003  if (side == invalid_uint && !new_node)
1004  {
1005  unsigned int worst_side = libMesh::invalid_uint;
1006  Real worst_cos = 1;
1007  for (auto s : make_range(3u))
1008  {
1009  // We never snap to a non-domain-boundary
1010  if (cavity_elem->neighbor_ptr(s))
1011  continue;
1012 
1013  Real ax = cavity_elem->point(s)(0) - new_pt(0),
1014  ay = cavity_elem->point(s)(1) - new_pt(1),
1015  bx = cavity_elem->point((s+1)%3)(0) - new_pt(0),
1016  by = cavity_elem->point((s+1)%3)(1) - new_pt(1);
1017  const Real my_cos = (ax*bx+ay*by) /
1018  std::sqrt(ax*ax+ay*ay) /
1019  std::sqrt(bx*bx+by*by);
1020 
1021  if (my_cos < worst_cos)
1022  {
1023  worst_side = s;
1024  worst_cos = my_cos;
1025  }
1026  }
1027 
1028  // If we'd create a sliver element on the side, let's just
1029  // refine the side instead, if we're allowed.
1030  if (worst_cos < -0.6) // -0.5 is the best we could enforce?
1031  {
1032  side = worst_side;
1033 
1034  if (this->is_refine_boundary_allowed(boundary_info,
1035  *cavity_elem,
1036  side))
1037  {
1038  // Let's just try bisecting for now
1039  new_pt = (cavity_elem->point(side) +
1040  cavity_elem->point((side+1)%3)) / 2;
1041  new_node = add_point(new_pt);
1042  boundary_refine(side);
1043  }
1044  else // Do the best we can under these restrictions
1045  {
1046  new_pt = cavity_elem->vertex_average();
1047  new_node = add_point(new_pt);
1048 
1049  // This was going to be a side refinement but it's
1050  // now an internal refinement
1051  side = invalid_uint;
1052  }
1053  }
1054  else
1055  new_node = add_point(new_pt);
1056  }
1057  else
1058  libmesh_assert(new_node);
1059 
1060  // Find the Delaunay cavity around the new point.
1061  std::set<Elem *, decltype(comp)> cavity(comp);
1062 
1063  std::set<Elem *, decltype(comp)> unchecked_cavity ({cavity_elem}, comp);
1064  while (!unchecked_cavity.empty())
1065  {
1066  std::set<Elem *, decltype(comp)> checking_cavity(comp);
1067  checking_cavity.swap(unchecked_cavity);
1068  for (Elem * checking_elem : checking_cavity)
1069  {
1070  for (auto s : make_range(3u))
1071  {
1072  Elem * neigh = checking_elem->neighbor_ptr(s);
1073  if (!neigh)
1074  continue;
1075  if (checking_cavity.find(neigh) !=
1076  checking_cavity.end())
1077  continue;
1078  if (cavity.find(neigh) !=
1079  cavity.end())
1080  continue;
1081 
1082  if (in_circumcircle(*neigh, new_pt, TOLERANCE*TOLERANCE))
1083  unchecked_cavity.insert(neigh);
1084  }
1085  }
1086 
1087  libmesh_merge_move(cavity, checking_cavity);
1088  }
1089 
1090  // Retriangulate the Delaunay cavity.
1091  // Each of our cavity triangle edges that are exterior to the
1092  // cavity will be a source of one new triangle.
1093 
1094  // Set of elements that might need Delaunay swaps
1095  std::set<Elem *, decltype(comp)> check_delaunay_on(comp);
1096 
1097  // Keep maps for doing neighbor pointer assignment. Not going
1098  // to iterate through these so hashing pointers is fine.
1099  std::unordered_map<Node *, std::pair<Elem *, boundary_id_type>>
1100  neighbors_CCW, neighbors_CW;
1101 
1102  for (Elem * old_elem : cavity)
1103  {
1104  for (auto s : make_range(3u))
1105  {
1106  Elem * neigh = old_elem->neighbor_ptr(s);
1107  if (!neigh || !cavity.count(neigh))
1108  {
1109  Node * node_CW = old_elem->node_ptr(s),
1110  * node_CCW = old_elem->node_ptr((s+1)%3);
1111 
1112  auto set_neighbors =
1113  [&neighbors_CW, &neighbors_CCW, &node_CW,
1114  &node_CCW, &boundary_info]
1115  (Elem * new_neigh, boundary_id_type bcid)
1116  {
1117  // Set clockwise neighbor and vice-versa if possible
1118  auto CW_it = neighbors_CW.find(node_CW);
1119  if (CW_it == neighbors_CW.end())
1120  {
1121  libmesh_assert(!neighbors_CCW.count(node_CW));
1122  neighbors_CCW[node_CW] = std::make_pair(new_neigh, bcid);
1123  }
1124  else
1125  {
1126  Elem * neigh_CW = CW_it->second.first;
1127  if (new_neigh)
1128  {
1129  new_neigh->set_neighbor(0, neigh_CW);
1130  boundary_id_type bcid_CW = CW_it->second.second;
1131  if (bcid_CW != BoundaryInfo::invalid_id)
1132  boundary_info.add_side(new_neigh, 0, bcid_CW);
1133 
1134  }
1135  if (neigh_CW)
1136  {
1137  neigh_CW->set_neighbor(2, new_neigh);
1138  if (bcid != BoundaryInfo::invalid_id)
1139  boundary_info.add_side(neigh_CW, 2, bcid);
1140  }
1141  neighbors_CW.erase(CW_it);
1142  }
1143 
1144  // Set counter-CW neighbor and vice-versa if possible
1145  auto CCW_it = neighbors_CCW.find(node_CCW);
1146  if (CCW_it == neighbors_CCW.end())
1147  {
1148  libmesh_assert(!neighbors_CW.count(node_CCW));
1149  neighbors_CW[node_CCW] = std::make_pair(new_neigh, bcid);
1150  }
1151  else
1152  {
1153  Elem * neigh_CCW = CCW_it->second.first;
1154  if (new_neigh)
1155  {
1156  boundary_id_type bcid_CCW = CCW_it->second.second;
1157  new_neigh->set_neighbor(2, neigh_CCW);
1158  if (bcid_CCW != BoundaryInfo::invalid_id)
1159  boundary_info.add_side(new_neigh, 2, bcid_CCW);
1160  }
1161  if (neigh_CCW)
1162  {
1163  neigh_CCW->set_neighbor(0, new_neigh);
1164  if (bcid != BoundaryInfo::invalid_id)
1165  boundary_info.add_side(neigh_CCW, 0, bcid);
1166  }
1167  neighbors_CCW.erase(CCW_it);
1168  }
1169  };
1170 
1171  // We aren't going to try to add a sliver element if we
1172  // have a new boundary node here. We do need to
1173  // keep track of other elements' neighbors, though.
1174  if (old_elem == cavity_elem &&
1175  s == side)
1176  {
1177  std::vector<boundary_id_type> bcids;
1178  boundary_info.boundary_ids(old_elem, s, bcids);
1179  libmesh_assert_equal_to(bcids.size(), 1);
1180  set_neighbors(nullptr, bcids[0]);
1181  continue;
1182  }
1183 
1184  auto new_elem = Elem::build_with_id(TRI3, ne++);
1185  new_elem->set_node(0) = new_node;
1186  new_elem->set_node(1) = node_CW;
1187  new_elem->set_node(2) = node_CCW;
1188  libmesh_assert(!new_elem->is_flipped());
1189 
1190  // Set in-and-out-of-cavity neighbor pointers
1191  new_elem->set_neighbor(1, neigh);
1192  if (neigh)
1193  {
1194  const unsigned int neigh_s =
1195  neigh->which_neighbor_am_i(old_elem);
1196  neigh->set_neighbor(neigh_s, new_elem.get());
1197  }
1198  else
1199  {
1200  std::vector<boundary_id_type> bcids;
1201  boundary_info.boundary_ids(old_elem, s, bcids);
1202  boundary_info.add_side(new_elem.get(), 1, bcids);
1203  }
1204 
1205  // Set in-cavity neighbors' neighbor pointers
1206  set_neighbors(new_elem.get(), BoundaryInfo::invalid_id);
1207 
1208  // C++ allows function argument evaluation in any
1209  // order, but we need get() to precede move
1210  Elem * new_elem_ptr = new_elem.get();
1211  new_elems.emplace(new_elem_ptr, std::move(new_elem));
1212 
1213  check_delaunay_on.insert(new_elem_ptr);
1214  }
1215  }
1216 
1217  boundary_info.remove(old_elem);
1218  }
1219 
1220  // Now that we're done using our cavity elems (including with a
1221  // cavity.find() that used a comparator that dereferences the
1222  // elmeents!) it's safe to delete them.
1223  for (Elem * old_elem : cavity)
1224  {
1225  auto it = new_elems.find(old_elem);
1226  if (it == new_elems.end())
1227  mesh.delete_elem(old_elem);
1228  else
1229  new_elems.erase(it);
1230  }
1231 
1232  // Everybody found their match?
1233  libmesh_assert(neighbors_CW.empty());
1234  libmesh_assert(neighbors_CCW.empty());
1235 
1236  // Because we're preserving boundaries here, our naive cavity
1237  // triangulation might not be a Delaunay triangulation. Let's
1238  // check and if necessary fix that; we depend on it when doing
1239  // future point insertions.
1240  restore_delaunay(check_delaunay_on, boundary_info);
1241 
1242  // This is too expensive to do on every cavity in devel mode
1243 #ifdef DEBUG
1244  libmesh_assert_delaunay(mesh, new_elems);
1245 #endif
1246  }
1247 
1248  // If we added any new boundary nodes, we're going to need to keep
1249  // track of the changes they made to the outer polyline and/or to
1250  // any holes.
1251  if (!next_boundary_node.empty())
1252  {
1253  auto checked_emplace = [this](dof_id_type new_first,
1254  dof_id_type new_second)
1255  {
1256 #ifdef DEBUG
1257  for (auto [first, second] : this->segments)
1258  {
1259  libmesh_assert_not_equal_to(first, new_first);
1260  libmesh_assert_not_equal_to(second, new_second);
1261  }
1262  if (!this->segments.empty())
1263  libmesh_assert_equal_to(this->segments.back().second, new_first);
1264 #endif
1265  libmesh_assert_not_equal_to(new_first, new_second);
1266 
1267  this->segments.emplace_back (new_first, new_second);
1268  };
