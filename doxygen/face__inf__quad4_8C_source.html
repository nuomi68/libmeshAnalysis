107 bool InfQuad4::contains_point (const Point & p, Real tol) const
108 {
109  /*
110  * make use of the fact that infinite elements do not
111  * live inside the envelope. Use a fast scheme to
112  * check whether point \p p is inside or outside
113  * our relevant part of the envelope. Note that
114  * this is not exclusive: the point may be outside
115  * the envelope, but contained in another infinite element.
116  * Therefore, if the distance is greater, do fall back
117  * to the scheme of using inverse_map().
118  */
119  const Point my_origin (this->origin());
120 
121  /*
122  * determine the minimal distance of the base from the origin
123  * use norm_sq() instead of norm(), it is slightly faster
124  */
125  const Real min_distance_sq = std::min((Point(this->point(0)-my_origin)).norm_sq(),
126  (Point(this->point(1)-my_origin)).norm_sq());
127 
128  /*
129  * work with 1% allowable deviation. Can still fall
130  * back to the InfFE::inverse_map()
131  */
132  const Real conservative_p_dist_sq = 1.01 * (Point(p-my_origin).norm_sq());
133 
134  if (conservative_p_dist_sq < min_distance_sq)
135  {
136  /*
137  * the physical point is definitely not contained
138  * in the element, return false.
139  */
140  return false;
141  }
142  else
143  {
144  /*
145  * cannot say anything, fall back to the inverse_map()
146  *
147  * Declare a basic FEType. Will use default in the base,
148  * and something else (not important) in radial direction.
149  */
150  FEType fe_type(default_order());
151 
152  const Point mapped_point = InfFEMap::inverse_map(dim(), this, p,
153  tol, false);
154 
155  return FEInterface::on_reference_element(mapped_point, this->type(), tol);
156  }
157 }

200  libmesh_error();
201 #endif // LIBMESH_ENABLE_DEPRECATED
202  }
203  else
204  {
205  switch (i)
206  {
207  case 0:
208  {
209  edge = std::make_unique<Edge2>();
210  break;
211  }
212 
213  // adjacent to another infinite element
214  case 1:
215  case 2:
216  {
217  edge = std::make_unique<InfEdge2>();

221  default:

224 
225  // Set the nodes
226  for (auto n : edge->node_index_range())
227  edge->set_node(n) = this->node_ptr(InfQuad4::side_nodes_map[i][n]);
228  }

230 #ifdef LIBMESH_ENABLE_DEPRECATED

232 #endif
233  edge->set_parent(nullptr);
234  edge->set_interior_parent(this);
235 
236  edge->subdomain_id() = this->subdomain_id();
237  edge->set_mapping_type(this->mapping_type());
238 #ifdef LIBMESH_ENABLE_AMR

240 #endif

255 void InfQuad4::connectivity(const unsigned int libmesh_dbg_var(sf),
256  const IOPackage iop,
257  std::vector<dof_id_type> & conn) const
258 {
259  libmesh_assert_less (sf, this->n_sub_elem());
260  libmesh_assert_not_equal_to (iop, INVALID_IO_PACKAGE);
261 
262  conn.resize(4);
263 
264  switch (iop)
265  {
266  case TECPLOT:
267  {
268  conn[0] = this->node_id(0)+1;
269  conn[1] = this->node_id(1)+1;
270  conn[2] = this->node_id(3)+1;
271  conn[3] = this->node_id(2)+1;
272  return;
273  }
274  case VTK:
275  {
276  conn[0] = this->node_id(0);
277  conn[1] = this->node_id(1);
278  conn[2] = this->node_id(3);
279  conn[3] = this->node_id(2);
280  return;
281  }
282  default:
283  libmesh_error_msg("Unsupported IO package " << iop);
284  }
285 }
