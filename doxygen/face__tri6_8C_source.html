143 bool Tri6::has_affine_map() const
144 {
145  // Make sure edges are straight
146  Point v = this->point(1) - this->point(0);
147  if (!v.relative_fuzzy_equals
148  ((this->point(3) - this->point(0))*2, affine_tol))
149  return false;
150  v = this->point(2) - this->point(1);
151  if (!v.relative_fuzzy_equals
152  ((this->point(4) - this->point(1))*2, affine_tol))
153  return false;
154  v = this->point(2) - this->point(0);
155  if (!v.relative_fuzzy_equals
156  ((this->point(5) - this->point(0))*2, affine_tol))
157  return false;
158 
159  return true;
160 }

226 void Tri6::connectivity(const unsigned int sf,
227  const IOPackage iop,
228  std::vector<dof_id_type> & conn) const
229 {
230  libmesh_assert_less (sf, this->n_sub_elem());
231  libmesh_assert_not_equal_to (iop, INVALID_IO_PACKAGE);
232 
233  switch (iop)
234  {
235  case TECPLOT:
236  {
237  conn.resize(4);
238  switch(sf)
239  {
240  case 0:
241  // linear sub-triangle 0
242  conn[0] = this->node_id(0)+1;
243  conn[1] = this->node_id(3)+1;
244  conn[2] = this->node_id(5)+1;
245  conn[3] = this->node_id(5)+1;
246 
247  return;
248 
249  case 1:
250  // linear sub-triangle 1
251  conn[0] = this->node_id(3)+1;
252  conn[1] = this->node_id(1)+1;
253  conn[2] = this->node_id(4)+1;
254  conn[3] = this->node_id(4)+1;
255 
256  return;
257 
258  case 2:
259  // linear sub-triangle 2
260  conn[0] = this->node_id(5)+1;
261  conn[1] = this->node_id(4)+1;
262  conn[2] = this->node_id(2)+1;
263  conn[3] = this->node_id(2)+1;
264 
265  return;
266 
267  case 3:
268  // linear sub-triangle 3
269  conn[0] = this->node_id(3)+1;
270  conn[1] = this->node_id(4)+1;
271  conn[2] = this->node_id(5)+1;
272  conn[3] = this->node_id(5)+1;
273 
274  return;
275 
276  default:
277  libmesh_error_msg("Invalid sf = " << sf);
278  }
279  }
280 
281  case VTK:
282  {
283  // VTK_QUADRATIC_TRIANGLE has same numbering as libmesh TRI6
284  conn.resize(Tri6::num_nodes);
285  for (auto i : index_range(conn))
286  conn[i] = this->node_id(i);
287  return;
288  }
289 
290  default:
291  libmesh_error_msg("Unsupported IO package " << iop);
292  }
293 }

297 BoundingBox Tri6::loose_bounding_box () const
298 {
299  // This might have curved edges, or might be a curved surface in
300  // 3-space, in which case the full bounding box can be larger than
301  // the bounding box of just the nodes.
302  //
303  //
304  // FIXME - I haven't yet proven the formula below to be correct for
305  // quadratics in 2D - RHS
306  Point pmin, pmax;
307 
308  for (unsigned d=0; d<LIBMESH_DIM; ++d)
309  {
310  Real center = this->point(0)(d);
311  for (unsigned int p=1; p != 6; ++p)
312  center += this->point(p)(d);
313  center /= 6;
314 
315  Real hd = std::abs(center - this->point(0)(d));
316  for (unsigned int p=1; p != 6; ++p)
317  hd = std::max(hd, std::abs(center - this->point(p)(d)));
318 
319  pmin(d) = center - hd;
320  pmax(d) = center + hd;
321  }
322 
323  return BoundingBox(pmin, pmax);
324 }

328 Real Tri6::volume () const
329 {
330  // This specialization is good for Lagrange mappings only
331  if (this->mapping_type() != LAGRANGE_MAP)
332  return this->Elem::volume();
333 
334  Real vol=0.;
335 
336 #if LIBMESH_DIM > 1
337  // Make copies of our points. It makes the subsequent calculations a bit
338  // shorter and avoids dereferencing the same pointer multiple times.
339  Point
340  x0 = point(0), x1 = point(1), x2 = point(2),
341  x3 = point(3), x4 = point(4), x5 = point(5);
342 
343  // Construct constant data vectors.
344  // \vec{x}_{\xi} = \vec{a1}*xi + \vec{b1}*eta + \vec{c1}
345  // \vec{x}_{\eta} = \vec{a2}*xi + \vec{b2}*eta + \vec{c2}
346  Point
347  a1 = 4*x0 + 4*x1 - 8*x3,
348  b1 = 4*x0 - 4*x3 + 4*x4 - 4*x5, /*=a2*/
349  c1 = -3*x0 - 1*x1 + 4*x3,
350  b2 = 4*x0 + 4*x2 - 8*x5,
351  c2 = -3*x0 - 1*x2 + 4*x5;
352 
353  // If a1 == b1 == a2 == b2 == 0, this is a TRI6 with straight sides,
354  // and we can use the TRI3 formula to compute the volume.
355  if (a1.relative_fuzzy_equals(Point(0,0,0)) &&
356  b1.relative_fuzzy_equals(Point(0,0,0)) &&
357  b2.relative_fuzzy_equals(Point(0,0,0)))
358  return 0.5 * cross_norm(c1, c2);
359 
360  // 7-point rule, exact for quintics.
361  const unsigned int N = 7;
362 
363  // Parameters of the quadrature rule
364  static const Real
365  w1 = Real(31)/480 + Real(std::sqrt(15.0L)/2400),
366  w2 = Real(31)/480 - Real(std::sqrt(15.0L)/2400),
367  q1 = Real(2)/7 + Real(std::sqrt(15.0L)/21),
368  q2 = Real(2)/7 - Real(std::sqrt(15.0L)/21);
369 
370  static const Real xi[N] = {Real(1)/3, q1, q1, 1-2*q1, q2, q2, 1-2*q2};
371  static const Real eta[N] = {Real(1)/3, q1, 1-2*q1, q1, q2, 1-2*q2, q2};
372  static const Real wts[N] = {Real(9)/80, w1, w1, w1, w2, w2, w2};
373 
374  // Approximate the area with quadrature
375  for (unsigned int q=0; q<N; ++q)
376  vol += wts[q] * cross_norm(xi[q]*a1 + eta[q]*b1 + c1,
377  xi[q]*b1 + eta[q]*b2 + c2);
378 #endif // LIBMESH_DIM > 1
379 
380  return vol;
381 }
