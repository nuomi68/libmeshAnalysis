73 dof_id_type Prism::key (const unsigned int s) const
74 {
75  libmesh_assert_less (s, this->n_sides());
76 
77  switch (s)
78  {
79  case 0: // the triangular face at z=0
80  case 4: // the triangular face at z=1
81  return this->compute_key (this->node_id(Prism6::side_nodes_map[s][0]),
82  this->node_id(Prism6::side_nodes_map[s][1]),
83  this->node_id(Prism6::side_nodes_map[s][2]));
84 
85  case 1: // the quad face at y=0
86  case 2: // the other quad face
87  case 3: // the quad face at x=0

127 std::unique_ptr<Elem> Prism::side_ptr (const unsigned int i)
128 {
129  libmesh_assert_less (i, this->n_sides());
130 
131  std::unique_ptr<Elem> face;
132 
133  // Set up the type of element
134  switch (i)
135  {
136  case 0: // the triangular face at z=0
137  case 4: // the triangular face at z=1
138  {
139  face = std::make_unique<Tri3>();
140  break;
141  }
142  case 1: // the quad face at y=0
143  case 2: // the other quad face
144  case 3: // the quad face at x=0
145  {
146  face = std::make_unique<Quad4>();
147  break;
148  }
149  default:
150  libmesh_error_msg("Invalid side i = " << i);
151  }
152 
153  // Set the nodes
154  for (auto n : face->node_index_range())
155  face->set_node(n) = this->node_ptr(Prism6::side_nodes_map[i][n]);
156 
157  return face;
158 }

162 void Prism::side_ptr (std::unique_ptr<Elem> & side,
163  const unsigned int i)
164 {
165  libmesh_assert_less (i, this->n_sides());
166 
167  switch (i)
168  {
169  // the base face
170  case 0: // the triangular face at z=0
171  case 4: // the triangular face at z=1
172  {
173  if (!side.get() || side->type() != TRI3)
174  {
175  side = this->side_ptr(i);
176  return;
177  }
178  break;
179  }
180 
181  case 1: // the quad face at y=0
182  case 2: // the other quad face
183  case 3: // the quad face at x=0
184  {
185  if (!side.get() || side->type() != QUAD4)
186  {
187  side = this->side_ptr(i);
188  return;
189  }
190  break;
191  }
192 
193  default:
194  libmesh_error_msg("Invalid side i = " << i);
195  }
196 
197  side->subdomain_id() = this->subdomain_id();
198 
199  // Set the nodes
200  for (auto n : side->node_index_range())
201  side->set_node(n) = this->node_ptr(Prism6::side_nodes_map[i][n]);
202 }
