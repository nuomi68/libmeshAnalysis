52 // Helper function for all_second_order, all_complete_order
53 void transfer_elem(const Elem & lo_elem,
54  std::unique_ptr<Elem> hi_elem,
55 #ifdef LIBMESH_ENABLE_UNIQUE_ID
56  unique_id_type max_unique_id,
57  unique_id_type max_new_nodes_per_elem,
58 #endif
59  UnstructuredMesh & mesh,
60  std::map<std::vector<dof_id_type>, Node *> & adj_vertices_to_hi_nodes)
61 {
62  libmesh_assert_equal_to (lo_elem.n_vertices(), hi_elem->n_vertices());
63 
64  const processor_id_type my_pid = mesh.processor_id();
65  const processor_id_type lo_pid = lo_elem.processor_id();
66 
67  /*
68  * form a vector that will hold the node id's of
69  * the vertices that are adjacent to the nth
70  * higher-order node.
71  */
72  std::vector<dof_id_type> adjacent_vertices_ids;
73 
74  /*
75  * Now handle the additional higher-order nodes. This
76  * is simply handled through a map that remembers
77  * the already-added nodes. This map maps the global
78  * ids of the vertices (that uniquely define this
79  * higher-order node) to the new node.
80  * Notation: hon = high-order node
81  */
82  const unsigned int hon_begin = lo_elem.n_nodes();
83  const unsigned int hon_end = hi_elem->n_nodes();
84 
85  for (unsigned int hon=hon_begin; hon<hon_end; hon++)
86  {
87  const unsigned int n_adjacent_vertices =
88  hi_elem->n_second_order_adjacent_vertices(hon);
89 
90  adjacent_vertices_ids.resize(n_adjacent_vertices);
91 
92  for (unsigned int v=0; v<n_adjacent_vertices; v++)
93  adjacent_vertices_ids[v] =
94  hi_elem->node_id( hi_elem->second_order_adjacent_vertex(hon,v) );
95 
96  /*
97  * \p adjacent_vertices_ids is now in order of the current
98  * side. sort it, so that comparisons with the
99  * \p adjacent_vertices_ids created through other elements'
100  * sides can match
101  */
102  std::sort(adjacent_vertices_ids.begin(),
103  adjacent_vertices_ids.end());
104 
105 
106  // does this set of vertices already have a mid-node added?
107  auto pos = adj_vertices_to_hi_nodes.equal_range (adjacent_vertices_ids);
108 
109  // no, not added yet
110  if (pos.first == pos.second)
111  {
112  /*
113  * for this set of vertices, there is no
114  * second_order node yet. Add it.
115  *
116  * compute the location of the new node as
117  * the average over the adjacent vertices.
118  */
119  Point new_location = mesh.point(adjacent_vertices_ids[0]);
120  for (unsigned int v=1; v<n_adjacent_vertices; v++)
121  new_location += mesh.point(adjacent_vertices_ids[v]);
122 
123  new_location /= static_cast<Real>(n_adjacent_vertices);
124 
125  /* Add the new point to the mesh.
126  *
127  * If we are on a serialized mesh, then we're doing this
128  * all in sync, and the node processor_id will be
129  * consistent between processors.
130  *
131  * If we are on a distributed mesh, we can fix
132  * inconsistent processor ids later, but only if every
133  * processor gives new nodes a *locally* consistent
134  * processor id, so we'll give the new node the
135  * processor id of an adjacent element for now and then
136  * we'll update that later if appropriate.
137  */
138  Node * hi_node = mesh.add_point
139  (new_location, DofObject::invalid_id, lo_pid);
140 
141  /* Come up with a unique unique_id for a potentially new
142  * node. On a distributed mesh we don't yet know what
143  * processor_id will definitely own it, so we can't let
144  * the pid determine the unique_id. But we're not
145  * adding unpartitioned nodes in sync, so we can't let
146  * the mesh autodetermine a unique_id for a new
147  * unpartitioned node either. So we have to pick unique
148  * unique_id values manually.
149  *
150  * We don't have to pick the *same* unique_id value as
151  * will be picked on other processors, though; we'll
152  * sync up each node later. We just need to make sure
153  * we don't duplicate any unique_id that might be chosen
154  * by the same process elsewhere.
155  */
156 #ifdef LIBMESH_ENABLE_UNIQUE_ID
157  unique_id_type new_unique_id = max_unique_id +
158  max_new_nodes_per_elem * lo_elem.id() +
159  hon - hon_begin;
160 
161  hi_node->set_unique_id(new_unique_id);
162 #endif
163 
164  /*
165  * insert the new node with its defining vertex
166  * set into the map, and relocate pos to this
167  * new entry, so that the hi_elem can use
168  * \p pos for inserting the node
169  */
170  adj_vertices_to_hi_nodes.emplace_hint
171  (pos.first, adjacent_vertices_ids, hi_node);
172 
173  hi_elem->set_node(hon) = hi_node;
174  }
175  // yes, already added.
176  else
177  {
178  Node * hi_node = pos.first->second;
179  libmesh_assert(hi_node);
180 
181  hi_elem->set_node(hon) = hi_node;
182 
183  // We need to ensure that the processor who should own a
184  // node *knows* they own the node. And because
185  // Node::choose_processor_id() may depend on Node id,
186  // which may not yet be authoritative, we still have to
187  // use a dumb-but-id-independent partitioning heuristic.
188  processor_id_type chosen_pid =
189  std::min (hi_node->processor_id(), lo_pid);
190 
191  // Plus, if we just discovered that we own this node,
192  // then on a distributed mesh we need to make sure to
193  // give it a valid id, not just a placeholder id!
194  if (!mesh.is_replicated() &&
195  hi_node->processor_id() != my_pid &&
196  chosen_pid == my_pid)
197  mesh.own_node(*hi_node);
198 
199  hi_node->processor_id() = chosen_pid;
200  }
201  }
202 
203  /*
204  * find_neighbors relies on remote_elem neighbor links being
205  * properly maintained.
206  */
207  for (auto s : lo_elem.side_index_range())
208  {
209  if (lo_elem.neighbor_ptr(s) == remote_elem)
210  hi_elem->set_neighbor(s, const_cast<RemoteElem*>(remote_elem));
211  }
212 
224  mesh.get_boundary_info().copy_boundary_ids
225  (mesh.get_boundary_info(), &lo_elem, hi_elem.get());
226 
227  /*
228  * The new second-order element is ready.
229  * Inserting it into the mesh will replace and delete
230  * the first-order element.
231  */
232  hi_elem->set_id(lo_elem.id());
233 #ifdef LIBMESH_ENABLE_UNIQUE_ID
234  hi_elem->set_unique_id(lo_elem.unique_id());
235 #endif
236  hi_elem->processor_id() = lo_pid;
237  hi_elem->subdomain_id() = lo_elem.subdomain_id();
238 
239  const unsigned int nei = lo_elem.n_extra_integers();
240  hi_elem->add_extra_integers(nei);
241  for (unsigned int i=0; i != nei; ++i)
242  hi_elem->set_extra_integer(i, lo_elem.get_extra_integer(i));
243 
244  // This might not help anything but shouldn't hurt.
245  hi_elem->set_mapping_type(lo_elem.mapping_type());
246  hi_elem->set_mapping_data(lo_elem.mapping_data());
247 
248  mesh.insert_elem(std::move(hi_elem));
249 }

311 void UnstructuredMesh::copy_nodes_and_elements(const MeshBase & other_mesh,
312  const bool skip_find_neighbors,
313  dof_id_type element_id_offset,
314  dof_id_type node_id_offset,
315  unique_id_type
316 #ifdef LIBMESH_ENABLE_UNIQUE_ID
317  unique_id_offset
318 #endif
319  ,
320  std::unordered_map<subdomain_id_type, subdomain_id_type> *
321  id_remapping)
322 {
323  LOG_SCOPE("copy_nodes_and_elements()", "UnstructuredMesh");
324 
325  std::pair<std::vector<unsigned int>, std::vector<unsigned int>>
326  extra_int_maps = this->merge_extra_integer_names(other_mesh);
327 
328  const unsigned int n_old_node_ints = extra_int_maps.second.size(),
329  n_new_node_ints = _node_integer_names.size(),
330  n_old_elem_ints = extra_int_maps.first.size(),
331  n_new_elem_ints = _elem_integer_names.size();
332 
333  // If we are partitioned into fewer parts than the incoming mesh,
334  // then we need to "wrap" the other Mesh's processor ids to fit
335  // within our range. This can happen, for example, while stitching
336  // meshes with small numbers of elements in parallel...
337  bool wrap_proc_ids = (this->n_partitions() <
338  other_mesh.n_partitions());
339 
340  // We're assuming the other mesh has proper element number ordering,
341  // so that we add parents before their children, and that the other
342  // mesh is consistently partitioned.
343 #ifdef DEBUG
344  MeshTools::libmesh_assert_valid_amr_elem_ids(other_mesh);
345  MeshTools::libmesh_assert_valid_procids<Node>(other_mesh);
346 #endif
347 
348  //Copy in Nodes
349  {
350  //Preallocate Memory if necessary
351  this->reserve_nodes(other_mesh.n_nodes());
352 
353  for (const auto & oldn : other_mesh.node_ptr_range())
354  {
355  processor_id_type added_pid = cast_int<processor_id_type>
356  (wrap_proc_ids ? oldn->processor_id() % _n_parts : oldn->processor_id());
357 
358  // Add new nodes in old node Point locations
359  Node * newn =
360  this->add_point(*oldn,
361  oldn->id() + node_id_offset,
362  added_pid);
363 
364  newn->add_extra_integers(n_new_node_ints);
365  for (unsigned int i = 0; i != n_old_node_ints; ++i)
366  newn->set_extra_integer(extra_int_maps.second[i],
367  oldn->get_extra_integer(i));
368 
369 #ifdef LIBMESH_ENABLE_UNIQUE_ID
370  newn->set_unique_id(oldn->unique_id() + unique_id_offset);
371 #endif
372  }
373  }
374 
375  //Copy in Elements
376  {
377  //Preallocate Memory if necessary
378  this->reserve_elem(other_mesh.n_elem());
379 
380  // Declare a map linking old and new elements, needed to copy the neighbor lists
381  typedef std::unordered_map<const Elem *, Elem *> map_type;
382  map_type old_elems_to_new_elems, ip_map;
383 
384  // Loop over the elements
385  for (const auto & old : other_mesh.element_ptr_range())
386  {
387  // Build a new element
388  Elem * newparent = old->parent() ?
389  this->elem_ptr(old->parent()->id() + element_id_offset) :
390  nullptr;
391  auto el = Elem::build(old->type(), newparent);
392 
393  subdomain_id_type sbd_id = old->subdomain_id();
394  if (id_remapping)
395  {
396  auto remapping_it = id_remapping->find(sbd_id);
397  if (remapping_it != id_remapping->end())
398  sbd_id = remapping_it->second;
399  }
400  el->subdomain_id() = sbd_id;
401 
402  // Hold off on trying to set the interior parent because we may actually
403  // add lower dimensional elements before their interior parents
404  if (el->dim() < other_mesh.mesh_dimension() && old->interior_parent())
405  ip_map[old] = el.get();
406 
407 #ifdef LIBMESH_ENABLE_AMR
408  if (old->has_children())
409  for (unsigned int c = 0, nc = old->n_children(); c != nc; ++c)
410  if (old->child_ptr(c) == remote_elem)
411  el->add_child(const_cast<RemoteElem *>(remote_elem), c);
412 
413  //Create the parent's child pointers if necessary
414  if (newparent)
415  {
416  unsigned int oldc = old->parent()->which_child_am_i(old);
417  newparent->add_child(el.get(), oldc);
418  }
419 
420  // Copy the refinement flags
421  el->set_refinement_flag(old->refinement_flag());
422 
423  // Use hack_p_level since we may not have sibling elements
424  // added yet
425  el->hack_p_level(old->p_level());
426 
427  el->set_p_refinement_flag(old->p_refinement_flag());
428 #endif // #ifdef LIBMESH_ENABLE_AMR
429 
430  //Assign all the nodes
431  for (auto i : el->node_index_range())
432  el->set_node(i) =
433  this->node_ptr(old->node_id(i) + node_id_offset);
434 
435  // And start it off with the same processor id (mod _n_parts).
436  el->processor_id() = cast_int<processor_id_type>
437  (wrap_proc_ids ? old->processor_id() % _n_parts : old->processor_id());
438 
439  // Give it the same element and unique ids
440  el->set_id(old->id() + element_id_offset);
441 
442  el->add_extra_integers(n_new_elem_ints);
443  for (unsigned int i = 0; i != n_old_elem_ints; ++i)
444  el->set_extra_integer(extra_int_maps.first[i],
445  old->get_extra_integer(i));
446 
447 #ifdef LIBMESH_ENABLE_UNIQUE_ID
448  el->set_unique_id(old->unique_id() + unique_id_offset);
449 #endif
450 
451  el->set_mapping_type(old->mapping_type());
452  el->set_mapping_data(old->mapping_data());
453 
454  //Hold onto it
455  if (!skip_find_neighbors)
456  {
457  for (auto s : old->side_index_range())
458  if (old->neighbor_ptr(s) == remote_elem)
459  el->set_neighbor(s, const_cast<RemoteElem *>(remote_elem));
460  this->add_elem(std::move(el));
461  }
462  else
463  {
464  Elem * new_el = this->add_elem(std::move(el));
465  old_elems_to_new_elems[old] = new_el;
466  }
467  }
468 
469  for (auto & elem_pair : ip_map)
470  elem_pair.second->set_interior_parent(
471  this->elem_ptr(elem_pair.first->interior_parent()->id() + element_id_offset));
472 
473  // Loop (again) over the elements to fill in the neighbors
474  if (skip_find_neighbors)
475  {
476  old_elems_to_new_elems[remote_elem] = const_cast<RemoteElem*>(remote_elem);
477 
478  for (const auto & old_elem : other_mesh.element_ptr_range())
479  {
480  Elem * new_elem = old_elems_to_new_elems[old_elem];
481  for (auto s : old_elem->side_index_range())
482  {
483  const Elem * old_neighbor = old_elem->neighbor_ptr(s);
484  Elem * new_neighbor = old_elems_to_new_elems[old_neighbor];
485  new_elem->set_neighbor(s, new_neighbor);
486  }
487  }
488  }
489  }
490 
491 #ifdef LIBMESH_ENABLE_UNIQUE_ID
492  // We set the unique ids of nodes after adding them to the mesh such that our value of
493  // _next_unique_id may be wrong. So we amend that here
494  this->set_next_unique_id(other_mesh.parallel_max_unique_id() + unique_id_offset + 1);
495 #endif
496 
497  //Finally prepare the new Mesh for use. Keep the same numbering and
498  //partitioning for now.
499  this->allow_renumbering(false);
500  this->allow_remote_element_removal(false);
501  this->allow_find_neighbors(!skip_find_neighbors);
502 
503  // We should generally be able to skip *all* partitioning here
504  // because we're only adding one already-consistent mesh to another.
505  this->skip_partitioning(true);
506 
507  this->prepare_for_use();
508 
509  //But in the long term, use the same renumbering and partitioning
510  //policies as our source mesh.
511  this->allow_find_neighbors(other_mesh.allow_find_neighbors());
512  this->allow_renumbering(other_mesh.allow_renumbering());
513  this->allow_remote_element_removal(other_mesh.allow_remote_element_removal());
514  this->skip_partitioning(other_mesh.skip_partitioning());
515 }

530 void UnstructuredMesh::find_neighbors (const bool reset_remote_elements,
531  const bool reset_current_list)
532 {
533  // We might actually want to run this on an empty mesh
534  // (e.g. the boundary mesh for a nonexistent bcid!)
535  // libmesh_assert_not_equal_to (this->n_nodes(), 0);
536  // libmesh_assert_not_equal_to (this->n_elem(), 0);
537 
538  // This function must be run on all processors at once
539  parallel_object_only();
540 
541  LOG_SCOPE("find_neighbors()", "Mesh");
542 
543  //TODO:[BSK] This should be removed later?!
544  if (reset_current_list)
545  for (const auto & e : this->element_ptr_range())
546  for (auto s : e->side_index_range())
547  if (e->neighbor_ptr(s) != remote_elem || reset_remote_elements)
548  e->set_neighbor(s, nullptr);
549 
550  // Find neighboring elements by first finding elements
551  // with identical side keys and then check to see if they
552  // are neighbors
553  {
554  // data structures -- Use the hash_multimap if available
555  typedef dof_id_type key_type;
556  typedef std::pair<Elem *, unsigned char> val_type;
557  typedef std::unordered_multimap<key_type, val_type> map_type;
558 
559  // A map from side keys to corresponding elements & side numbers
560  map_type side_to_elem_map;
561 
562  // Pull objects out of the loop to reduce heap operations
563  std::unique_ptr<Elem> my_side, their_side;
564 
565  for (const auto & element : this->element_ptr_range())
566  {
567  for (auto ms : element->side_index_range())
568  {
569  next_side:
570  // If we haven't yet found a neighbor on this side, try.
571  // Even if we think our neighbor is remote, that
572  // information may be out of date.
573  if (element->neighbor_ptr(ms) == nullptr ||
574  element->neighbor_ptr(ms) == remote_elem)
575  {
576  // Get the key for the side of this element
577  const dof_id_type key = element->key(ms);
578 
579  // Look for elements that have an identical side key
580  auto bounds = side_to_elem_map.equal_range(key);
581 
582  // May be multiple keys, check all the possible
583  // elements which _might_ be neighbors.
584  if (bounds.first != bounds.second)
585  {
586  // Get the side for this element
587  element->side_ptr(my_side, ms);
588 
589  // Look at all the entries with an equivalent key
590  while (bounds.first != bounds.second)
591  {
592  // Get the potential element
593  Elem * neighbor = bounds.first->second.first;
594 
595  // Get the side for the neighboring element
596  const unsigned int ns = bounds.first->second.second;
597  neighbor->side_ptr(their_side, ns);
598  //libmesh_assert(my_side.get());
599  //libmesh_assert(their_side.get());
600 
601  // If found a match with my side
602  //
603  // In 1D, since parents and children have an
604  // equal side (i.e. a node) we need to check
605  // for matching level() to avoid setting our
606  // neighbor pointer to any of our neighbor's
607  // descendants.
608  if ((*my_side == *their_side) &&
609  (element->level() == neighbor->level()))
610  {
611  // So share a side. Is this a mixed pair
612  // of subactive and active/ancestor
613  // elements?
614  // If not, then we're neighbors.
615  // If so, then the subactive's neighbor is
616 
617  if (element->subactive() ==
618  neighbor->subactive())
619  {
620  // an element is only subactive if it has
621  // been coarsened but not deleted
622  element->set_neighbor (ms,neighbor);
623  neighbor->set_neighbor(ns,element);
624  }
625  else if (element->subactive())
626  {
627  element->set_neighbor(ms,neighbor);
628  }
629  else if (neighbor->subactive())
630  {
631  neighbor->set_neighbor(ns,element);
632  }
633  side_to_elem_map.erase (bounds.first);
634 
635  // get out of this nested crap
636  goto next_side;
637  }
638 
639  ++bounds.first;
640  }
641  }
642 
643  // didn't find a match...
644  // Build the map entry for this element
645  side_to_elem_map.emplace
646  (key, std::make_pair(element, cast_int<unsigned char>(ms)));
647  }
648  }
649  }
650  }

653 
681  const unsigned int n_levels = MeshTools::n_levels(*this);
682  for (unsigned int level = 1; level < n_levels; ++level)
683  {
684  for (auto & current_elem : as_range(level_elements_begin(level),
685  level_elements_end(level)))
686  {
687  libmesh_assert(current_elem);
688  Elem * parent = current_elem->parent();
689  libmesh_assert(parent);
690  const unsigned int my_child_num = parent->which_child_am_i(current_elem);
691 
692  for (auto s : current_elem->side_index_range())
693  {
694  if (current_elem->neighbor_ptr(s) == nullptr ||
695  (current_elem->neighbor_ptr(s) == remote_elem &&
696  parent->is_child_on_side(my_child_num, s)))
697  {
698  Elem * neigh = parent->neighbor_ptr(s);
699 
700  // If neigh was refined and had non-subactive children
701  // made remote earlier, then our current elem should
702  // actually have one of those remote children as a
703  // neighbor
704  if (neigh &&
705  (neigh->ancestor() ||
706  // If neigh has subactive children which should have
707  // matched as neighbors of the current element but
708  // did not, then those likewise must be remote
709  // children.
710  (current_elem->subactive() && neigh->has_children() &&
711  (neigh->level()+1) == current_elem->level())))
712  {
713 #ifdef DEBUG
714  // Let's make sure that "had children made remote"
715  // situation is actually the case
716  libmesh_assert(neigh->has_children());
717  bool neigh_has_remote_children = false;
718  for (auto & child : neigh->child_ref_range())
719  if (&child == remote_elem)
720  neigh_has_remote_children = true;
721  libmesh_assert(neigh_has_remote_children);
722 
723  // And let's double-check that we don't have
724  // a remote_elem neighboring an active local element
725  if (current_elem->active())
726  libmesh_assert_not_equal_to (current_elem->processor_id(),
727  this->processor_id());
728 #endif // DEBUG
729  neigh = const_cast<RemoteElem *>(remote_elem);
730  }
731  // If neigh and current_elem are more than one level
732  // apart, figuring out whether we have a remote
733  // neighbor here becomes much harder.
734  else if (neigh && (current_elem->subactive() &&
735  neigh->has_children()))
736  {
737  // Find the deepest descendant of neigh which
738  // we could consider for a neighbor. If we run
739  // out of neigh children, then that's our
740  // neighbor. If we find a potential neighbor
741  // with remote_children and we don't find any
742  // potential neighbors among its non-remote
743  // children, then our neighbor must be remote.
744  while (neigh != remote_elem &&
745  neigh->has_children())
746  {
747  bool found_neigh = false;
748  for (unsigned int c = 0, nc = neigh->n_children();
749  !found_neigh && c != nc; ++c)
750  {
751  Elem * child = neigh->child_ptr(c);
752  if (child == remote_elem)
753  continue;
754  for (auto ncn : child->neighbor_ptr_range())
755  {
756  if (ncn != remote_elem &&
757  ncn->is_ancestor_of(current_elem))
758  {
759  neigh = ncn;
760  found_neigh = true;
761  break;
762  }
763  }
764  }
765  if (!found_neigh)
766  neigh = const_cast<RemoteElem *>(remote_elem);
767  }
768  }
769  current_elem->set_neighbor(s, neigh);
770 #ifdef DEBUG
771  if (neigh != nullptr && neigh != remote_elem)
772  // We ignore subactive elements here because
773  // we don't care about neighbors of subactive element.
774  if ((!neigh->active()) && (!current_elem->subactive()))
775  {
776  libMesh::err << "On processor " << this->processor_id()
777  << std::endl;
778  libMesh::err << "Bad element ID = " << current_elem->id()
779  << ", Side " << s << ", Bad neighbor ID = " << neigh->id() << std::endl;
780  libMesh::err << "Bad element proc_ID = " << current_elem->processor_id()
781  << ", Bad neighbor proc_ID = " << neigh->processor_id() << std::endl;
782  libMesh::err << "Bad element size = " << current_elem->hmin()
783  << ", Bad neighbor size = " << neigh->hmin() << std::endl;
784  libMesh::err << "Bad element center = " << current_elem->vertex_average()
785  << ", Bad neighbor center = " << neigh->vertex_average() << std::endl;
786  libMesh::err << "ERROR: "
787  << (current_elem->active()?"Active":"Ancestor")
788  << " Element at level "
789  << current_elem->level() << std::endl;
790  libMesh::err << "with "
791  << (parent->active()?"active":
792  (parent->subactive()?"subactive":"ancestor"))
793  << " parent share "
794  << (neigh->subactive()?"subactive":"ancestor")
795  << " neighbor at level " << neigh->level()
796  << std::endl;
797  NameBasedIO(*this).write ("bad_mesh.gmv");
798  libmesh_error_msg("Problematic mesh written to bad_mesh.gmv.");
799  }
800 #endif // DEBUG
801  }
802  }
803 
804  // We can skip to the next element if we're full-dimension
805  // and therefore don't have any interior parents
806  if (current_elem->dim() >= LIBMESH_DIM)
807  continue;
808 
809  // We have no interior parents unless we can find one later
810  current_elem->set_interior_parent(nullptr);
811 
812  Elem * pip = parent->interior_parent();
813 
814  if (!pip)
815  continue;
816 
817  // If there's no interior_parent children, whether due to a
818  // remote element or a non-conformity, then there's no
819  // children to search.
820  if (pip == remote_elem || pip->active())
821  {
822  current_elem->set_interior_parent(pip);
823  continue;
824  }
825 
826  // For node comparisons we'll need a sensible tolerance
827  Real node_tolerance = current_elem->hmin() * TOLERANCE;
828 
829  // Otherwise our interior_parent should be a child of our
830  // parent's interior_parent.
831  for (auto & child : pip->child_ref_range())
832  {
833  // If we have a remote_elem, that might be our
834  // interior_parent. We'll set it provisionally now and
835  // keep trying to find something better.
836  if (&child == remote_elem)
837  {
838  current_elem->set_interior_parent
839  (const_cast<RemoteElem *>(remote_elem));
840  continue;
841  }
842 
843  bool child_contains_our_nodes = true;
844  for (auto & n : current_elem->node_ref_range())
845  {
846  bool child_contains_this_node = false;
847  for (auto & cn : child.node_ref_range())
848  if (cn.absolute_fuzzy_equals
849  (n, node_tolerance))
850  {
851  child_contains_this_node = true;
852  break;
853  }
854  if (!child_contains_this_node)
855  {
856  child_contains_our_nodes = false;
857  break;
858  }
859  }
860  if (child_contains_our_nodes)
861  {
862  current_elem->set_interior_parent(&child);
863  break;
864  }
865  }
866 
867  // We should have found *some* interior_parent at this
868  // point, whether semilocal or remote.
869  libmesh_assert(current_elem->interior_parent());
870  }
871  }
872 
873 #endif // AMR
874 
875 
876 #ifdef DEBUG
877  MeshTools::libmesh_assert_valid_neighbors(*this,
878  !reset_remote_elements);
879  MeshTools::libmesh_assert_valid_amr_interior_parents(*this);
880 #endif
881 }

1130 void UnstructuredMesh::all_first_order ()
1131 {
1132  LOG_SCOPE("all_first_order()", "Mesh");
1133 
1134  /*
1135  * when the mesh is not prepared,
1136  * at least renumber the nodes and
1137  * elements, so that the node ids
1138  * are correct
1139  */
1140  if (!this->_is_prepared)
1141  this->renumber_nodes_and_elements ();
1142 
1146  std::vector<bool> node_touched_by_me(this->max_node_id(), false);
1147 
1148  // Loop over the high-ordered elements.
1149  // First make sure they _are_ indeed high-order, and then replace
1150  // them with an equivalent first-order element.
1151  for (auto & so_elem : element_ptr_range())
1152  {
1153  libmesh_assert(so_elem);
1154 
1155  /*
1156  * build the first-order equivalent, add to
1157  * the new_elements list.
1158  */
1159  auto lo_elem = Elem::build
1160  (Elem::first_order_equivalent_type
1161  (so_elem->type()), so_elem->parent());
1162 
1163  const unsigned short n_sides = so_elem->n_sides();
1164 
1165  for (unsigned short s=0; s != n_sides; ++s)
1166  if (so_elem->neighbor_ptr(s) == remote_elem)
1167  lo_elem->set_neighbor(s, const_cast<RemoteElem *>(remote_elem));
1168 
1169 #ifdef LIBMESH_ENABLE_AMR
1170  /*
1171  * Reset the parent links of any child elements
1172  */
1173  if (so_elem->has_children())
1174  for (unsigned int c = 0, nc = so_elem->n_children(); c != nc; ++c)
1175  {
1176  Elem * child = so_elem->child_ptr(c);
1177  if (child != remote_elem)
1178  child->set_parent(lo_elem.get());
1179  lo_elem->add_child(child, c);
1180  }
1181 
1182  /*
1183  * Reset the child link of any parent element
1184  */
1185  if (so_elem->parent())
1186  {
1187  unsigned int c =
1188  so_elem->parent()->which_child_am_i(so_elem);
1189  lo_elem->parent()->replace_child(lo_elem.get(), c);
1190  }
1191 
1192  /*
1193  * Copy as much data to the new element as makes sense
1194  */
1195  lo_elem->set_p_level(so_elem->p_level());
1196  lo_elem->set_refinement_flag(so_elem->refinement_flag());
1197  lo_elem->set_p_refinement_flag(so_elem->p_refinement_flag());
1198 #endif
1199 
1200  libmesh_assert_equal_to (lo_elem->n_vertices(), so_elem->n_vertices());
1201 
1202  /*
1203  * By definition the vertices of the linear and
1204  * second order element are identically numbered.
1205  * transfer these.
1206  */
1207  for (unsigned int v=0, snv=so_elem->n_vertices(); v < snv; v++)
1208  {
1209  lo_elem->set_node(v) = so_elem->node_ptr(v);
1210  node_touched_by_me[lo_elem->node_id(v)] = true;
1211  }
1212 
1213  /*
1214  * find_neighbors relies on remote_elem neighbor links being
1215  * properly maintained.
1216  */
1217  for (unsigned short s=0; s != n_sides; s++)
1218  {
1219  if (so_elem->neighbor_ptr(s) == remote_elem)
1220  lo_elem->set_neighbor(s, const_cast<RemoteElem*>(remote_elem));
1221  }
1222 
1229  this->get_boundary_info().copy_boundary_ids
1230  (this->get_boundary_info(), so_elem, lo_elem.get());
1231 
1232  /*
1233  * The new first-order element is ready.
1234  * Inserting it into the mesh will replace and delete
1235  * the second-order element.
1236  */
1237  lo_elem->set_id(so_elem->id());
1238 #ifdef LIBMESH_ENABLE_UNIQUE_ID
1239  lo_elem->set_unique_id(so_elem->unique_id());
1240 #endif
1241  lo_elem->processor_id() = so_elem->processor_id();
1242  lo_elem->subdomain_id() = so_elem->subdomain_id();
1243 
1244  const unsigned int nei = so_elem->n_extra_integers();
1245  lo_elem->add_extra_integers(nei);
1246  for (unsigned int i=0; i != nei; ++i)
1247  lo_elem->set_extra_integer(i, so_elem->get_extra_integer(i));
1248 
1249  // This is probably moot but shouldn't hurt
1250  lo_elem->set_mapping_type(so_elem->mapping_type());
1251  lo_elem->set_mapping_data(so_elem->mapping_data());
1252 
1253  this->insert_elem(std::move(lo_elem));
1254  }
1255 
1256  // Deleting nodes does not invalidate iterators, so this is safe.
1257  for (const auto & node : this->node_ptr_range())
1258  if (!node_touched_by_me[node->id()])
1259  this->delete_node(node);
1260 
1261  // If crazy people applied boundary info to non-vertices and then
1262  // deleted those non-vertices, we should make sure their boundary id
1263  // caches are correct.
1264  this->get_boundary_info().regenerate_id_sets();
1265 
1266  // On hanging nodes that used to also be second order nodes, we
1267  // might now have an invalid nodal processor_id()
1268  Partitioner::set_node_processor_ids(*this);
1269 
1270  // delete or renumber nodes if desired
1271  this->prepare_for_use();
1272 }

1277 UnstructuredMesh::all_second_order_range (const SimpleRange<element_iterator> & range,
1278  const bool full_ordered)
1279 {
1280  // This function must be run on all processors at once
1281  parallel_object_only();
1282 
1283  LOG_SCOPE("all_second_order_range()", "Mesh");
1284 
1285  /*
1286  * when the mesh is not prepared,
1287  * at least renumber the nodes and
1288  * elements, so that the node ids
1289  * are correct
1290  */
1291  if (!this->_is_prepared)
1292  this->renumber_nodes_and_elements ();
1293 
1294  /*
1295  * If the mesh is empty
1296  * then we have nothing to do
1297  */
1298  if (!this->n_elem())
1299  return;
1300 
1301  // If every element in the range _on every proc_ is already second
1302  // order then we have nothing to do. However, if any proc has some
1303  // first-order elements in the range, then _all_ processors need to
1304  // continue this function because it is parallel_only(). Note:
1305  // std::all_of() returns true for an empty range, which can happen
1306  // for example in the DistributedMesh case when there are more
1307  // processors than elements. In the case of an empty range we
1308  // therefore set already_second_order to true on that proc.
1309  bool already_second_order =
1310  std::all_of(range.begin(), range.end(),
1311  [](const auto & elem){ return elem->default_order() == SECOND; });
1312

1314  this->comm().min(already_second_order);
1315  if (already_second_order)
1316  return;
1317 
1318  /*
1319  * this map helps in identifying second order
1320  * nodes. Namely, a second-order node:
1321  * - edge node
1322  * - face node
1323  * - bubble node
1324  * is uniquely defined through a set of adjacent
1325  * vertices. This set of adjacent vertices is
1326  * used to identify already added higher-order
1327  * nodes. We are safe to use node id's since we
1328  * make sure that these are correctly numbered.
1329  */
1330  std::map<std::vector<dof_id_type>, Node *> adj_vertices_to_so_nodes;
1331 
1332  /*
1333  * The maximum number of new second order nodes we might be adding,
1334  * for use when picking unique unique_id values later. This variable
1335  * is not used unless unique ids are enabled.
1336  */
1337 #ifdef LIBMESH_ENABLE_UNIQUE_ID
1338  unsigned int max_new_nodes_per_elem;
1339
