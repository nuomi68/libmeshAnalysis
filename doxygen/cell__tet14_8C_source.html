155 bool Tet14::has_affine_map() const
156 {
157  // Make sure edges are straight
158  Point v10 = this->point(1) - this->point(0);
159  if (!v10.relative_fuzzy_equals
160  ((this->point(4) - this->point(0))*2))
161  return false;
162  Point v21 = this->point(2) - this->point(1);
163  if (!v21.relative_fuzzy_equals
164  ((this->point(5) - this->point(1))*2))
165  return false;
166  Point v20 = this->point(2) - this->point(0);
167  if (!v20.relative_fuzzy_equals
168  ((this->point(6) - this->point(0))*2))
169  return false;
170  Point v30 = this->point(3) - this->point(0);
171  if (!v30.relative_fuzzy_equals
172  ((this->point(7) - this->point(0))*2))
173  return false;
174  Point v31 = this->point(3) - this->point(1);
175  if (!v31.relative_fuzzy_equals
176  ((this->point(8) - this->point(1))*2))
177  return false;
178  Point v32 = this->point(3) - this->point(2);
179  if (!v32.relative_fuzzy_equals
180  ((this->point(9) - this->point(2))*2))
181  return false;
182 
183  // Make sure midface nodes are midface
184  if (!(v20+v10).relative_fuzzy_equals
185  ((this->point(10) - this->point(0))*3))
186  return false;
187  if (!(v30+v10).relative_fuzzy_equals
188  ((this->point(11) - this->point(0))*3))
189  return false;
190  if (!(v31+v21).relative_fuzzy_equals
191  ((this->point(12) - this->point(1))*3))
192  return false;
193  if (!(v30+v20).relative_fuzzy_equals
194  ((this->point(13) - this->point(0))*3))
195  return false;
196 
197  return true;
198 }

261 void Tet14::connectivity(const unsigned int sc,
262  const IOPackage iop,
263  std::vector<dof_id_type> & conn) const
264 {
265  libmesh_assert(_nodes);
266  libmesh_assert_less (sc, this->n_sub_elem());
267  libmesh_assert_not_equal_to (iop, INVALID_IO_PACKAGE);
268 
269  switch (iop)
270  {
271  case TECPLOT:
272  {
273  conn.resize(8);
274  switch (sc)
275  {
276 
277 
278  // Linear sub-tet 0
279  case 0:
280 
281  conn[0] = this->node_id(0)+1;
282  conn[1] = this->node_id(4)+1;
283  conn[2] = this->node_id(6)+1;
284  conn[3] = this->node_id(6)+1;
285  conn[4] = this->node_id(7)+1;
286  conn[5] = this->node_id(7)+1;
287  conn[6] = this->node_id(7)+1;
288  conn[7] = this->node_id(7)+1;
289 
290  return;
291 
292  // Linear sub-tet 1
293  case 1:
294 
295  conn[0] = this->node_id(4)+1;
296  conn[1] = this->node_id(1)+1;
297  conn[2] = this->node_id(5)+1;
298  conn[3] = this->node_id(5)+1;
299  conn[4] = this->node_id(8)+1;
300  conn[5] = this->node_id(8)+1;
301  conn[6] = this->node_id(8)+1;
302  conn[7] = this->node_id(8)+1;
303 
304  return;
305 
306  // Linear sub-tet 2
307  case 2:
308 
309  conn[0] = this->node_id(5)+1;
310  conn[1] = this->node_id(2)+1;
311  conn[2] = this->node_id(6)+1;
312  conn[3] = this->node_id(6)+1;
313  conn[4] = this->node_id(9)+1;
314  conn[5] = this->node_id(9)+1;
315  conn[6] = this->node_id(9)+1;
316  conn[7] = this->node_id(9)+1;
317 
318  return;
319 
320  // Linear sub-tet 3
321  case 3:
322 
323  conn[0] = this->node_id(7)+1;
324  conn[1] = this->node_id(8)+1;
325  conn[2] = this->node_id(9)+1;
326  conn[3] = this->node_id(9)+1;
327  conn[4] = this->node_id(3)+1;
328  conn[5] = this->node_id(3)+1;
329  conn[6] = this->node_id(3)+1;
330  conn[7] = this->node_id(3)+1;
331 
332  return;
333 
334  // Linear sub-tet 4
335  case 4:
336 
337  conn[0] = this->node_id(4)+1;
338  conn[1] = this->node_id(8)+1;
339  conn[2] = this->node_id(6)+1;
340  conn[3] = this->node_id(6)+1;
341  conn[4] = this->node_id(7)+1;
342  conn[5] = this->node_id(7)+1;
343  conn[6] = this->node_id(7)+1;
344  conn[7] = this->node_id(7)+1;
345 
346  return;
347 
348  // Linear sub-tet 5
349  case 5:
350 
351  conn[0] = this->node_id(4)+1;
352  conn[1] = this->node_id(5)+1;
353  conn[2] = this->node_id(6)+1;
354  conn[3] = this->node_id(6)+1;
355  conn[4] = this->node_id(8)+1;
356  conn[5] = this->node_id(8)+1;
357  conn[6] = this->node_id(8)+1;
358  conn[7] = this->node_id(8)+1;
359 
360  return;
361 
362  // Linear sub-tet 6
363  case 6:
364 
365  conn[0] = this->node_id(5)+1;
366  conn[1] = this->node_id(9)+1;
367  conn[2] = this->node_id(6)+1;
368  conn[3] = this->node_id(6)+1;
369  conn[4] = this->node_id(8)+1;
370  conn[5] = this->node_id(8)+1;
371  conn[6] = this->node_id(8)+1;
372  conn[7] = this->node_id(8)+1;
373 
374  return;
375 
376  // Linear sub-tet 7
377  case 7:
378 
379  conn[0] = this->node_id(7)+1;
380  conn[1] = this->node_id(6)+1;
381  conn[2] = this->node_id(9)+1;
382  conn[3] = this->node_id(9)+1;
383  conn[4] = this->node_id(8)+1;
384  conn[5] = this->node_id(8)+1;
385  conn[6] = this->node_id(8)+1;
386  conn[7] = this->node_id(8)+1;
387 
388  return;
389 
390 
391  default:
392  libmesh_error_msg("Invalid sc = " << sc);
393  }
394  }
395 
396  case VTK:
397  {
398  // VTK has vtkHigherOrderTetra which might have the same
399  // connectivity as our Tet14, but this has not been tested
400  // yet.
401  libmesh_experimental();
402  conn.resize(Tet14::num_nodes);
403  for (auto i : index_range(conn))
404  conn[i] = this->node_id(i);
405  return;
406  }
407 
408  default:
409  libmesh_error_msg("Unsupported IO package " << iop);
410  }

660 Real Tet14::embedding_matrix (const unsigned int i,
661  const unsigned int j,
662  const unsigned int k) const
663 {
664  // Choose an optimal diagonal, if one has not already been selected
665  this->choose_diagonal();
666 
667  // Permuted j and k indices
668  unsigned int
669  jp=j,
670  kp=k;
671 
672  if ((i>3) && (this->_diagonal_selection!=DIAG_02_13))
673  {
674  // Just the enum value cast to an unsigned int...
675  const unsigned ds = static_cast<unsigned int>(this->_diagonal_selection); // == 1 or 2
676 
677  // Instead of doing a lot of arithmetic, use these
678  // straightforward arrays for the permutations. Note that 3 ->
679  // 3 and 10 -> 10, and the first array consists of "forward"
680  // permutations of the sets {0,1,2}, {4,5,6}, {7,8,9}, and
681  // {11,12,13} while the second array consists of "reverse"
682  // permutations of the same sets.
683 
684  const unsigned int perms[2][14] =
685  {
686  {1, 2, 0, 3, 5, 6, 4, 8, 9, 7, 10, 12, 13, 11},
687  {2, 0, 1, 3, 6, 4, 5, 9, 7, 8, 10, 13, 11, 12}
688  };
689 
690  // Permute j
691  jp = perms[ds-1][j];
692  // if (jp<3)
693  // jp = (jp+ds)%3;
694  // else if (jp>3)
695  // jp = (jp-1+ds)%3 + 1 + 3*((jp-1)/3);
696 
697  // Permute k
698  kp = perms[ds-1][k];
699  // if (kp<3)
700  // kp = (kp+ds)%3;
701  // else if (kp>3)
702  // kp = (kp-1+ds)%3 + 1 + 3*((kp-1)/3);
703  }
704 
705  // Debugging:
706  // libMesh::err << "Selected diagonal " << _diagonal_selection << std::endl;
707  // libMesh::err << "j=" << j << std::endl;
708  // libMesh::err << "k=" << k << std::endl;
709  // libMesh::err << "jp=" << jp << std::endl;
710  // libMesh::err << "kp=" << kp << std::endl;
711 
712  // Call embedding matrix with permuted indices
713  return this->_embedding_matrix[i][jp][kp];
714 }

747 void Tet14::permute(unsigned int perm_num)
748 {
749  libmesh_assert_less (perm_num, 12);
750 
751  const unsigned int side = perm_num % 4;
752  const unsigned int rotate = perm_num / 4;
753 
754  for (unsigned int i = 0; i != rotate; ++i)
755  {
756  swap3nodes(0,1,2);
757  swap3nodes(4,5,6);
758  swap3nodes(7,8,9);
759  swap3nodes(11,12,13);
760  swap3neighbors(1,2,3);
761  }
762 
763  switch (side) {
764  case 0:
765  break;
766  case 1:
767  swap3nodes(0,2,3);
768  swap3nodes(4,5,8);
769  swap3nodes(6,9,7);
770  swap3nodes(10,12,11);
771  swap3neighbors(0,2,1);
772  break;
773  case 2:
774  swap3nodes(2,0,3);
775  swap3nodes(5,4,8);
776  swap3nodes(6,7,9);
777  swap3nodes(10,11,12);
778  swap3neighbors(0,1,2);
779  break;
780  case 3:
781  swap3nodes(2,1,3);
782  swap3nodes(5,8,9);
783  swap3nodes(6,4,7);
784  swap3nodes(10,11,13);
785  swap3neighbors(0,1,3);
786  break;
787  default:
788  libmesh_error();
789  }
