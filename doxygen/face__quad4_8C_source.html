142 bool Quad4::has_invertible_map(Real tol) const
143 {
144  // At the moment this only makes sense for Lagrange elements
145  libmesh_assert_equal_to(this->mapping_type(), LAGRANGE_MAP);
146 
147  // Side vectors
148  Point s0 = this->point(1) - this->point(0);
149  Point s1 = this->point(2) - this->point(1);
150  Point s2 = this->point(3) - this->point(2);
151  Point s3 = this->point(0) - this->point(3);
152 
153  // Cross products of side vectors
154  Point v1 = s3.cross(s0);
155  Point v2 = s2.cross(s0);
156  Point v3 = s3.cross(s1);
157 
158  // A unit vector in the direction of:
159  // f(xi, eta) = (v1 + xi*v2 + eta*v3)
160  // at the midpoint (xi, eta) = (1/2, 1/2) of the element. (Note that
161  // we are using the [0,1]^2 reference element definition for the
162  // Quad4 instead of the [-1,1]^2 reference element that is typically
163  // used for FEM calculations.) We use this as a "reference" vector
164  // and compare the sign of dot(n,f) at each vertex.
165  Point n = v1 + Real(.5) * (v2 + v3);
166  Real norm_n = n.norm();
167 
168  // If the Jacobian vector at the midpoint of the element is zero,
169  // then it must be either zero for the entire element, or change
170  // sign at the vertices. Either way the element is non-invertible.
171  if (norm_n <= tol)
172  return false;
173 
174  n /= norm_n;
175 
176  // Debugging
177  // std::cout << "n=" << n << std::endl;
178 
179  // Compute scalar quantity n * (v1 + xi*v2 + eta*v3) at each
180  // vertex. If it is non-zero and has the same sign at each
181  // vertex, the the element is invertible, otherwise it is not.
182  std::array<Real, 4> vertex_vals;
183  unsigned int ctr = 0;
184  for (unsigned int i=0; i<2; ++i)
185  for (unsigned int j=0; j<2; ++j)
186  vertex_vals[ctr++] = n * (v1 + Real(i)*v2 + Real(j)*v3);
187 
188  // Debugging:
189  // std::cout << "Vertex values: ";
190  // for (const auto & val : vertex_vals)
191  // std::cout << val << " ";
192  // std::cout << std::endl;
193 
194  auto result = std::minmax_element(vertex_vals.begin(), vertex_vals.end());
195  Real min_vertex = *(result.first);
196  Real max_vertex = *(result.second);
197 
198  // Debugging
199  // std::cout << "min_vertex=" << min_vertex << std::endl;
200  // std::cout << "max_vertex=" << max_vertex << std::endl;
201 
202  // If max and min are both on the same side of 0, we are invertible, otherwise we are not.
203  return ((max_vertex > 0 && min_vertex > 0) || (max_vertex < 0 && min_vertex < 0));
204 }

232  const IOPackage iop,

268 Point Quad4::true_centroid () const
269 {
270  // Convenient references to our points
271  const Point
272  &x0 = point(0), &x1 = point(1),
273  &x2 = point(2), &x3 = point(3);
274 
275  // Construct "dx/d(xi)" and "dx/d(eta)" vectors which are columns of the Jacobian.
276  // \vec{x}_{\xi} = \vec{a1}*eta + \vec{b1}
277  // \vec{x}_{\eta} = \vec{a2}*xi + \vec{b2}
278  // This is copy-pasted directly from the output of a Python script. Note: we are off
279  // by a factor of (1/4) here, but since the final result should have the same error
280  // in the numerator and denominator, it should cancel out while saving us some math
281  // operations.
282  Point
283  a1 = x0 - x1 + x2 - x3,
284  b1 = -x0 + x1 + x2 - x3,
285  a2 = a1,
286  b2 = -x0 - x1 + x2 + x3;
287 
288  // Use 2x2 quadrature to compute the integral of each basis function
289  // (as defined on the [-1,1]^2 reference domain). We use a 4-point
290  // rule, which is exact for bi-cubics. The weights for this rule are
291  // all equal to 1.
292  const Real q[2] = {-std::sqrt(3.)/3, std::sqrt(3.)/3.};
293 
294  // Nodal areas
295  Real A0 = 0., A1 = 0., A2 = 0., A3 = 0.;
296 
297  for (const auto & xi : q)
298  for (const auto & eta : q)
299  {
300  Real jxw = cross_norm(eta*a1 + b1, xi*a2 + b2);
301 
302  A0 += jxw * (1-xi) * (1-eta); // 4 * phi_0
303  A1 += jxw * (1+xi) * (1-eta); // 4 * phi_1
304  A2 += jxw * (1+xi) * (1+eta); // 4 * phi_2
305  A3 += jxw * (1-xi) * (1+eta); // 4 * phi_3
306  }
307 
308  // Compute centroid
309  return Point(x0(0)*A0 + x1(0)*A1 + x2(0)*A2 + x3(0)*A3,
310  x0(1)*A0 + x1(1)*A1 + x2(1)*A2 + x3(1)*A3,
311  x0(2)*A0 + x1(2)*A1 + x2(2)*A2 + x3(2)*A3) / (A0 + A1 + A2 + A3);
312 }

316 Real Quad4::volume () const
317 {
318  // Make copies of our points. It makes the subsequent calculations a bit
319  // shorter and avoids dereferencing the same pointer multiple times.
320  Point
321  x0 = point(0), x1 = point(1),
322  x2 = point(2), x3 = point(3);
323 
324  // Construct constant data vectors.
325  // \vec{x}_{\xi} = \vec{a1}*eta + \vec{b1}
326  // \vec{x}_{\eta} = \vec{a2}*xi + \vec{b2}
327  // This is copy-pasted directly from the output of a Python script.
328  Point
329  a1 = x0/4 - x1/4 + x2/4 - x3/4,
330  b1 = -x0/4 + x1/4 + x2/4 - x3/4,
331  a2 = a1,
332  b2 = -x0/4 - x1/4 + x2/4 + x3/4;
333 
334  // Check for quick return for parallelogram QUAD4.
335  if (a1.relative_fuzzy_equals(Point(0,0,0)))
336  return 4. * b1.cross(b2).norm();
337 
338  // Otherwise, use 2x2 quadrature to approximate the surface area.
339 
340  // 4-point rule, exact for bi-cubics. The weights for this rule are
341  // all equal to 1.
342  const Real q[2] = {-std::sqrt(3.)/3, std::sqrt(3.)/3.};
343 
344  Real vol=0.;
345  for (unsigned int i=0; i<2; ++i)
346  for (unsigned int j=0; j<2; ++j)
347  vol += cross_norm(q[j]*a1 + b1,
348  q[i]*a2 + b2);
349 
350  return vol;
351 }
