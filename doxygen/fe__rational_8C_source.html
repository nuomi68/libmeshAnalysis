37 void rational_nodal_soln(const Elem * elem,
38  const Order order,
39  const std::vector<Number> & elem_soln,
40  std::vector<Number> & nodal_soln,
41  const bool add_p_level)
42 {
43  const unsigned int n_nodes = elem->n_nodes();
44 
45  const ElemType elem_type = elem->type();
46 
47  nodal_soln.resize(n_nodes);
48 
49  const Order totalorder = static_cast<Order>(order + add_p_level*elem->p_level());
50 
51  // FEType object to be passed to various FEInterface functions below.
52  FEType fe_type(order, _underlying_fe_family);
53  FEType p_refined_fe_type(totalorder, _underlying_fe_family);
54 
55  switch (totalorder)
56  {
57  // Constant shape functions
58  case CONSTANT:
59  {
60  libmesh_assert_equal_to (elem_soln.size(), 1);
61 
62  const Number val = elem_soln[0];
63 
64  for (unsigned int n=0; n<n_nodes; n++)
65  nodal_soln[n] = val;
66 
67  return;
68  }
69 
70 
71  // For other bases do interpolation at the nodes
72  // explicitly.
73  default:
74  {
75  const unsigned int n_sf =
76  FEInterface::n_shape_functions(fe_type, elem);
77 
78  std::vector<Point> refspace_nodes;
79  FEBase::get_refspace_nodes(elem_type,refspace_nodes);
80  libmesh_assert_equal_to (refspace_nodes.size(), n_nodes);
81  libmesh_assert_equal_to (n_sf, n_nodes);
82  libmesh_assert_equal_to (elem_soln.size(), n_sf);
83 
84  std::vector<Real> node_weights(n_nodes);
85 
86  const unsigned char weight_index = elem->mapping_data();
87 
88  for (unsigned int n=0; n<n_nodes; n++)
89  node_weights[n] =
90  elem->node_ref(n).get_extra_datum<Real>(weight_index);
91 
92  for (unsigned int n=0; n<n_nodes; n++)
93  {
94  std::vector<Real> weighted_shape(n_sf);
95  Real weighted_sum = 0;
96 
97  for (unsigned int i=0; i<n_sf; i++)
98  {
99  weighted_shape[i] = node_weights[i] *
100  FEInterface::shape(fe_type, elem, i, refspace_nodes[n]);
101  weighted_sum += weighted_shape[i];
102  }
103 
104  // Zero before summation
105  nodal_soln[n] = 0;
106 
107  // u_i = Sum (alpha_i w_i phi_i) / Sum (w_j phi_j)
108  for (unsigned int i=0; i<n_sf; i++)
109  nodal_soln[n] += elem_soln[i] * weighted_shape[i];
110  nodal_soln[n] /= weighted_sum;
111  }
112 
113  return;
114  }
115  }
116 } // rational_nodal_soln()
