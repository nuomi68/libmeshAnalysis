167  diff_solver->minimum_linear_tolerance = linear_min_tol;
168  }

173 void NonlinearImplicitSystem::solve ()
174 {
175  // Log how long the nonlinear solve takes.
176  LOG_SCOPE("solve()", "System");
177 
178  this->set_solver_parameters();
179 
180  if (diff_solver.get())
181  {
182  diff_solver->solve();
183 
184  // Store the number of nonlinear iterations required to
185  // solve and the final residual.
186  _n_nonlinear_iterations = diff_solver->total_outer_iterations();
187  _final_nonlinear_residual = 0.; // FIXME - support this!
188  }
189  else
190  {
191  if (libMesh::on_command_line("--solver-system-names"))
192  nonlinear_solver->init((this->name()+"_").c_str());
193  else
194  nonlinear_solver->init();
195 
196  // FIXME - this is necessary for petsc_auto_fieldsplit
197  // nonlinear_solver->init_names(*this);
198 
199  // Solve the nonlinear system.
200  // Store the number of nonlinear iterations required to
201  // solve and the final residual.
202  std::tie(_n_nonlinear_iterations, _final_nonlinear_residual) =
203  nonlinear_solver->solve (*matrix, *solution, *rhs,
204  nonlinear_solver->relative_residual_tolerance,
205  nonlinear_solver->max_linear_iterations);
206  }
207 
208  // Update the system after the solve
209  this->update();
210 }

225 void NonlinearImplicitSystem::assembly(bool get_residual,
226  bool get_jacobian,
227  bool /*apply_heterogeneous_constraints*/,
228  bool /*apply_no_constraints*/)
229 {
230  // Get current_local_solution in sync
231  this->update();
232 
233  //-----------------------------------------------------------------------------
234  // if the user has provided both function pointers and objects only the pointer
235  // will be used, so catch that as an error
236  libmesh_error_msg_if(nonlinear_solver->jacobian && nonlinear_solver->jacobian_object,
237  "ERROR: cannot specify both a function and object to compute the Jacobian!");
238 
239  libmesh_error_msg_if(nonlinear_solver->residual && nonlinear_solver->residual_object,
240  "ERROR: cannot specify both a function and object to compute the Residual!");
241 
242  libmesh_error_msg_if(nonlinear_solver->matvec && nonlinear_solver->residual_and_jacobian_object,
243  "ERROR: cannot specify both a function and object to compute the combined Residual & Jacobian!");
244 
245 
246  if (get_jacobian)
247  {
248  if (nonlinear_solver->jacobian != nullptr)
249  nonlinear_solver->jacobian (*current_local_solution.get(), *matrix, *this);
250 
251  else if (nonlinear_solver->jacobian_object != nullptr)
252  nonlinear_solver->jacobian_object->jacobian (*current_local_solution.get(), *matrix, *this);
253 
254  else if (nonlinear_solver->matvec != nullptr)
255  nonlinear_solver->matvec (*current_local_solution.get(), get_residual ? rhs : nullptr, matrix, *this);
256 
257  else if (nonlinear_solver->residual_and_jacobian_object != nullptr)
258  nonlinear_solver->residual_and_jacobian_object->residual_and_jacobian (*current_local_solution.get(), get_residual ? rhs : nullptr, matrix, *this);
259 
260  else
261  libmesh_error_msg("Error! Unable to compute residual and/or Jacobian!");
262  }
263 
264  if (get_residual)
265  {
266  if (nonlinear_solver->residual != nullptr)
267  nonlinear_solver->residual (*current_local_solution.get(), *rhs, *this);
268 
269  else if (nonlinear_solver->residual_object != nullptr)
270  nonlinear_solver->residual_object->residual (*current_local_solution.get(), *rhs, *this);
271 
272  else if (nonlinear_solver->matvec != nullptr)
273  {
274  // we might have already grabbed the residual and jacobian together
275  if (!get_jacobian)
276  nonlinear_solver->matvec (*current_local_solution.get(), rhs, nullptr, *this);
277  }
278 
279  else if (nonlinear_solver->residual_and_jacobian_object != nullptr)
280  {
281  // we might have already grabbed the residual and jacobian together
282  if (!get_jacobian)
283  nonlinear_solver->residual_and_jacobian_object->residual_and_jacobian (*current_local_solution.get(), rhs, nullptr, *this);
284  }
285 
286  else
287  libmesh_error_msg("Error! Unable to compute residual and/or Jacobian!");
288  }
289  else
290  libmesh_assert(get_jacobian); // I can't believe you really wanted to assemble *nothing*
291 }
