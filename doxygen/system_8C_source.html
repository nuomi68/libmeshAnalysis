215 void System::init_data ()
216 {
217  parallel_object_only();
218 
219  MeshBase & mesh = this->get_mesh();
220 
221  // Add all variable groups to our underlying DofMap
222  unsigned int n_dof_map_vg = _dof_map->n_variable_groups();
223  for (auto vg : make_range(this->n_variable_groups()))
224  {
225  const VariableGroup & group = this->variable_group(vg);
226  if (vg < n_dof_map_vg)
227  libmesh_assert(group == _dof_map->variable_group(vg));
228  else
229  _dof_map->add_variable_group(group);
230  }
231 
232  // Distribute the degrees of freedom on the mesh
233  auto total_dofs = _dof_map->distribute_dofs (mesh);
234 
235  // Throw an error if the total number of DOFs is not capable of
236  // being indexed by our solution vector.
237  auto max_allowed_id = solution->max_allowed_id();
238  libmesh_error_msg_if(total_dofs > max_allowed_id,
239  "Cannot allocate a NumericVector with " << total_dofs << " degrees of freedom. "
240  "The vector can only index up to " << max_allowed_id << " entries.");
241 
242  // Recreate any user or internal constraints
243  this->reinit_constraints();
244 
245  // And clean up the send_list before we first use it
246  _dof_map->prepare_send_list();
247 
248  // Resize the solution conformal to the current mesh
249  solution->init (this->n_dofs(), this->n_local_dofs(), false, PARALLEL);
250 
251  // Resize the current_local_solution for the current mesh
252 #ifdef LIBMESH_ENABLE_GHOSTED
253  current_local_solution->init (this->n_dofs(), this->n_local_dofs(),
254  _dof_map->get_send_list(), /*fast=*/false,
255  GHOSTED);
256 #else
257  current_local_solution->init (this->n_dofs(), false, SERIAL);
258 #endif
259 
260  // from now on, adding additional vectors or variables can't be done
261  // without immediately initializing them
262  _is_initialized = true;
263 
264  // initialize & zero other vectors, if necessary
265  for (auto & [vec_name, vec] : _vectors)
266  {
267  libmesh_ignore(vec_name); // spurious warning from old gcc
268  const ParallelType type = vec->type();
269 
270  if (type == GHOSTED)
271  {
272 #ifdef LIBMESH_ENABLE_GHOSTED
273  vec->init (this->n_dofs(), this->n_local_dofs(),
274  _dof_map->get_send_list(), /*fast=*/false,
275  GHOSTED);
276 #else
277  libmesh_error_msg("Cannot initialize ghosted vectors when they are not enabled.");
278 #endif
279  }
280  else if (type == SERIAL)
281  {
282  vec->init (this->n_dofs(), false, type);
283  }
284  else
285  {
286  libmesh_assert_equal_to(type, PARALLEL);
287  vec->init (this->n_dofs(), this->n_local_dofs(), false, type);
288  }
289  }
290 
291  // Add matrices
292  this->add_matrices();
293 
294  // Clear any existing matrices
295  for (auto & pr : _matrices)
296  pr.second->clear();
297 
298  // Initialize the matrices for the system
299  if (!_basic_system_only)
300  this->init_matrices();
301 }

325 void System::init_matrices ()
326 {
327  parallel_object_only();
328 
329  // No matrices to init
330  if (_matrices.empty())
331  {
332  // any future matrices to be added will need their own
333  // initialization
334  _matrices_initialized = true;
335 
336  return;
337  }
338 
339  // Check for quick return in case the first matrix
340  // (and by extension all the matrices) has already
341  // been initialized
342  if (_matrices.begin()->second->initialized())
343  {
344  libmesh_assert(_matrices_initialized);
345  return;
346  }
347 
348  _matrices_initialized = true;
349 
350  // Tell the matrices about the dof map, and vice versa
351  for (auto & pr : _matrices)
352  {
353  SparseMatrix<Number> & m = *(pr.second);
354  libmesh_assert (!m.initialized());
355 
356  // We want to allow repeated init() on systems, but we don't
357  // want to attach the same matrix to the DofMap twice
358  if (!this->get_dof_map().is_attached(m))
359  this->get_dof_map().attach_matrix(m);
360  }
361 
362  // Compute the sparsity pattern for the current
363  // mesh and DOF distribution. This also updates
364  // additional matrices, \p DofMap now knows them
365  this->get_dof_map().compute_sparsity(this->get_mesh());
366 
367  // Initialize matrices and set to zero
368  for (auto & pr : _matrices)
369  {
370  pr.second->init(_matrix_types[pr.first]);
371  pr.second->zero();
372  }
373 }

377 void System::restrict_vectors ()
378 {
379  parallel_object_only();
380 
381 #ifdef LIBMESH_ENABLE_AMR
382  // Restrict the _vectors on the coarsened cells
383  for (auto & [vec_name, vec] : _vectors)
384  {
385  NumericVector<Number> * v = vec.get();
386 
387  if (_vector_projections[vec_name])
388  {
389  this->project_vector (*v, this->vector_is_adjoint(vec_name));
390  }
391  else
392  {
393  const ParallelType type = vec->type();
394 
395  if (type == GHOSTED)
396  {
397 #ifdef LIBMESH_ENABLE_GHOSTED
398  vec->init (this->n_dofs(), this->n_local_dofs(),
399  _dof_map->get_send_list(), /*fast=*/false,
400  GHOSTED);
401 #else
402  libmesh_error_msg("Cannot initialize ghosted vectors when they are not enabled.");
403 #endif
404  }
405  else
406  vec->init (this->n_dofs(), this->n_local_dofs(), false, type);
407  }
408  }
409 
410  const std::vector<dof_id_type> & send_list = _dof_map->get_send_list ();
411 
412  // Restrict the solution on the coarsened cells
413  if (_solution_projection)
414  this->project_vector (*solution);
415  // Or at least make sure the solution vector is the correct size
416  else
417  solution->init (this->n_dofs(), this->n_local_dofs(), true, PARALLEL);
418 
419 #ifdef LIBMESH_ENABLE_GHOSTED
420  current_local_solution->init(this->n_dofs(),
421  this->n_local_dofs(), send_list,
422  false, GHOSTED);
423 #else
424  current_local_solution->init(this->n_dofs());
425 #endif
426 
427  if (_solution_projection)
428  solution->localize (*current_local_solution, send_list);
429 
430 #endif // LIBMESH_ENABLE_AMR
431 }

600 bool System::compare (const System & other_system,
601  const Real threshold,
602  const bool verbose) const
603 {
604  // we do not care for matrices, but for vectors
605  libmesh_assert (_is_initialized);
606  libmesh_assert (other_system._is_initialized);
607 
608  if (verbose)
609  {
610  libMesh::out << " Systems \"" << _sys_name << "\"" << std::endl;
611  libMesh::out << " comparing matrices not supported." << std::endl;
612  libMesh::out << " comparing names...";
613  }
614 
615  // compare the name: 0 means identical
616  const int name_result = _sys_name.compare(other_system.name());
617  if (verbose)
618  {
619  if (name_result == 0)
620  libMesh::out << " identical." << std::endl;
621  else
622  libMesh::out << " names not identical." << std::endl;
623  libMesh::out << " comparing solution vector...";
624  }
625 
626 
627  // compare the solution: -1 means identical
628  const int solu_result = solution->compare (*other_system.solution.get(),
629  threshold);
630 
631  if (verbose)
632  {
633  if (solu_result == -1)
634  libMesh::out << " identical up to threshold." << std::endl;
635  else
636  libMesh::out << " first difference occurred at index = "
637  << solu_result << "." << std::endl;
638  }
639 
640 
641  // safety check, whether we handle at least the same number
642  // of vectors
643  std::vector<int> ov_result;
644 
645  if (this->n_vectors() != other_system.n_vectors())
646  {
647  if (verbose)
648  {
649  libMesh::out << " Fatal difference. This system handles "
650  << this->n_vectors() << " add'l vectors," << std::endl
651  << " while the other system handles "
652  << other_system.n_vectors()
653  << " add'l vectors." << std::endl
654  << " Aborting comparison." << std::endl;
655  }
656  return false;
657  }
658  else if (this->n_vectors() == 0)
659  {
660  // there are no additional vectors...
661  ov_result.clear ();
662  }
663  else
664  {
665  // compare other vectors
666  for (auto & [vec_name, vec] : _vectors)
667  {
668  if (verbose)
669  libMesh::out << " comparing vector \""
670  << vec_name << "\" ...";
671 
672  // assume they have the same name
673  const NumericVector<Number> & other_system_vector =
674  other_system.get_vector(vec_name);
675 
676  ov_result.push_back(vec->compare(other_system_vector, threshold));
677 
678  if (verbose)
679  {
680  if (ov_result[ov_result.size()-1] == -1)
681  libMesh::out << " identical up to threshold." << std::endl;
682  else
683  libMesh::out << " first difference occurred at" << std::endl
684  << " index = " << ov_result[ov_result.size()-1] << "." << std::endl;
685  }
686  }
687  } // finished comparing additional vectors
688 
689 
690  bool overall_result;
691 
692  // sum up the results
693  if ((name_result==0) && (solu_result==-1))
694  {
695  if (ov_result.size()==0)
696  overall_result = true;
697  else
698  {
699  bool ov_identical;
700  unsigned int n = 0;
701  do
702  {
703  ov_identical = (ov_result[n]==-1);
704  n++;
705  }
706  while (ov_identical && n<ov_result.size());
707  overall_result = ov_identical;
708  }
709  }
710  else
711  overall_result = false;
712 
713  if (verbose)
714  {
715  libMesh::out << " finished comparisons, ";
716  if (overall_result)
717  libMesh::out << "found no differences." << std::endl << std::endl;
718  else
719  libMesh::out << "found differences." << std::endl << std::endl;
720  }
721 
722  return overall_result;
723 }

771  // we'll ignore any type setting. If we're in parallel, we
772  // might have a type change to deal with.
773

776  // If the type setting has changed in a way we can't
777  // perceive as an upgrade or a downgrade, scream.
778  libmesh_assert_equal_to(type == SERIAL,
779  vec.type() == SERIAL);
780

785  // to risk zeroing data.

788  if (!vec.closed())

798  _dof_map->get_send_list(), /*fast=*/false,
799  GHOSTED);
800 #else
801  libmesh_error_msg("Cannot initialize ghosted vectors when they are not enabled.");
802 #endif
803 
804  *new_vec = vec;
805  vec.swap(*new_vec);
806  }
807  else
808  vec.type() = type;
809  }
810  }
811 
812  // Any upgrades are done; we're happy here.
813  return vec;
814  }
815 
816  // Otherwise build the vector
817  auto pr = _vectors.emplace(vec_name, NumericVector<Number>::build(this->comm()));
818  auto buf = pr.first->second.get();
819  _vector_projections.emplace(vec_name, projections);
820  buf->type() = type;
821

823  _vector_is_adjoint.emplace(vec_name, -1);
824 
825  // Initialize it if necessary
826  if (_is_initialized)
827  {
828  if (type == GHOSTED)
829  {
830 #ifdef LIBMESH_ENABLE_GHOSTED
831  buf->init (this->n_dofs(), this->n_local_dofs(),
832  _dof_map->get_send_list(), /*fast=*/false,
833  GHOSTED);

836 #endif
837  }
838  else
839  buf->init (this->n_dofs(), this->n_local_dofs(), false, type);
840  }
841

1318  // or if it is, that it's the type we want
1319  for (auto v : make_range(this->n_vars()))
1320  if (this->variable_name(v) == var)
1321  {
1322  if (this->variable_type(v) == type)
1323  {
1324  // Check whether the existing variable's active subdomains also matches
1325  // the incoming variable's active subdomains. If they don't match, then
1326  // either it is an error by the user or the user is trying to change the
1327  // subdomain restriction after the variable has already been added, which
1328  // is not supported.
1329  const Variable & existing_var = this->variable(v);

1332  bool check1 = (!active_subdomains || active_subdomains->empty()) && existing_var.implicitly_active();
1333 
1334  // Check if the provided active_subdomains is equal to the existing_var's active_subdomains
1335  bool check2 = (active_subdomains && (*active_subdomains == existing_var.active_subdomains()));
1336 
1337  // If either of these checks passed, then we already have this variable
1338  if (check1 || check2)
1339  return _variables[v].number();
1340  }
1341 
1342  libmesh_error_msg("ERROR: incompatible variable " << var << " has already been added for this system!");
1343  }

1348  {

1350  // variables of the same FEType and subdomain restriction, catch
1351  // that here and add them as members of the same VariableGroup.
1352  //
1353  // start by setting this flag to whatever the user has requested
1354  // and then consider the conditions which should negate it.
1355  bool should_be_in_vg = this->identify_variable_groups();

1365  if (vg.type() != type)
1366  should_be_in_vg = false;
1367 
1368  // they are restricted, we aren't?
1369  if (their_active_subdomains &&
1370  (!active_subdomains || (active_subdomains && active_subdomains->empty())))
1371  should_be_in_vg = false;
1372 
1373  // they aren't restricted, we are?
1374  if (!their_active_subdomains && (active_subdomains && !active_subdomains->empty()))
1375  should_be_in_vg = false;
1376 
1377  if (their_active_subdomains && active_subdomains)
1378  // restricted to different sets?
1379  if (*their_active_subdomains != *active_subdomains)
1380  should_be_in_vg = false;
1381

1399  return this->add_variables (std::vector<std::string>(1, std::string(var)),

1693 Real System::calculate_norm(const NumericVector<Number> & v,
1694  const SystemNorm & norm,
1695  std::set<unsigned int> * skip_dimensions) const
1696 {
1697  // This function must be run on all processors at once
1698  parallel_object_only();
1699 
1700  LOG_SCOPE ("calculate_norm()", "System");
1701 
1702  // Zero the norm before summation
1703  Real v_norm = 0.;
1704 
1705  if (norm.is_discrete())
1706  {
1707  //Check to see if all weights are 1.0 and all types are equal
1708  FEMNormType norm_type0 = norm.type(0);
1709  unsigned int check_var = 0, check_end = this->n_vars();
1710  for (; check_var != check_end; ++check_var)
1711  if ((norm.weight(check_var) != 1.0) || (norm.type(check_var) != norm_type0))
1712  break;
1713 
1714  //All weights were 1.0 so just do the full vector discrete norm
1715  if (check_var == this->n_vars())
1716  {
1717  if (norm_type0 == DISCRETE_L1)
1718  return v.l1_norm();
1719  if (norm_type0 == DISCRETE_L2)
1720  return v.l2_norm();
1721  if (norm_type0 == DISCRETE_L_INF)
1722  return v.linfty_norm();
1723  else
1724  libmesh_error_msg("Invalid norm_type0 = " << Utility::enum_to_string(norm_type0));
1725  }
1726 
1727  for (auto var : make_range(this->n_vars()))
1728  {
1729  // Skip any variables we don't need to integrate
1730  if (norm.weight(var) == 0.0)
1731  continue;
1732 
1733  v_norm += norm.weight(var) * discrete_var_norm(v, var, norm.type(var));
1734  }
1735 
1736  return v_norm;
1737  }
1738 
1739  // Localize the potentially parallel vector
1740  std::unique_ptr<NumericVector<Number>> local_v = NumericVector<Number>::build(this->comm());
1741  local_v->init(v.size(), v.local_size(), _dof_map->get_send_list(),
1742  true, GHOSTED);
1743  v.localize (*local_v, _dof_map->get_send_list());
1744 
1745  // I'm not sure how best to mix Hilbert norms on some variables (for
1746  // which we'll want to square then sum then square root) with norms
1747  // like L_inf (for which we'll just want to take an absolute value
1748  // and then sum).
1749  bool using_hilbert_norm = true,
1750  using_nonhilbert_norm = true;
1751 
1752  // Loop over all variables
1753  for (auto var : make_range(this->n_vars()))
1754  {
1755  // Skip any variables we don't need to integrate
1756  Real norm_weight_sq = norm.weight_sq(var);
1757  if (norm_weight_sq == 0.0)
1758  continue;
1759  Real norm_weight = norm.weight(var);
1760 
1761  // Check for unimplemented norms (rather than just returning 0).
1762  FEMNormType norm_type = norm.type(var);
1763  if ((norm_type==H1) ||
1764  (norm_type==H2) ||
1765  (norm_type==L2) ||
1766  (norm_type==H1_SEMINORM) ||
1767  (norm_type==H2_SEMINORM))
1768  {
1769  if (!using_hilbert_norm)
1770  libmesh_not_implemented();
1771  using_nonhilbert_norm = false;
1772  }
1773  else if ((norm_type==L1) ||
1774  (norm_type==L_INF) ||
1775  (norm_type==W1_INF_SEMINORM) ||
1776  (norm_type==W2_INF_SEMINORM))
1777  {
1778  if (!using_nonhilbert_norm)
1779  libmesh_not_implemented();
1780  using_hilbert_norm = false;
1781  }
1782  else
1783  libmesh_not_implemented();
1784 
1785  const FEType & fe_type = this->get_dof_map().variable_type(var);
1786 
1787  // Allow space for dims 0-3, even if we don't use them all
1788  std::vector<std::unique_ptr<FEBase>> fe_ptrs(4);
1789  std::vector<std::unique_ptr<QBase>> q_rules(4);
1790 
1791  const std::set<unsigned char> & elem_dims = _mesh.elem_dimensions();
1792 
1793  // Prepare finite elements for each dimension present in the mesh
1794  for (const auto & dim : elem_dims)
1795  {
1796  if (skip_dimensions && skip_dimensions->find(dim) != skip_dimensions->end())
1797  continue;
1798 
1799  // Construct quadrature and finite element objects
1800  q_rules[dim] = fe_type.default_quadrature_rule (dim);
1801  fe_ptrs[dim] = FEBase::build(dim, fe_type);
1802 
1803  // Attach quadrature rule to FE object
1804  fe_ptrs[dim]->attach_quadrature_rule (q_rules[dim].get());
1805  }
1806 
1807  std::vector<dof_id_type> dof_indices;
1808 
1809  // Begin the loop over the elements
1810  for (const auto & elem : this->get_mesh().active_local_element_ptr_range())
1811  {
1812  const unsigned int dim = elem->dim();
1813 
1814  // One way for implementing this would be to exchange the fe with the FEInterface- class.
1815  // However, it needs to be discussed whether integral-norms make sense for infinite elements.
1816  // or in which sense they could make sense.
1817  if (elem->infinite() )
1818  libmesh_not_implemented();
1819 
1820  if (skip_dimensions && skip_dimensions->find(dim) != skip_dimensions->end())
1821  continue;
1822 
1823  FEBase * fe = fe_ptrs[dim].get();
1824  QBase * qrule = q_rules[dim].get();
1825  libmesh_assert(fe);
1826  libmesh_assert(qrule);
1827 
1828  const std::vector<Real> & JxW = fe->get_JxW();
1829  const std::vector<std::vector<Real>> * phi = nullptr;
1830  if (norm_type == H1 ||
1831  norm_type == H2 ||
1832  norm_type == L2 ||
1833  norm_type == L1 ||
1834  norm_type == L_INF)
1835  phi = &(fe->get_phi());
1836 
1837  const std::vector<std::vector<RealGradient>> * dphi = nullptr;
1838  if (norm_type == H1 ||
1839  norm_type == H2 ||
1840  norm_type == H1_SEMINORM ||
1841  norm_type == W1_INF_SEMINORM)
1842  dphi = &(fe->get_dphi());
1843 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
1844  const std::vector<std::vector<RealTensor>> * d2phi = nullptr;
1845  if (norm_type == H2 ||
1846  norm_type == H2_SEMINORM ||
1847  norm_type == W2_INF_SEMINORM)
1848  d2phi = &(fe->get_d2phi());
1849 #endif
1850 
1851  fe->reinit (elem);
1852 
1853  this->get_dof_map().dof_indices (elem, dof_indices, var);
1854 
1855  const unsigned int n_qp = qrule->n_points();
1856 
1857  const unsigned int n_sf = cast_int<unsigned int>
1858  (dof_indices.size());
1859 
1860  // Begin the loop over the Quadrature points.
1861  for (unsigned int qp=0; qp<n_qp; qp++)
1862  {
1863  if (norm_type == L1)
1864  {
1865  Number u_h = 0.;
1866  for (unsigned int i=0; i != n_sf; ++i)
1867  u_h += (*phi)[i][qp] * (*local_v)(dof_indices[i]);
1868  v_norm += norm_weight *
1869  JxW[qp] * std::abs(u_h);
1870  }
1871 
1872  if (norm_type == L_INF)
1873  {
1874  Number u_h = 0.;
1875  for (unsigned int i=0; i != n_sf; ++i)
1876  u_h += (*phi)[i][qp] * (*local_v)(dof_indices[i]);
1877  v_norm = std::max(v_norm, norm_weight * std::abs(u_h));
1878  }
1879 
1880  if (norm_type == H1 ||
1881  norm_type == H2 ||
1882  norm_type == L2)
1883  {
1884  Number u_h = 0.;
1885  for (unsigned int i=0; i != n_sf; ++i)
1886  u_h += (*phi)[i][qp] * (*local_v)(dof_indices[i]);
1887  v_norm += norm_weight_sq *
1888  JxW[qp] * TensorTools::norm_sq(u_h);
1889  }
1890 
1891  if (norm_type == H1 ||
1892  norm_type == H2 ||
1893  norm_type == H1_SEMINORM)
1894  {
1895  Gradient grad_u_h;
1896  for (unsigned int i=0; i != n_sf; ++i)
1897  grad_u_h.add_scaled((*dphi)[i][qp], (*local_v)(dof_indices[i]));
1898  v_norm += norm_weight_sq *
1899  JxW[qp] * grad_u_h.norm_sq();
1900  }
1901 
1902  if (norm_type == W1_INF_SEMINORM)
1903  {
1904  Gradient grad_u_h;
1905  for (unsigned int i=0; i != n_sf; ++i)
1906  grad_u_h.add_scaled((*dphi)[i][qp], (*local_v)(dof_indices[i]));
1907  v_norm = std::max(v_norm, norm_weight * grad_u_h.norm());
1908  }
1909 
1910 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
1911  if (norm_type == H2 ||
1912  norm_type == H2_SEMINORM)
1913  {
1914  Tensor hess_u_h;
1915  for (unsigned int i=0; i != n_sf; ++i)
1916  hess_u_h.add_scaled((*d2phi)[i][qp], (*local_v)(dof_indices[i]));
1917  v_norm += norm_weight_sq *
1918  JxW[qp] * hess_u_h.norm_sq();
1919  }
1920 
1921  if (norm_type == W2_INF_SEMINORM)
1922  {
1923  Tensor hess_u_h;
1924  for (unsigned int i=0; i != n_sf; ++i)
1925  hess_u_h.add_scaled((*d2phi)[i][qp], (*local_v)(dof_indices[i]));
1926  v_norm = std::max(v_norm, norm_weight * hess_u_h.norm());
1927  }
1928 #endif
1929  }
1930  }
1931  }
1932 
1933  if (using_hilbert_norm)
1934  {
1935  this->comm().sum(v_norm);
1936  v_norm = std::sqrt(v_norm);
1937  }
1938  else
1939  {
1940  this->comm().max(v_norm);
1941  }
1942 
1943  return v_norm;
1944 }
