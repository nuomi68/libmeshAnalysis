60 struct NodalVolumeIntegral
61 {
62  // API
63  void accumulate_qp(Real xi, Real eta, Real zeta, Real JxW)
64  {
65  // Copied from fe_lagrange_shape_3D.C
66  static const unsigned int i0[] = {0, 0, 0, 1, 1, 1};
67  static const unsigned int i1[] = {0, 1, 2, 0, 1, 2};
68 
69  // Copied from fe_lagrange_shape_2D.C
70  auto tri3_phi = [](int i, Real x, Real y)
71  {
72  switch(i)
73  {
74  case 0:
75  return 1 - x - y;
76 
77  case 1:
78  return x;
79 
80  case 2:
81  return y;
82 
83  default:
84  libmesh_error_msg("Invalid shape function index i = " << i);
85  }
86  };
87 
88  for (int i=0; i<6; ++i)
89  {
90  Real phi =
91  tri3_phi(i1[i], xi, eta) * fe_lagrange_1D_linear_shape(i0[i], zeta);
92 
93  V[i] += phi * JxW;
94  }
95  };
96 
97  // nodal volumes
98  std::array<Real, 6> V {};
99 };

220 std::unique_ptr<Elem> Prism6::build_side_ptr (const unsigned int i,
221  bool proxy)
222 {
223  libmesh_assert_less (i, this->n_sides());
224 
225  std::unique_ptr<Elem> face;
226  if (proxy)
227  {
228 #ifdef LIBMESH_ENABLE_DEPRECATED
229  libmesh_deprecated();
230  switch(i)
231  {
232  case 0:
233  case 4:
234  {
235  face = std::make_unique<Side<Tri3,Prism6>>(this,i);
236  break;
237  }
238 
239  case 1:
240  case 2:
241  case 3:
242  {
243  face = std::make_unique<Side<Quad4,Prism6>>(this,i);
244  break;
245  }
246 
247  default:
248  libmesh_error_msg("Invalid side i = " << i);
249  }
250 #else
251  libmesh_error();
252 #endif // LIBMESH_ENABLE_DEPRECATED
253  }
254  else
255  {
256  switch (i)
257  {
258  case 0: // the triangular face at z=-1
259  case 4: // the triangular face at z=1
260  {
261  face = std::make_unique<Tri3>();
262  break;
263  }
264  case 1: // the quad face at y=0
265  case 2: // the other quad face
266  case 3: // the quad face at x=0
267  {
268  face = std::make_unique<Quad4>();
269  break;
270  }
271  default:
272  libmesh_error_msg("Invalid side i = " << i);
273  }
274 
275  // Set the nodes
276  for (auto n : face->node_index_range())
277  face->set_node(n) = this->node_ptr(Prism6::side_nodes_map[i][n]);
278  }
279 
280 #ifdef LIBMESH_ENABLE_DEPRECATED
281  if (!proxy) // proxy sides used to leave parent() set
282 #endif
283  face->set_parent(nullptr);
284  face->set_interior_parent(this);
285 
286  face->subdomain_id() = this->subdomain_id();
287  face->set_mapping_type(this->mapping_type());
288 #ifdef LIBMESH_ENABLE_AMR
289  face->set_p_level(this->p_level());
290 #endif
291 
292  return face;
293 }

459 Point Prism6::true_centroid () const
460 {
461  NodalVolumeIntegral nvi;
462  cell_integral(this, nvi);
463 
464  // Compute centroid
465  Point cp;
466  Real vol = 0.;
467 
468  for (int i=0; i<6; ++i)
469  {
470  cp += this->point(i) * nvi.V[i];
471  vol += nvi.V[i];
472  }
473 
474  return cp / vol;
475 }

492 Prism6::permute(unsigned int perm_num)
493 {
494  libmesh_assert_less (perm_num, 6);
495  const unsigned int side = perm_num % 2;
496  const unsigned int rotate = perm_num / 2;
497 
498  for (unsigned int i = 0; i != rotate; ++i)
499  {
500  swap3nodes(0,1,2);
501  swap3nodes(3,4,5);
502  swap3neighbors(1,2,3);
503  }
504 
505  switch (side) {
506  case 0:
507  break;
508  case 1:
509  swap2nodes(1,3);
510  swap2nodes(0,4);
511  swap2nodes(2,5);
512  swap2neighbors(0,4);
513  swap2neighbors(2,3);
514  break;
515  default:
516  libmesh_error();
517  }

549 template <typename T>
550 void cell_integral(const Elem * elem, T & t)
551 {
552  // Conveient references to our points.
553  const Point
554  &x0 = elem->point(0), &x1 = elem->point(1), &x2 = elem->point(2),
555  &x3 = elem->point(3), &x4 = elem->point(4), &x5 = elem->point(5);
556 
557  // constant and zeta terms only. These are copied directly from a
558  // Python script.
559  Point dx_dxi[2] =
560  {
561  -x0/2 + x1/2 - x3/2 + x4/2, // constant
562  x0/2 - x1/2 - x3/2 + x4/2, // zeta
563  };
564 
565  // constant and zeta terms only. These are copied directly from a
566  // Python script.
567  Point dx_deta[2] =
568  {
569  -x0/2 + x2/2 - x3/2 + x5/2, // constant
570  x0/2 - x2/2 - x3/2 + x5/2, // zeta
571  };
572 
573  // Constant, xi, and eta terms
574  Point dx_dzeta[3] =
575  {
576  -x0/2 + x3/2, // constant
577  x0/2 - x2/2 - x3/2 + x5/2, // eta
578  x0/2 - x1/2 - x3/2 + x4/2 // xi
579  };
580 
581  // The quadrature rule for the Prism6 is a tensor product between a
582  // four-point TRI3 rule (in xi, eta) and a two-point EDGE2 rule (in
583  // zeta) which is capable of integrating cubics exactly.
584 
585  // Number of points in the 2D quadrature rule.
586  const int N2D = 4;
587 
588  static const Real w2D[N2D] =
589  {
590  Real(1.5902069087198858469718450103758e-01L),
591  Real(9.0979309128011415302815498962418e-02L),
592  Real(1.5902069087198858469718450103758e-01L),
593  Real(9.0979309128011415302815498962418e-02L)
594  };
595 
596  static const Real xi[N2D] =
597  {
598  Real(1.5505102572168219018027159252941e-01L),
599  Real(6.4494897427831780981972840747059e-01L),
600  Real(1.5505102572168219018027159252941e-01L),
601  Real(6.4494897427831780981972840747059e-01L)
602  };
603 
604  static const Real eta[N2D] =
605  {
606  Real(1.7855872826361642311703513337422e-01L),
607  Real(7.5031110222608118177475598324603e-02L),
608  Real(6.6639024601470138670269327409637e-01L),
609  Real(2.8001991549907407200279599420481e-01L)
610  };
611 
612  // Number of points in the 1D quadrature rule. The weights of the
613  // 1D quadrature rule are equal to 1.
614  const int N1D = 2;
615 
616  // Points of the 1D quadrature rule
617  static const Real zeta[N1D] =
618  {
619  -std::sqrt(3.)/3,
620  std::sqrt(3.)/3.
621  };
622 
623  for (int i=0; i<N2D; ++i)
624  {
625  // dx_dzeta depends only on the 2D quadrature rule points.
626  Point dx_dzeta_q = dx_dzeta[0] + eta[i]*dx_dzeta[1] + xi[i]*dx_dzeta[2];
627 
628  for (int j=0; j<N1D; ++j)
629  {
630  // dx_dxi and dx_deta only depend on the 1D quadrature rule points.
631  Point
632  dx_dxi_q = dx_dxi[0] + zeta[j]*dx_dxi[1],
633  dx_deta_q = dx_deta[0] + zeta[j]*dx_deta[1];
634 
635  // Compute t-specific contributions at current qp
636  t.accumulate_qp(xi[i], eta[i], zeta[j], w2D[i] * triple_product(dx_dxi_q, dx_deta_q, dx_dzeta_q));
637  }
638  }
639 }
