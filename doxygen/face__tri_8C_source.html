140 Real Tri::quality (const ElemQuality q) const
141 {
142  switch (q)
143  {
144 
148  case DISTORTION:
149  case STRETCH:
150  {
151  const Point & p1 = this->point(0);
152  const Point & p2 = this->point(1);
153  const Point & p3 = this->point(2);
154 
155  Point v1 = p2 - p1;
156  Point v2 = p3 - p1;
157  Point v3 = p3 - p2;
158  const Real l1 = v1.norm();
159  const Real l2 = v2.norm();
160  const Real l3 = v3.norm();
161 
162  // if one length is 0, quality is quite bad!
163  if ((l1 <=0.) || (l2 <= 0.) || (l3 <= 0.))
164  return 0.;
165 
166  const Real s1 = std::sin(std::acos(v1*v2/l1/l2)/2.);
167  v1 *= -1;
168  const Real s2 = std::sin(std::acos(v1*v3/l1/l3)/2.);
169  const Real s3 = std::sin(std::acos(v2*v3/l2/l3)/2.);
170 
171  return 8. * s1 * s2 * s3;
172 
173  }
174 
175  // From: P. Knupp, "Algebraic mesh quality metrics for
176  // unstructured initial meshes," Finite Elements in Analysis
177  // and Design 39, 2003, p. 217-241, Section 3.2.
178  case SHAPE:
179  {
180  // Unlike Quads, the Tri SHAPE metric is independent of the
181  // node at which it is computed, we choose to compute it for
182  // node 0.
183 
184  // The nodal Jacobian matrix A is a 3x2 matrix, hence we
185  // represent it by a std:array with 6 entries.
186  Point
187  d01 = point(1) - point(0),
188  d02 = point(2) - point(0);
189 
190  std::array<Real, 6> A =
191  {{d01(0), d02(0),
192  d01(1), d02(1),
193  d01(2), d02(2)}};
194 
195  // Compute metric tensor entries, T = A^T * A.
196  // This is a symmetric 2x2 matrix so we only
197  // compute one of the off-diagonal entries.
198  // As in the paper, we define lambda_ij := T_ij.
199  Real
200  lambda11 = A[0]*A[0] + A[2]*A[2] + A[4]*A[4],
201  lambda12 = A[0]*A[1] + A[2]*A[3] + A[4]*A[5],
202  lambda22 = A[1]*A[1] + A[3]*A[3] + A[5]*A[5];
203 
204  // Compute the denominator of the metric. If it is exactly
205  // zero then return 0 (lowest quality) for this metric.
206  Real den = lambda11 + lambda22 - lambda12;
207  if (den == 0.0)
208  return 0.;
209 
210  // Compute the nodal area
211  Real alpha = std::sqrt(lambda11 * lambda22 - lambda12 * lambda12);
212 
213  // Finally, compute and return the metric.
214  return std::sqrt(3) * alpha / den;
215  }
216 
217  default:
218  return Elem::quality(q);
219  }
220 
221  // We won't get here.
222  return Elem::quality(q);
223 }

231 {

255  break;
256

258  case SHAPE:
