141 bool Prism21::has_affine_map() const
142 {
143  // Make sure z edges are affine
144  Point v = this->point(3) - this->point(0);
145  if (!v.relative_fuzzy_equals(this->point(4) - this->point(1), affine_tol) ||
146  !v.relative_fuzzy_equals(this->point(5) - this->point(2), affine_tol))
147  return false;
148 
149  // Make sure edges are straight
150  v /= 2;
151  if (!v.relative_fuzzy_equals(this->point(9) - this->point(0), affine_tol) ||
152  !v.relative_fuzzy_equals(this->point(10) - this->point(1), affine_tol) ||
153  !v.relative_fuzzy_equals(this->point(11) - this->point(2), affine_tol) ||
154  !v.relative_fuzzy_equals(this->point(15) - this->point(6), affine_tol) ||
155  !v.relative_fuzzy_equals(this->point(16) - this->point(7), affine_tol) ||
156  !v.relative_fuzzy_equals(this->point(17) - this->point(8), affine_tol))
157  return false;
158  v = (this->point(1) - this->point(0))/2;
159  if (!v.relative_fuzzy_equals(this->point(6) - this->point(0), affine_tol) ||
160  !v.relative_fuzzy_equals(this->point(12) - this->point(3), affine_tol))
161  return false;
162  v = (this->point(2) - this->point(0))/2;
163  if (!v.relative_fuzzy_equals(this->point(8) - this->point(0), affine_tol) ||
164  !v.relative_fuzzy_equals(this->point(14) - this->point(3), affine_tol))
165  return false;
166  v = (this->point(2) - this->point(1))/2;
167  if (!v.relative_fuzzy_equals(this->point(7) - this->point(1), affine_tol) ||
168  !v.relative_fuzzy_equals(this->point(13) - this->point(4), affine_tol))
169  return false;
170 
171  // Make sure triangle face midpoints are centered
172  v = this->point(2) + this->point(1) - 2*this->point(0);
173  if (!v.relative_fuzzy_equals((this->point(18)-this->point(0))*3))
174  return false;
175  v = this->point(5) + this->point(4) - 2*this->point(3);
176  if (!v.relative_fuzzy_equals((this->point(19)-this->point(3))*3))
177  return false;
178  v = this->point(11) + this->point(10) - 2*this->point(9);
179  if (!v.relative_fuzzy_equals((this->point(20)-this->point(9))*3))
180  return false;
181 
182  return true;
183 }
