52 template<typename OutputShape>
53 void HDivFETransformation<OutputShape>::map_phi(const unsigned int dim,
54  const Elem * const elem,
55  const std::vector<Point> & qp,
56  const FEGenericBase<OutputShape> & fe,
57  std::vector<std::vector<OutputShape>> & phi,
58  const bool /*add_p_level*/) const
59 {
60  switch (dim)
61  {
62  case 0:
63  case 1:
64  libmesh_error_msg("These element transformations only make sense in 2D and 3D.");
65 
66  case 2:
67  {
68  const std::vector<RealGradient> & dxyz_dxi = fe.get_fe_map().get_dxyzdxi();
69  const std::vector<RealGradient> & dxyz_deta = fe.get_fe_map().get_dxyzdeta();
70 
71  const std::vector<Real> & J = fe.get_fe_map().get_jacobian();
72 
73  // phi = J^{-1} * (dx/dxi) * \hat{phi}
74  for (auto i : index_range(phi))
75  for (auto p : index_range(phi[i]))
76  {
77  Real dx_dxi = dxyz_dxi[p](0);
78  Real dx_deta = dxyz_deta[p](0);
79 
80  Real dy_dxi = dxyz_dxi[p](1);
81  Real dy_deta = dxyz_deta[p](1);
82 
83  // Need to temporarily cache reference shape functions
84  // We are computing mapping basis functions, so we explicitly ignore
85  // any non-zero p_level() the Elem might have.
86  OutputShape phi_ref;
87  FEInterface::shape(fe.get_fe_type(), /*extra_order=*/0, elem, i, qp[p], phi_ref);
88 
89  phi[i][p](0) = (dx_dxi*phi_ref(0) + dx_deta*phi_ref(1))/J[p];
90  phi[i][p](1) = (dy_dxi*phi_ref(0) + dy_deta*phi_ref(1))/J[p];
91  }
92 
93  break;
94  }
95  case 3:
96  {
97  const std::vector<RealGradient> & dxyz_dxi = fe.get_fe_map().get_dxyzdxi();
98  const std::vector<RealGradient> & dxyz_deta = fe.get_fe_map().get_dxyzdeta();
99  const std::vector<RealGradient> & dxyz_dzeta = fe.get_fe_map().get_dxyzdzeta();
100 
101  const std::vector<Real> & J = fe.get_fe_map().get_jacobian();
102 
103  // phi = J^{-1} * (dx/dxi) * \hat{phi}
104  for (auto i : index_range(phi))
105  for (auto p : index_range(phi[i]))
106  {
107  Real dx_dxi = dxyz_dxi[p](0);
108  Real dx_deta = dxyz_deta[p](0);
109  Real dx_dzeta = dxyz_dzeta[p](0);
110 
111  Real dy_dxi = dxyz_dxi[p](1);
112  Real dy_deta = dxyz_deta[p](1);
113  Real dy_dzeta = dxyz_dzeta[p](1);
114 
115  Real dz_dxi = dxyz_dxi[p](2);
116  Real dz_deta = dxyz_deta[p](2);
117  Real dz_dzeta = dxyz_dzeta[p](2);
118 
119  // Need to temporarily cache reference shape functions
120  // We are computing mapping basis functions, so we explicitly ignore
121  // any non-zero p_level() the Elem might have.
122  OutputShape phi_ref;
123  FEInterface::shape(fe.get_fe_type(), /*extra_order=*/0, elem, i, qp[p], phi_ref);
124 
125  phi[i][p](0) = (dx_dxi*phi_ref(0) + dx_deta*phi_ref(1) + dx_dzeta*phi_ref(2))/J[p];
126  phi[i][p](1) = (dy_dxi*phi_ref(0) + dy_deta*phi_ref(1) + dy_dzeta*phi_ref(2))/J[p];
127  phi[i][p](2) = (dz_dxi*phi_ref(0) + dz_deta*phi_ref(1) + dz_dzeta*phi_ref(2))/J[p];
128  }
129 
130  break;
131  }
132 
133  default:
134  libmesh_error_msg("Invalid dim = " << dim);
135  } // switch(dim)
136 }

138 template<typename OutputShape>
139 void HDivFETransformation<OutputShape>::map_div(const unsigned int dim,
140  const Elem * const,
141  const std::vector<Point> &,
142  const FEGenericBase<OutputShape> & fe,
143  std::vector<std::vector<typename FEGenericBase<OutputShape>::OutputDivergence>> & div_phi) const
144 {
145  switch (dim)
146  {
147  case 0:
148  case 1:
149  libmesh_error_msg("These element transformations only make sense in 2D and 3D.");
150 
151  case 2:
152  {
153  const std::vector<std::vector<OutputShape>> & dphi_dxi = fe.get_dphidxi();
154  const std::vector<std::vector<OutputShape>> & dphi_deta = fe.get_dphideta();
155 
156  const std::vector<Real> & J = fe.get_fe_map().get_jacobian();
157 
158  // div(phi) = J^{-1} * div(\hat{phi})
159  for (auto i : index_range(div_phi))
160  for (auto p : index_range(div_phi[i]))
161  {
162  div_phi[i][p] = (dphi_dxi[i][p](0) + dphi_deta[i][p](1))/J[p];
163  }
164 
165  break;
166  }
167  case 3:
168  {
169  const std::vector<std::vector<OutputShape>> & dphi_dxi = fe.get_dphidxi();
170  const std::vector<std::vector<OutputShape>> & dphi_deta = fe.get_dphideta();
171  const std::vector<std::vector<OutputShape>> & dphi_dzeta = fe.get_dphidzeta();
172 
173  const std::vector<Real> & J = fe.get_fe_map().get_jacobian();
174 
175  // div(phi) = J^{-1} * div(\hat{phi})
176  for (auto i : index_range(div_phi))
177  for (auto p : index_range(div_phi[i]))
178  {
179  div_phi[i][p] = (dphi_dxi[i][p](0) + dphi_deta[i][p](1) + dphi_dzeta[i][p](2))/J[p];
180  }
181 
182  break;
183  }
184 
185  default:
186  libmesh_error_msg("Invalid dim = " << dim);
187  } // switch(dim)
