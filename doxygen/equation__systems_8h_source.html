657 template <typename T_sys>
658 inline
659 T_sys & EquationSystems::add_system (std::string_view name)
660 {
661  if (!_systems.count(name))
662  {
663  const unsigned int sys_num = this->n_systems();
664 
665  auto result = _systems.emplace
666  (name, std::make_unique<T_sys>(*this, std::string(name),
667  sys_num));
668 
669  if (!_enable_default_ghosting)
670  this->_remove_default_ghosting(sys_num);
671 
672  // Tell all the \p DofObject entities to add a system.
673  this->_add_system_to_nodes_and_elems();
674 
675  // Return reference to newly added item
676  auto it = result.first;
677  auto & sys_ptr = it->second;
678  return cast_ref<T_sys &>(*sys_ptr);
679  }
680  else
681  {
682  // We now allow redundant add_system calls, to make it
683  // easier to load data from files for user-derived system
684  // subclasses
685  return this->get_system<T_sys>(name);
686  }
687 }

702 template <typename T_sys>
703 inline
704 const T_sys & EquationSystems::get_system (const unsigned int num) const
705 {
706  libmesh_assert_less (num, this->n_systems());
707 
708  for (auto & pr : _systems)
709  {
710  const auto & sys_ptr = pr.second;
711  if (sys_ptr->number() == num)
712  return cast_ref<const T_sys &>(*sys_ptr);
713  }
714  // Error if we made it here
715  libmesh_error_msg("ERROR: no system number " << num << " found!");
716 }
