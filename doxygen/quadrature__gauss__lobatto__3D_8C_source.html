28 void QGaussLobatto::init_3D(const ElemType, unsigned int)
29 {
30  switch (_type)
31  {
32  case HEX8:
33  case HEX20:
34  case HEX27:
35  {
36  // We compute the 3D quadrature rule as a tensor
37  // product of the 1D quadrature rule.
38  QGaussLobatto q1D(1, _order);
39  q1D.init(EDGE2, _p_level);
40  tensor_product_hex(q1D);
41  return;
42  }
43 
44  // We fall back on a Gauss type rule for other types of elements,
45  // but we warn the user (once) that we are doing this instead of
46  // silently switching out quadrature rules on them.
47  default:
48  {
49  libmesh_warning("Warning: QGaussLobatto falling back on QGauss rule "
50  "for unsupported Elem type: " << Utility::enum_to_string(_type));
51 
52  QGauss gauss_rule(_dim, _order);
53  gauss_rule.init(_type, _p_level);
54 
55  // Swap points and weights with the about-to-be destroyed rule.
56  _points.swap (gauss_rule.get_points() );
57  _weights.swap(gauss_rule.get_weights());
58  }
59  }
60 }
