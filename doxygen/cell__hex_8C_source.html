239 Real Hex::quality (const ElemQuality q) const
240 {
241  switch (q)
242  {
243 
244 #if LIBMESH_DIM >= 3
245 
249  case DIAGONAL:
250  {
251  // Diagonal between node 0 and node 6
252  const Real d06 = this->length(0,6);
253 
254  // Diagonal between node 3 and node 5
255  const Real d35 = this->length(3,5);
256 
257  // Diagonal between node 1 and node 7
258  const Real d17 = this->length(1,7);
259 
260  // Diagonal between node 2 and node 4
261  const Real d24 = this->length(2,4);
262 
263  // Find the biggest and smallest diagonals
264  const Real min = std::min(d06, std::min(d35, std::min(d17, d24)));
265  const Real max = std::max(d06, std::max(d35, std::max(d17, d24)));
266 
267  libmesh_assert_not_equal_to (max, 0.0);
268 
269  return min / max;
270 
271  break;
272  }
273 
278  case TAPER:
279  {
280 
284  const Real d01 = this->length(0,1);
285  const Real d12 = this->length(1,2);
286  const Real d23 = this->length(2,3);
287  const Real d03 = this->length(0,3);
288  const Real d45 = this->length(4,5);
289  const Real d56 = this->length(5,6);
290  const Real d67 = this->length(6,7);
291  const Real d47 = this->length(4,7);
292  const Real d04 = this->length(0,4);
293  const Real d15 = this->length(1,5);
294  const Real d37 = this->length(3,7);
295  const Real d26 = this->length(2,6);
296 
297  std::vector<Real> edge_ratios(12);
298  // Front
299  edge_ratios[0] = std::min(d01, d45) / std::max(d01, d45);
300  edge_ratios[1] = std::min(d04, d15) / std::max(d04, d15);
301 
302  // Right
303  edge_ratios[2] = std::min(d15, d26) / std::max(d15, d26);
304  edge_ratios[3] = std::min(d12, d56) / std::max(d12, d56);
305 
306  // Back
307  edge_ratios[4] = std::min(d67, d23) / std::max(d67, d23);
308  edge_ratios[5] = std::min(d26, d37) / std::max(d26, d37);
309 
310  // Left
311  edge_ratios[6] = std::min(d04, d37) / std::max(d04, d37);
312  edge_ratios[7] = std::min(d03, d47) / std::max(d03, d47);
313 
314  // Bottom
315  edge_ratios[8] = std::min(d01, d23) / std::max(d01, d23);
316  edge_ratios[9] = std::min(d03, d12) / std::max(d03, d12);
317 
318  // Top
319  edge_ratios[10] = std::min(d45, d67) / std::max(d45, d67);
320  edge_ratios[11] = std::min(d56, d47) / std::max(d56, d47);
321 
322  return *(std::min_element(edge_ratios.begin(), edge_ratios.end())) ;
323 
324  break;
325  }
326 
327 
332  case STRETCH:
333  {
334  const Real sqrt3 = 1.73205080756888;
335 
339  const Real d06 = this->length(0,6);
340  const Real d17 = this->length(1,7);
341  const Real d35 = this->length(3,5);
342  const Real d24 = this->length(2,4);
343  const Real max_diag = std::max(d06, std::max(d17, std::max(d35, d24)));
344 
345  libmesh_assert_not_equal_to ( max_diag, 0.0 );
346 
350  std::vector<Real> edges(12);
351  edges[0] = this->length(0,1);
352  edges[1] = this->length(1,2);
353  edges[2] = this->length(2,3);
354  edges[3] = this->length(0,3);
355  edges[4] = this->length(4,5);
356  edges[5] = this->length(5,6);
357  edges[6] = this->length(6,7);
358  edges[7] = this->length(4,7);
359  edges[8] = this->length(0,4);
360  edges[9] = this->length(1,5);
361  edges[10] = this->length(2,6);
362  edges[11] = this->length(3,7);
363 
364  const Real min_edge = *(std::min_element(edges.begin(), edges.end()));
365  return sqrt3 * min_edge / max_diag ;
366  }
367 
368 
369  case SHAPE:
370  case SKEW:
371  {
372  // From: P. Knupp, "Algebraic mesh quality metrics for
373  // unstructured initial meshes," Finite Elements in Analysis
374  // and Design 39, 2003, p. 217-241, Sections 6.2 and 6.3.
375 
376  // Make local copies of points, we will access these several
377  // times below.
378  const Point
379  x0 = point(0), x1 = point(1), x2 = point(2), x3 = point(3),
380  x4 = point(4), x5 = point(5), x6 = point(6), x7 = point(7);
381 
382  // The columns of the Jacobian matrices are:
383  // \vec{x}_{\xi} = \vec{a1}*eta*zeta + \vec{b1}*eta + \vec{c1}*zeta + \vec{d1}
384  // \vec{x}_{\eta} = \vec{a2}*xi*zeta + \vec{b2}*xi + \vec{c2}*zeta + \vec{d2}
385  // \vec{x}_{\zeta} = \vec{a3}*xi*eta + \vec{b3}*xi + \vec{c3}*eta + \vec{d3}
386  // where the ai, bi, ci, and di are constants defined below.
387  const Point a1 = -x0 + x1 - x2 + x3 + x4 - x5 + x6 - x7;
388  const Point b1 = x0 - x1 + x2 - x3 + x4 - x5 + x6 - x7;
389  const Point c1 = x0 - x1 - x2 + x3 - x4 + x5 + x6 - x7;
390  const Point d1 = -x0 + x1 + x2 - x3 - x4 + x5 + x6 - x7;
391 
392  const Point a2 = a1;
393  const Point b2 = b1;
394  const Point c2 = x0 + x1 - x2 - x3 - x4 - x5 + x6 + x7;
395  const Point d2 = -x0 - x1 + x2 + x3 - x4 - x5 + x6 + x7;
396 
397  const Point a3 = a1;
398  const Point b3 = c1;
399  const Point c3 = c2;
400  const Point d3 = -x0 - x1 - x2 - x3 + x4 + x5 + x6 + x7;
401 
402  // Form the nodal Jacobians. These were computed using a
403  // Python script and the formulas above. Note that we are
404  // actually computing the Jacobian _columns_ and passing them
405  // to the RealTensor constructor which expects _rows_, but
406  // it's OK because we are only interested in determinants and
407  // products which are not affected by taking the transpose.
408  std::array<RealTensor, 8> A =
409  {{
410  RealTensor(d1, d2, d3),
411  RealTensor(d1, b2 + d2, b3 + d3),
412  RealTensor(b1 + d1, b2 + d2, a3 + b3 + c3 + d3),
413  RealTensor(b1 + d1, d2, c3 + d3),
414  RealTensor(c1 + d1, c2 + d2, d3),
415  RealTensor(c1 + d1, a2 + b2 + c2 + d2, b3 + d3),
416  RealTensor(a1 + b1 + c1 + d1, a2 + b2 + c2 + d2, a3 + b3 + c3 + d3),
417  RealTensor(a1 + b1 + c1 + d1, c2 + d2, c3 + d3)
418  }};
419 
420  // Compute Nodal areas, alpha_k = det(A_k).
421  // If any of these are zero or negative, we return zero
422  // (lowest possible value) for the quality, since the formulas
423  // below require positive nodal areas.
424  std::array<Real, 8> alpha;
425  for (unsigned int k=0; k<alpha.size(); ++k)
426  {
427  alpha[k] = A[k].det();
428  if (alpha[k] <= 0.)
429  return 0.;
430  }
431 
432  // Compute metric tensors, T_k = A_k^T * A_k.
433  std::array<RealTensor, 8> T;
434  for (unsigned int k=0; k<T.size(); ++k)
435  T[k] = A[k] * A[k].transpose();
436 
437  // Compute and return the shape metric. These only use the
438  // diagonal entries of the T_k.
439  Real den = 0.;
440  if (q == SHAPE)
441  {
442  for (unsigned int k=0; k<T.size(); ++k)
443  den += T[k].tr() / std::pow(alpha[k], 2./3.);
444  return (den == 0.) ? 0 : (24. / den);
445  }
446  else
447  {
448  for (unsigned int k=0; k<T.size(); ++k)
449  den += std::pow(std::sqrt(T[k](0,0) * T[k](1,1) * T[k](2,2)) / alpha[k], 2./3.);
450  return (den == 0.) ? 0 : (8. / den);
451  }
452  }
453 #endif // LIBMESH_DIM >= 3
454 
459  default:
460  return Elem::quality(q);
461  }
462 }
