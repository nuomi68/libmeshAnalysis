105 template <unsigned int Dim, FEFamily T>
106 void FE<Dim,T>::reinit(const Elem * elem,
107  const unsigned int s,
108  const Real /* tolerance */,
109  const std::vector<Point> * const pts,
110  const std::vector<Real> * const weights)
111 {
112  libmesh_assert(elem);
113  libmesh_assert (this->qrule != nullptr || pts != nullptr);
114  // We now do this for 1D elements!
115  // libmesh_assert_not_equal_to (Dim, 1);
116 
117  // If we called this function redundantly (e.g. in an FEMContext
118  // that is asked not to do any side calculations) then let's skip the
119  // whole inverse_map process that calculates side points
120  if (this->calculating_nothing())
121  {
122  this->calculations_started = true; // Ironic
123  return;
124  }
125 
126  // We're (possibly re-) calculating now! FIXME - we currently
127  // expect to be able to use side_map and JxW later, but we could
128  // optimize further here.
129  this->_fe_map->add_calculations();
130  this->_fe_map->get_JxW();
131  this->_fe_map->get_xyz();
132  this->determine_calculations();
133 
134  // Build the side of interest
135  const std::unique_ptr<const Elem> side(elem->build_side_ptr(s));
136 
137  // Find the max p_level to select
138  // the right quadrature rule for side integration
139  unsigned int side_p_level = elem->p_level();
140  if (elem->neighbor_ptr(s) != nullptr)
141  side_p_level = std::max(side_p_level, elem->neighbor_ptr(s)->p_level());
142 
143  // Initialize the shape functions at the user-specified
144  // points
145  if (pts != nullptr)
146  {
147  // The shape functions do not correspond to the qrule
148  this->shapes_on_quadrature = false;
149 
150  // Initialize the face shape functions
151  this->_fe_map->template init_face_shape_functions<Dim>(*pts, side.get());
152 
153  // Compute the Jacobian*Weight on the face for integration
154  if (weights != nullptr)
155  {
156  this->_fe_map->compute_face_map (Dim, *weights, side.get());
157  }
158  else
159  {
160  std::vector<Real> dummy_weights (pts->size(), 1.);
161  this->_fe_map->compute_face_map (Dim, dummy_weights, side.get());
162  }
163  }
164  // If there are no user specified points, we use the
165  // quadrature rule
166  else
167  {
168  // initialize quadrature rule
169  this->qrule->init(side->type(), side_p_level);
170 
171  if (this->qrule->shapes_need_reinit())
172  this->shapes_on_quadrature = false;
173 
174  // FIXME - could this break if the same FE object was used
175  // for both volume and face integrals? - RHS
176  // We might not need to reinitialize the shape functions
177  if ((this->get_type() != elem->type()) ||
178  (side->type() != last_side) ||
179  (this->_elem_p_level != side_p_level) ||
180  this->shapes_need_reinit() ||
181  !this->shapes_on_quadrature)
182  {
183  // Set the element type and p_level
184  this->elem_type = elem->type();
185  this->_elem_p_level = side_p_level;
186 
187  // Set the last_side
188  last_side = side->type();
189 
190  // Set the last p level
191  this->_p_level = this->_add_p_level_in_reinit * side_p_level;
192 
193  // Initialize the face shape functions
194  this->_fe_map->template init_face_shape_functions<Dim>(this->qrule->get_points(), side.get());
195  }
196 
197  // Compute the Jacobian*Weight on the face for integration
198  this->_fe_map->compute_face_map (Dim, this->qrule->get_weights(), side.get());
199 
200  // The shape functions correspond to the qrule
201  this->shapes_on_quadrature = true;
202  }
203 
204  // make a copy of the Jacobian for integration
205  const std::vector<Real> JxW_int(this->_fe_map->get_JxW());
206 
207  // make a copy of shape on quadrature info
208  bool shapes_on_quadrature_side = this->shapes_on_quadrature;
209 
210  // Find where the integration points are located on the
211  // full element.
212  const std::vector<Point> * ref_qp;
213  if (pts != nullptr)
214  ref_qp = pts;
215  else
216  ref_qp = &this->qrule->get_points();
217 
218  std::vector<Point> qp;
219  this->side_map(elem, side.get(), s, *ref_qp, qp);
220 
221  // compute the shape function and derivative values
222  // at the points qp
223  this->reinit (elem, &qp);
224 
225  this->shapes_on_quadrature = shapes_on_quadrature_side;
226 
227  // copy back old data
228  this->_fe_map->get_JxW() = JxW_int;
229 }

332 template <unsigned int Dim, FEFamily T>
333 void FE<Dim,T>::side_map (const Elem * elem,
334  const Elem * side,
335  const unsigned int s,
336  const std::vector<Point> & reference_side_points,
337  std::vector<Point> & reference_points)
338 {
339  // We're calculating mappings - we need at least first order info
340  this->calculate_phi = true;
341  this->determine_calculations();
342 
343  unsigned int side_p_level = elem->p_level();
344  if (elem->neighbor_ptr(s) != nullptr)
345  side_p_level = std::max(side_p_level, elem->neighbor_ptr(s)->p_level());
346 
347  if (side->type() != last_side ||
348  side_p_level != this->_elem_p_level ||
349  !this->shapes_on_quadrature)
350  {
351  // Set the element type
352  this->elem_type = elem->type();
353  this->_elem_p_level = side_p_level;
354  this->_p_level = this->_add_p_level_in_reinit * side_p_level;
355 
356  // Set the last_side
357  last_side = side->type();
358 
359  // Initialize the face shape functions
360  this->_fe_map->template init_face_shape_functions<Dim>(reference_side_points, side);
361  }
362 
363  const unsigned int n_points =
364  cast_int<unsigned int>(reference_side_points.size());
365  reference_points.resize(n_points);
366  for (unsigned int i = 0; i < n_points; i++)
367  reference_points[i].zero();
368 
369  std::vector<Point> refspace_nodes;
370  this->get_refspace_nodes(elem->type(), refspace_nodes);
371 
372  const std::vector<std::vector<Real>> & psi_map = this->_fe_map->get_psi();
373 
374  // sum over the nodes
375  for (auto i : index_range(psi_map))
376  {
377  const Point & side_node = refspace_nodes[elem->local_side_node(s,i)];
378  for (unsigned int p=0; p<n_points; p++)
379  reference_points[p].add_scaled (side_node, psi_map[i][p]);
380  }

433 template<unsigned int Dim>
434 void FEMap::init_face_shape_functions(const std::vector<Point> & qp,
435  const Elem * side)
436 {
437  // Start logging the shape function initialization
438  LOG_SCOPE("init_face_shape_functions()", "FEMap");
439 
440  libmesh_assert(side);
441 
442  // We're calculating now!
443  this->determine_calculations();
444 
445  // The element type and order to use in
446  // the map
447  const FEFamily mapping_family = FEMap::map_fe_type(*side);
448  const FEType map_fe_type(side->default_order(), mapping_family);
449 
450  // The number of quadrature points.
451  const unsigned int n_qp = cast_int<unsigned int>(qp.size());
452 
453  // Do not use the p_level(), if any, that is inherited by the side.
454  const unsigned int n_mapping_shape_functions =
455  FEInterface::n_shape_functions(map_fe_type, /*extra_order=*/0, side);
456 
457  // resize the vectors to hold current data
458  // Psi are the shape functions used for the FE mapping
459  if (calculate_xyz)
460  this->psi_map.resize (n_mapping_shape_functions);
461 
462  if (Dim > 1)
463  {
464  if (calculate_dxyz)
465  this->dpsidxi_map.resize (n_mapping_shape_functions);
466 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
467  if (calculate_d2xyz)
468  this->d2psidxi2_map.resize (n_mapping_shape_functions);
469 #endif
470  }
471 
472  if (Dim == 3)
473  {
474  if (calculate_dxyz)
475  this->dpsideta_map.resize (n_mapping_shape_functions);
476 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
477  if (calculate_d2xyz)
478  {
479  this->d2psidxideta_map.resize (n_mapping_shape_functions);
480  this->d2psideta2_map.resize (n_mapping_shape_functions);
481  }
482 #endif
483  }
484 
485  FEInterface::shape_ptr shape_ptr =
486  FEInterface::shape_function(map_fe_type, side);
487 
488  FEInterface::shape_deriv_ptr shape_deriv_ptr =
489  FEInterface::shape_deriv_function(map_fe_type, side);
490 
491 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
492  FEInterface::shape_second_deriv_ptr shape_second_deriv_ptr =
493  FEInterface::shape_second_deriv_function(map_fe_type, side);
494 #endif
495 
496  for (unsigned int i=0; i<n_mapping_shape_functions; i++)
497  {
498  // Allocate space to store the values of the shape functions
499  // and their first and second derivatives at the quadrature points.
500  if (calculate_xyz)
501  this->psi_map[i].resize (n_qp);
502  if (Dim > 1)
503  {
504  if (calculate_dxyz)
505  this->dpsidxi_map[i].resize (n_qp);
506 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
507  if (calculate_d2xyz)
508  this->d2psidxi2_map[i].resize (n_qp);
509 #endif
510  }
511  if (Dim == 3)
512  {
513  if (calculate_dxyz)
514  this->dpsideta_map[i].resize (n_qp);
515 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
516  if (calculate_d2xyz)
517  {
518  this->d2psidxideta_map[i].resize (n_qp);
519  this->d2psideta2_map[i].resize (n_qp);
520  }
521 #endif
522  }
523 
524 
525  // Compute the value of mapping shape function i, and its first
526  // and second derivatives at quadrature point p
527  for (unsigned int p=0; p<n_qp; p++)
528  {
529  if (calculate_xyz)
530  this->psi_map[i][p] = shape_ptr (map_fe_type, side, i, qp[p], false);
531  if (Dim > 1)
532  {
533  if (calculate_dxyz)
534  this->dpsidxi_map[i][p] = shape_deriv_ptr (map_fe_type, side, i, 0, qp[p], false);
535 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
536  if (calculate_d2xyz)
537  this->d2psidxi2_map[i][p] = shape_second_deriv_ptr(map_fe_type, side, i, 0, qp[p], false);
538 #endif
539  }
540  // libMesh::out << "this->d2psidxi2_map["<<i<<"][p]=" << d2psidxi2_map[i][p] << std::endl;
541 
542  // If we are in 3D, then our sides are 2D faces.
543  // For the second derivatives, we must also compute the cross
544  // derivative d^2() / dxi deta
545  if (Dim == 3)
546  {
547  if (calculate_dxyz)
548  this->dpsideta_map[i][p] = shape_deriv_ptr (map_fe_type, side, i, 1, qp[p], false);
549 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
550  if (calculate_d2xyz)
551  {
552  this->d2psidxideta_map[i][p] = shape_second_deriv_ptr(map_fe_type, side, i, 1, qp[p], false);
553  this->d2psideta2_map[i][p] = shape_second_deriv_ptr(map_fe_type, side, i, 2, qp[p], false);
554  }
555 #endif
556  }
557  }
558  }

583  FEInterface::n_shape_functions(map_fe_type, /*extra_order=*/0, edge);
584 
585  // resize the vectors to hold current data
586  // Psi are the shape functions used for the FE mapping
587  if (calculate_xyz)
588  this->psi_map.resize (n_mapping_shape_functions);
589  if (calculate_dxyz)
590  this->dpsidxi_map.resize (n_mapping_shape_functions);
591 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
592  if (calculate_d2xyz)
593  this->d2psidxi2_map.resize (n_mapping_shape_functions);
594 #endif
595 
596  FEInterface::shape_ptr shape_ptr =
597  FEInterface::shape_function(map_fe_type, edge);
598 
599  FEInterface::shape_deriv_ptr shape_deriv_ptr =
600  FEInterface::shape_deriv_function(map_fe_type, edge);
601 
602 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
603  FEInterface::shape_second_deriv_ptr shape_second_deriv_ptr =
604  FEInterface::shape_second_deriv_function(map_fe_type, edge);
605 #endif // LIBMESH_ENABLE_SECOND_DERIVATIVES
606 
607  for (unsigned int i=0; i<n_mapping_shape_functions; i++)
608  {
609  // Allocate space to store the values of the shape functions
610  // and their first and second derivatives at the quadrature points.
611  if (calculate_xyz)
612  this->psi_map[i].resize (n_qp);
613  if (calculate_dxyz)
614  this->dpsidxi_map[i].resize (n_qp);
615 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
616  if (calculate_d2xyz)
617  this->d2psidxi2_map[i].resize (n_qp);
618 #endif
619 
620  // Compute the value of mapping shape function i, and its first
621  // and second derivatives at quadrature point p
622  for (unsigned int p=0; p<n_qp; p++)
623  {
624  if (calculate_xyz)
625  this->psi_map[i][p] = shape_ptr (map_fe_type, edge, i, qp[p], false);
626  if (calculate_dxyz)
627  this->dpsidxi_map[i][p] = shape_deriv_ptr (map_fe_type, edge, i, 0, qp[p], false);
628 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
629  if (calculate_d2xyz)
630  this->d2psidxi2_map[i][p] = shape_second_deriv_ptr(map_fe_type, edge, i, 0, qp[p], false);
631 #endif
632  }
633  }

638 void FEMap::compute_face_map(int dim, const std::vector<Real> & qw,
639  const Elem * side)
640 {
641  libmesh_assert(side);
642 
643  // We're calculating now!
644  this->determine_calculations();
645 
646  LOG_SCOPE("compute_face_map()", "FEMap");
647 
648  // The number of quadrature points.
649  const unsigned int n_qp = cast_int<unsigned int>(qw.size());
650 
651  const FEFamily mapping_family = FEMap::map_fe_type(*side);
652  const Order mapping_order (side->default_order());
653  const FEType map_fe_type(mapping_order, mapping_family);
654 
655  // Do not use the p_level(), if any, that is inherited by the side.
656  const unsigned int n_mapping_shape_functions =
657  FEInterface::n_shape_functions(map_fe_type, /*extra_order=*/0, side);
658 
659  switch (dim)
660  {
661  case 1:
662  {
663  // A 1D finite element, currently assumed to be in 1D space
664  // This means the boundary is a "0D finite element", a
665  // NODEELEM.
666 
667  // Resize the vectors to hold data at the quadrature points
668  {
669  if (calculate_xyz)
670  this->xyz.resize(n_qp);
671  if (calculate_dxyz)
672  normals.resize(n_qp);
673 
674  if (calculate_dxyz)
675  this->JxW.resize(n_qp);
676  }
677 
678  // If we have no quadrature points, there's nothing else to do
679  if (!n_qp)
680  break;
681 
682  // We need to look back at the full edge to figure out the normal
683  // vector
684  const Elem * elem = side->interior_parent();
685  libmesh_assert (elem);
686  if (calculate_dxyz)
687  {
688  if (side->node_id(0) == elem->node_id(0))
689  normals[0] = Point(-1.);
690  else
691  {
692  libmesh_assert_equal_to (side->node_id(0),
693  elem->node_id(1));
694  normals[0] = Point(1.);
695  }
696  }
697 
698  // Calculate x at the point
699  if (calculate_xyz)
700  libmesh_assert_equal_to (this->psi_map.size(), 1);
701  // In the unlikely event we have multiple quadrature
702  // points, they'll be in the same place
703  for (unsigned int p=0; p<n_qp; p++)
704  {
705  if (calculate_xyz)
706  {
707  this->xyz[p].zero();
708  this->xyz[p].add_scaled (side->point(0), this->psi_map[0][p]);
709  }
710  if (calculate_dxyz)
711  {
712  normals[p] = normals[0];
713  this->JxW[p] = 1.0*qw[p];
714  }
715  }
716 
717  // done computing the map
718  break;
719  }
720 
721  case 2:
722  {
723  // A 2D finite element living in either 2D or 3D space.
724  // This means the boundary is a 1D finite element, i.e.
725  // and EDGE2 or EDGE3.
726  // Resize the vectors to hold data at the quadrature points
727  {
728  if (calculate_xyz)
729  this->xyz.resize(n_qp);
730  if (calculate_dxyz)
731  {
732  this->dxyzdxi_map.resize(n_qp);
733  this->tangents.resize(n_qp);
734  this->normals.resize(n_qp);
735 
736  this->JxW.resize(n_qp);
737  }
738 
739 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
740  if (calculate_d2xyz)
741  {
742  this->d2xyzdxi2_map.resize(n_qp);
743  this->curvatures.resize(n_qp);
744  }
745 #endif
746  }
747 
748  // Clear the entities that will be summed
749  // Compute the tangent & normal at the quadrature point
750  for (unsigned int p=0; p<n_qp; p++)
751  {
752  if (calculate_xyz)
753  this->xyz[p].zero();
754  if (calculate_dxyz)
755  {
756  this->tangents[p].resize(LIBMESH_DIM-1); // 1 Tangent in 2D, 2 in 3D
757  this->dxyzdxi_map[p].zero();
758  }
759 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
760  if (calculate_d2xyz)
761  this->d2xyzdxi2_map[p].zero();
762 #endif
763  }
764 
765  // compute x, dxdxi at the quadrature points
766  for (unsigned int i=0; i<n_mapping_shape_functions; i++) // sum over the nodes
767  {
768  const Point & side_point = side->point(i);
769 
770  for (unsigned int p=0; p<n_qp; p++) // for each quadrature point...
771  {
772  if (calculate_xyz)
773  this->xyz[p].add_scaled (side_point, this->psi_map[i][p]);
774  if (calculate_dxyz)
775  this->dxyzdxi_map[p].add_scaled (side_point, this->dpsidxi_map[i][p]);
776 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
777  if (calculate_d2xyz)
778  this->d2xyzdxi2_map[p].add_scaled(side_point, this->d2psidxi2_map[i][p]);
779 #endif
780  }
781  }
782 
783  // Compute the tangent & normal at the quadrature point
784  if (calculate_dxyz)
785  {
786  for (unsigned int p=0; p<n_qp; p++)
787  {
788  // The first tangent comes from just the edge's Jacobian
789  this->tangents[p][0] = this->dxyzdxi_map[p].unit();
790 
791 #if LIBMESH_DIM == 2
792  // For a 2D element living in 2D, the normal is given directly
793  // from the entries in the edge Jacobian.
794  this->normals[p] = (Point(this->dxyzdxi_map[p](1), -this->dxyzdxi_map[p](0), 0.)).unit();
795 
796 #elif LIBMESH_DIM == 3
797  // For a 2D element living in 3D, there is a second tangent.
798  // For the second tangent, we need to refer to the full
799  // element's (not just the edge's) Jacobian.
800  const Elem * elem = side->interior_parent();
801  libmesh_assert(elem);
802 
803  // Inverse map xyz[p] to a reference point on the parent...
804  Point reference_point = FEMap::inverse_map(2, elem, this->xyz[p]);
805 
806  // Get dxyz/dxi and dxyz/deta from the parent map.
807  Point dx_dxi = FEMap::map_deriv (2, elem, 0, reference_point);
808  Point dx_deta = FEMap::map_deriv (2, elem, 1, reference_point);
809 
810  // The second tangent vector is formed by crossing these vectors.
811  tangents[p][1] = dx_dxi.cross(dx_deta).unit();
812 
813  // Finally, the normal in this case is given by crossing these
814  // two tangents.
815  normals[p] = tangents[p][0].cross(tangents[p][1]).unit();
816 #endif
817 
818 
819 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
820  // The curvature is computed via the familiar Frenet formula:
821  // curvature = [d^2(x) / d (xi)^2] dot [normal]
822  // For a reference, see:
823  // F.S. Merritt, Mathematics Manual, 1962, McGraw-Hill, p. 310
824  //
825  // Note: The sign convention here is different from the
826  // 3D case. Concave-upward curves (smiles) have a positive
827  // curvature. Concave-downward curves (frowns) have a
828  // negative curvature. Be sure to take that into account!
829  if (calculate_d2xyz)
830  {
831  const Real numerator = this->d2xyzdxi2_map[p] * this->normals[p];
832  const Real denominator = this->dxyzdxi_map[p].norm_sq();
833  libmesh_assert_not_equal_to (denominator, 0);
834  curvatures[p] = numerator / denominator;
835  }
836 #endif
837  }
838 
839  // compute the jacobian at the quadrature points
840  for (unsigned int p=0; p<n_qp; p++)
841  {
842  const Real the_jac = this->dxyzdxi_map[p].norm();
843 
844  libmesh_assert_greater (the_jac, 0.);
845 
846  this->JxW[p] = the_jac*qw[p];
847  }
848  }
849 
850  // done computing the map
851  break;
852  }
853 
854 
855 
856  case 3:
857  {
858  // A 3D finite element living in 3D space.
859  // Resize the vectors to hold data at the quadrature points
860  {
861  if (calculate_xyz)
862  this->xyz.resize(n_qp);
863  if (calculate_dxyz)
864  {
865  this->dxyzdxi_map.resize(n_qp);
866  this->dxyzdeta_map.resize(n_qp);
867  this->tangents.resize(n_qp);
868  this->normals.resize(n_qp);
869  this->JxW.resize(n_qp);
870  }
871 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
872  if (calculate_d2xyz)
873  {
874  this->d2xyzdxi2_map.resize(n_qp);
875  this->d2xyzdxideta_map.resize(n_qp);
876  this->d2xyzdeta2_map.resize(n_qp);
877  this->curvatures.resize(n_qp);
878  }
879 #endif
880  }
881 
882  // Clear the entities that will be summed
883  for (unsigned int p=0; p<n_qp; p++)
884  {
885  if (calculate_xyz)
886  this->xyz[p].zero();
887  if (calculate_dxyz)
888  {
889  this->tangents[p].resize(LIBMESH_DIM-1); // 1 Tangent in 2D, 2 in 3D
890  this->dxyzdxi_map[p].zero();
891  this->dxyzdeta_map[p].zero();
892  }
893 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
894  if (calculate_d2xyz)
895  {
896  this->d2xyzdxi2_map[p].zero();
897  this->d2xyzdxideta_map[p].zero();
898  this->d2xyzdeta2_map[p].zero();
899  }
900 #endif
901  }
902 
903  // compute x, dxdxi at the quadrature points
904  for (unsigned int i=0; i<n_mapping_shape_functions; i++) // sum over the nodes
905  {
906  const Point & side_point = side->point(i);
907 
908  for (unsigned int p=0; p<n_qp; p++) // for each quadrature point...
909  {
910  if (calculate_xyz)
911  this->xyz[p].add_scaled (side_point, this->psi_map[i][p]);
912  if (calculate_dxyz)
913  {
914  this->dxyzdxi_map[p].add_scaled (side_point, this->dpsidxi_map[i][p]);
915  this->dxyzdeta_map[p].add_scaled(side_point, this->dpsideta_map[i][p]);
916  }
917 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
918  if (calculate_d2xyz)
919  {
920  this->d2xyzdxi2_map[p].add_scaled (side_point, this->d2psidxi2_map[i][p]);
921  this->d2xyzdxideta_map[p].add_scaled(side_point, this->d2psidxideta_map[i][p]);
922  this->d2xyzdeta2_map[p].add_scaled (side_point, this->d2psideta2_map[i][p]);
923  }
924 #endif
925  }
926  }
927 
928  // Compute the tangents, normal, and curvature at the quadrature point
929  if (calculate_dxyz)
930  {
931  for (unsigned int p=0; p<n_qp; p++)
932  {
933  const Point n = this->dxyzdxi_map[p].cross(this->dxyzdeta_map[p]);
934  this->normals[p] = n.unit();
935  this->tangents[p][0] = this->dxyzdxi_map[p].unit();
936  this->tangents[p][1] = n.cross(this->dxyzdxi_map[p]).unit();
937 
938 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
939  if (calculate_d2xyz)
940  {
941  // Compute curvature using the typical nomenclature
942  // of the first and second fundamental forms.
943  // For reference, see:
944  // 1) http://mathworld.wolfram.com/MeanCurvature.html
945  // (note -- they are using inward normal)
946  // 2) F.S. Merritt, Mathematics Manual, 1962, McGraw-Hill
947  const Real L = -this->d2xyzdxi2_map[p] * this->normals[p];
948  const Real M = -this->d2xyzdxideta_map[p] * this->normals[p];
949  const Real N = -this->d2xyzdeta2_map[p] * this->normals[p];
950  const Real E = this->dxyzdxi_map[p].norm_sq();
951  const Real F = this->dxyzdxi_map[p] * this->dxyzdeta_map[p];
952  const Real G = this->dxyzdeta_map[p].norm_sq();
953 
954  const Real numerator = E*N -2.*F*M + G*L;
955  const Real denominator = E*G - F*F;
956  libmesh_assert_not_equal_to (denominator, 0.);
957  curvatures[p] = 0.5*numerator/denominator;
958  }
959 #endif
960  }
961 
962  // compute the jacobian at the quadrature points, see
963  // http://sp81.msi.umn.edu:999/fluent/fidap/help/theory/thtoc.htm
964  for (unsigned int p=0; p<n_qp; p++)
965  {
966  const Real g11 = (dxdxi_map(p)*dxdxi_map(p) +
967  dydxi_map(p)*dydxi_map(p) +
968  dzdxi_map(p)*dzdxi_map(p));
969 
970  const Real g12 = (dxdxi_map(p)*dxdeta_map(p) +
971  dydxi_map(p)*dydeta_map(p) +
972  dzdxi_map(p)*dzdeta_map(p));
973 
974  const Real g21 = g12;
975 
976  const Real g22 = (dxdeta_map(p)*dxdeta_map(p) +
977  dydeta_map(p)*dydeta_map(p) +
978  dzdeta_map(p)*dzdeta_map(p));
979 
980 
981  const Real the_jac = std::sqrt(g11*g22 - g12*g21);
982 
983  libmesh_assert_greater (the_jac, 0.);
984 
985  this->JxW[p] = the_jac*qw[p];
986  }
987  }
988 
989  // done computing the map
990  break;
991  }
992 
993 
994  default:
995  libmesh_error_msg("Invalid dimension dim = " << dim);
996  }
