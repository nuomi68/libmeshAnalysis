110 std::unique_ptr<Elem> InfPrism6::build_side_ptr (const unsigned int i,
111  bool proxy)
112 {
113  libmesh_assert_less (i, this->n_sides());
114 
115  std::unique_ptr<Elem> face;
116  if (proxy)
117  {
118 #ifdef LIBMESH_ENABLE_DEPRECATED
119  libmesh_deprecated();
120  switch (i)
121  {
122  // base
123  case 0:
124  {
125  face = std::make_unique<Side<Tri3,InfPrism6>>(this,i);
126  break;
127  }
128 
129  // ifem sides
130  case 1:
131  case 2:
132  case 3:
133  {
134  face = std::make_unique<Side<InfQuad4,InfPrism6>>(this,i);
135  break;
136  }
137 
138  default:
139  libmesh_error_msg("Invalid side i = " << i);
140  }
141 #else
142  libmesh_error();
143 #endif // LIBMESH_ENABLE_DEPRECATED
144  }
145  else
146  {
147  switch (i)
148  {
149  case 0: // the triangular face at z=-1, base face
150  {
151  face = std::make_unique<Tri3>();
152  break;
153  }
154 
155  case 1: // the quad face at y=0
156  case 2: // the other quad face
157  case 3: // the quad face at x=0
158  {
159  face = std::make_unique<InfQuad4>();
160  break;
161  }
162 
163  default:
164  libmesh_error_msg("Invalid side i = " << i);
165  }
166 
167  // Set the nodes
168  for (auto n : face->node_index_range())
169  face->set_node(n) = this->node_ptr(InfPrism6::side_nodes_map[i][n]);
170  }
171 
172 #ifdef LIBMESH_ENABLE_DEPRECATED
173  if (!proxy) // proxy sides used to leave parent() set
174 #endif
175  face->set_parent(nullptr);
176  face->set_interior_parent(this);
177 
178  face->subdomain_id() = this->subdomain_id();
179  face->set_mapping_type(this->mapping_type());
180 #ifdef LIBMESH_ENABLE_AMR
181  face->set_p_level(this->p_level());
182 #endif
183 
184  return face;
185 }

257  const IOPackage iop,
258  std::vector<dof_id_type> & conn) const
259 {
260  libmesh_assert(_nodes);

273  conn[4] = this->node_id(3)+1;
