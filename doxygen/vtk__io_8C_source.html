61 // A convenient macro for comparing VTK versions. Returns 1 if the
62 // current VTK version is < major.minor.subminor and zero otherwise.
63 //
64 // It relies on the VTK version numbers detected during configure. Note that if
65 // LIBMESH_HAVE_VTK is not defined, none of the LIBMESH_DETECTED_VTK_VERSION_* variables will
66 // be defined either.
67 #define VTK_VERSION_LESS_THAN(major,minor,subminor) \
68  ((LIBMESH_DETECTED_VTK_VERSION_MAJOR < (major) || \
69  (LIBMESH_DETECTED_VTK_VERSION_MAJOR == (major) && (LIBMESH_DETECTED_VTK_VERSION_MINOR < (minor) || \
70  (LIBMESH_DETECTED_VTK_VERSION_MINOR == (minor) && \
71  LIBMESH_DETECTED_VTK_VERSION_SUBMINOR < (subminor))))) ? 1 : 0)
72 
73 #endif // LIBMESH_HAVE_VTK

124  std::map<ElemMappingType, VTKIO::ElementMaps> all_maps;

127  em.associate(EDGE2, VTK_LINE);

131  em.associate(QUAD4, VTK_QUAD);
132  em.associate(QUAD8, VTK_QUADRATIC_QUAD);

134  em.associate(TET10, VTK_QUADRATIC_TETRA);
135  em.associate(HEX8, VTK_HEXAHEDRON);
136  em.associate(HEX20, VTK_QUADRATIC_HEXAHEDRON);

138  em.associate(PRISM6, VTK_WEDGE);
139  em.associate(PRISM15, VTK_QUADRATIC_WEDGE);

184 void VTKIO::read (const std::string & name)
185 {
186  // This is a serial-only process for now;
187  // the Mesh should be read on processor 0 and
188  // broadcast later
189  libmesh_assert_equal_to (MeshOutput<MeshBase>::mesh().processor_id(), 0);
190 
191  // Keep track of what kinds of elements this file contains
192  elems_of_dimension.clear();
193  elems_of_dimension.resize(4, false);
194 
195  // Use a typedef, because these names are just crazy
196  typedef vtkSmartPointer<vtkXMLUnstructuredGridReader> MyReader;
197  MyReader reader = MyReader::New();
198 
199  // Pass the filename along to the reader
200  reader->SetFileName(name.c_str());
201 
202  // Force reading
203  reader->Update();
204 
205  // read in the grid
206  _vtk_grid = reader->GetOutput();
207 
208  // Get a reference to the mesh
209  MeshBase & mesh = MeshInput<MeshBase>::mesh();
210 
211  // Clear out any pre-existing data from the Mesh
212  mesh.clear();
213 
214  // Get the number of points from the _vtk_grid object
215  const unsigned int vtk_num_points = static_cast<unsigned int>(_vtk_grid->GetNumberOfPoints());
216 
217  // always numbered nicely so we can loop like this
218  for (unsigned int i=0; i<vtk_num_points; ++i)
219  {
220  // add to the id map
221  // and add the actual point
222  double pnt[3];
223  _vtk_grid->GetPoint(static_cast<vtkIdType>(i), pnt);
224  Point xyz(pnt[0], pnt[1], pnt[2]);
225  mesh.add_point(xyz, i);
226  }
227 
228  // Get the number of cells from the _vtk_grid object
229  const unsigned int vtk_num_cells = static_cast<unsigned int>(_vtk_grid->GetNumberOfCells());
230 
231  auto& element_map = libmesh_map_find(_element_maps, mesh.default_mapping_type());
232 
233  vtkSmartPointer<vtkGenericCell> cell = vtkSmartPointer<vtkGenericCell>::New();
234  for (unsigned int i=0; i<vtk_num_cells; ++i)
235  {
236  _vtk_grid->GetCell(i, cell);
237 
238  // Get the libMesh element type corresponding to this VTK element type.
239  ElemType libmesh_elem_type = element_map.find(cell->GetCellType());
240  auto elem = Elem::build(libmesh_elem_type);
241 
242  // get the straightforward numbering from the VTK cells
243  for (auto j : elem->node_index_range())
244  elem->set_node(j) =
245  mesh.node_ptr(cast_int<dof_id_type>(cell->GetPointId(j)));
246 
247  // then get the connectivity
248  std::vector<dof_id_type> conn;
249  elem->connectivity(0, VTK, conn);
250 
251  // then reshuffle the nodes according to the connectivity, this
252  // two-time-assign would evade the definition of the vtk_mapping
253  for (unsigned int j=0,
254  n_conn = cast_int<unsigned int>(conn.size());
255  j != n_conn; ++j)
256  elem->set_node(j) = mesh.node_ptr(conn[j]);
257 
258  elem->set_id(i);
259 
260  elems_of_dimension[elem->dim()] = true;
261 
262  mesh.add_elem(std::move(elem));
263  } // end loop over VTK cells
264 
265  // Set the mesh dimension to the largest encountered for an element
266  for (unsigned char i=0; i!=4; ++i)
267  if (elems_of_dimension[i])
268  mesh.set_mesh_dimension(i);
269 
270 #if LIBMESH_DIM < 3
271  libmesh_error_msg_if(mesh.mesh_dimension() > LIBMESH_DIM,
272  "Cannot open dimension "
273  << mesh.mesh_dimension()
274  << " mesh file when configured without "
275  << mesh.mesh_dimension()
276  << "D support.");
277 #endif // LIBMESH_DIM < 3
278 }

282 void VTKIO::write_nodal_data (const std::string & fname,
283  const std::vector<Number> & soln,
284  const std::vector<std::string> & names)
285 {
286  // Warn that the .pvtu file extension should be used. Paraview
287  // recognizes this, and it works in both serial and parallel. Only
288  // warn about this once.
289  if (fname.substr(fname.rfind("."), fname.size()) != ".pvtu")
290  libmesh_do_once(libMesh::err << "The .pvtu extension should be used when writing VTK files in libMesh.");
291 
292  // If there are variable names being written, the solution vector
293  // should not be empty, it should have been broadcast to all
294  // processors by the MeshOutput base class, since VTK is a parallel
295  // format. Verify this before going further.
296  libmesh_error_msg_if(!names.empty() && soln.empty(),
297  "Empty soln vector in VTKIO::write_nodal_data().");
298 
299  // Get a reference to the mesh
300  const MeshBase & mesh = MeshOutput<MeshBase>::mesh();
301 
302  // we only use Unstructured grids
303  _vtk_grid = vtkSmartPointer<vtkUnstructuredGrid>::New();
304  vtkSmartPointer<vtkXMLPUnstructuredGridWriter> writer = vtkSmartPointer<vtkXMLPUnstructuredGridWriter>::New();
305 #ifdef LIBMESH_HAVE_MPI
306  // Set VTK to the same communicator as libMesh
307  vtkSmartPointer<vtkMPICommunicator> vtk_comm = vtkSmartPointer<vtkMPICommunicator>::New();
308  MPI_Comm mpi_comm = mesh.comm().get();
309  vtkMPICommunicatorOpaqueComm vtk_opaque_comm(&mpi_comm);
310  vtk_comm->InitializeExternal(&vtk_opaque_comm);
311 
312  vtkSmartPointer<vtkMPIController> vtk_mpi_ctrl = vtkSmartPointer<vtkMPIController>::New();
313  vtk_mpi_ctrl->SetCommunicator(vtk_comm);
314 
315  writer->SetController(vtk_mpi_ctrl);
316 #endif
317 
318  // add nodes to the grid and update _local_node_map
319  _local_node_map.clear();
320  this->nodes_to_vtk();
321 
322  // add cells to the grid
323  this->cells_to_vtk();
324 
325  // add nodal solutions to the grid, if solutions are given
326  if (names.size() > 0)
327  {
328  std::size_t num_vars = names.size();
329  std::vector<Number> local_values;
330 
331 #ifdef LIBMESH_USE_COMPLEX_NUMBERS
332  std::vector<Real> local_real_values;
333 #endif
334 
335  for (std::size_t variable=0; variable<num_vars; ++variable)
336  {
337  get_local_node_values(local_values, variable, soln, names);
338 
339 #ifdef LIBMESH_USE_COMPLEX_NUMBERS
340  // write real part
341  local_real_values.resize(local_values.size());
342  std::transform(local_values.begin(), local_values.end(),
343  local_real_values.begin(),
344  [](Number x) { return x.real(); });
345  node_values_to_vtk(names[variable] + "_real", local_real_values);
346 
347  // write imaginary part
348  local_real_values.resize(local_values.size());
349  std::transform(local_values.begin(), local_values.end(),
350  local_real_values.begin(),
351  [](Number x) { return x.imag(); });
352  node_values_to_vtk(names[variable] + "_imag", local_real_values);
353 #else
354  node_values_to_vtk(names[variable], local_values);
355 #endif
356  }
357  }
358 
359  // Tell the writer how many partitions exist and on which processor
360  // we are currently
361  writer->SetNumberOfPieces(mesh.n_processors());
362  writer->SetStartPiece(mesh.processor_id());
363  writer->SetEndPiece(mesh.processor_id());
364 
365  // partitions overlap by one node
366  // FIXME: According to this document
367  // http://paraview.org/Wiki/images/5/51/SC07_tut107_ParaView_Handouts.pdf
368  // the ghosts are cells rather than nodes.
369  writer->SetGhostLevel(1);
370 
371  // VTK 6 replaces SetInput() with SetInputData(). See
372  // http://www.vtk.org/Wiki/VTK/VTK_6_Migration/Replacement_of_SetInput
373  // for the full explanation.
374 #if VTK_VERSION_LESS_THAN(6,0,0)
375  writer->SetInput(_vtk_grid);
376 #else
377  writer->SetInputData(_vtk_grid);
378 #endif
379 
380  writer->SetFileName(fname.c_str());
381  writer->SetDataModeToAscii();
382 
383  // compress the output, if desired (switches also to binary)
384  if (this->_compress)
385  {
386 #if !VTK_VERSION_LESS_THAN(5,6,0)
387  writer->SetCompressorTypeToZLib();
388  writer->SetDataModeToBinary();
389 #else
390  libmesh_do_once(libMesh::err << "Compression not implemented with old VTK libs!" << std::endl;);
391 #endif
392  }
393 
394  writer->Write();
395 
396 }

414 void VTKIO::nodes_to_vtk()
415 {
416  const MeshBase & mesh = MeshOutput<MeshBase>::mesh();
417 
418  // containers for points and coordinates of points
419  vtkSmartPointer<vtkPoints> points = vtkSmartPointer<vtkPoints>::New();
420  vtkSmartPointer<vtkDoubleArray> pcoords = vtkSmartPointer<vtkDoubleArray>::New();
421  // if this grid is to be used in VTK then the dimension of the points should be 3
422  pcoords->SetNumberOfComponents(LIBMESH_DIM);
423  pcoords->Allocate(3*mesh.n_local_nodes());
424  points->SetNumberOfPoints(mesh.n_local_nodes()); // it seems that it needs this to prevent a segfault
425 
426  // SetRationalWeights() was introduced in VTK 9.0
427 #if !VTK_VERSION_LESS_THAN(9,0,0)
428  bool have_weights = false;
429  int weight_index = 0;
430  vtkSmartPointer<vtkDoubleArray> rational_weights;
431 
432  if (mesh.default_mapping_type() == ElemMappingType::RATIONAL_BERNSTEIN_MAP)
433  {
434  rational_weights = vtkSmartPointer<vtkDoubleArray>::New();
435  rational_weights->SetName("RationalWeights");
436  rational_weights->SetNumberOfComponents(1);
437  weight_index = static_cast<int>(mesh.default_mapping_data());
438  have_weights = true;
439  }
440 #endif
441 
442  unsigned int local_node_counter = 0;
443 
444  for (const auto & node_ptr : mesh.local_node_ptr_range())
445  {
446  const Node & node = *node_ptr;
447 
448  double pnt[3] = {0, 0, 0};
449  for (unsigned int i=0; i<LIBMESH_DIM; ++i)
450  pnt[i] = node(i);
451 
452  // Fill mapping between global and local node numbers
453  _local_node_map[node.id()] = local_node_counter;
454 
455  // add point
456 #if VTK_VERSION_LESS_THAN(7,1,0)
457  pcoords->InsertNextTupleValue(pnt);
458 #else
459  pcoords->InsertNextTuple(pnt);
460 #endif
461 #if !VTK_VERSION_LESS_THAN(9,0,0)
462  if (have_weights)
463  {
464  Real weight = node.get_extra_datum<Real>(weight_index);
465  rational_weights->InsertTuple1(local_node_counter, weight);
466  }
467 #endif
468 
469  ++local_node_counter;
470  }
471 
472  // add coordinates to points
473  points->SetData(pcoords);
474 
475  // add points to grid
476  _vtk_grid->SetPoints(points);
477 #if !VTK_VERSION_LESS_THAN(9,0,0)
478  if (have_weights)
479  _vtk_grid->GetPointData()->SetRationalWeights(rational_weights);
480 
481 #endif
482 }

486 void VTKIO::cells_to_vtk()
487 {
488  const MeshBase & mesh = MeshOutput<MeshBase>::mesh();
489 
490  auto& element_map = libmesh_map_find(_element_maps, mesh.default_mapping_type());
491  vtkSmartPointer<vtkCellArray> cells = vtkSmartPointer<vtkCellArray>::New();
492  vtkSmartPointer<vtkIdList> pts = vtkSmartPointer<vtkIdList>::New();
493 
494  std::vector<int> types(mesh.n_active_local_elem());
495 
496  vtkSmartPointer<vtkIntArray> elem_id = vtkSmartPointer<vtkIntArray>::New();
497  elem_id->SetName("libmesh_elem_id");
498  elem_id->SetNumberOfComponents(1);
499 
500  vtkSmartPointer<vtkIntArray> subdomain_id = vtkSmartPointer<vtkIntArray>::New();
501  subdomain_id->SetName("subdomain_id");
502  subdomain_id->SetNumberOfComponents(1);
503 
504  vtkSmartPointer<vtkIntArray> elem_proc_id = vtkSmartPointer<vtkIntArray>::New();
505  elem_proc_id->SetName("processor_id");
506  elem_proc_id->SetNumberOfComponents(1);
507 
508  unsigned active_element_counter = 0;
509  for (const auto & elem : mesh.active_local_element_ptr_range())
510  {
511  // When using rational bernstein these hold the weights
512  if ( elem->type() == NODEELEM )
513  continue;
514 
515  pts->SetNumberOfIds(elem->n_nodes());
516 
517  // get the connectivity for this element
518  std::vector<dof_id_type> conn;
519  elem->connectivity(0, VTK, conn);
520 
521  for (unsigned int i=0,
522  n_conn = cast_int<unsigned int>(conn.size());
523  i != n_conn; ++i)
524  {
525  // If the node ID is not found in the _local_node_map, we'll
526  // add it to the _vtk_grid. NOTE[JWP]: none of the examples
527  // I have actually enters this section of code...
528  if (_local_node_map.find(conn[i]) == _local_node_map.end())
529  {
530  dof_id_type global_node_id = elem->node_id(i);
531 
532  const Point & the_node = mesh.point(global_node_id);
533 
534  // InsertNextPoint accepts either a double or float array of length 3.
535  double pt[3] = {0., 0., 0.};
536  for (unsigned int d=0; d<LIBMESH_DIM; ++d)
537  pt[d] = the_node(d);
538 
539  // Insert the point into the _vtk_grid
540  vtkIdType local = _vtk_grid->GetPoints()->InsertNextPoint(pt);
541 
542  // Update the _local_node_map with the ID returned by VTK
543  _local_node_map[global_node_id] =
544  cast_int<dof_id_type>(local);
545  }
546 
547  // Otherwise, the node ID was found in the _local_node_map, so
548  // insert it into the vtkIdList.
549  pts->InsertId(i, _local_node_map[conn[i]]);
550  }
551 
552  vtkIdType vtkcellid = cells->InsertNextCell(pts);
553  types[active_element_counter] = cast_int<int>(element_map.find(elem->type()));
554 
555  elem_id->InsertTuple1(vtkcellid, elem->id());
556  subdomain_id->InsertTuple1(vtkcellid, elem->subdomain_id());
557  elem_proc_id->InsertTuple1(vtkcellid, elem->processor_id());
558  ++active_element_counter;
559  } // end loop over active elements
560 
561  _vtk_grid->SetCells(types.data(), cells);
562  _vtk_grid->GetCellData()->AddArray(elem_id);
563  _vtk_grid->GetCellData()->AddArray(subdomain_id);
564  _vtk_grid->GetCellData()->AddArray(elem_proc_id);
565 }

567 void VTKIO::node_values_to_vtk(const std::string & name,
568  const std::vector<Real> & local_values)
569 {
570  vtkSmartPointer<vtkDoubleArray> data = vtkSmartPointer<vtkDoubleArray>::New();
571  data->SetName(name.c_str());
572 
573  libmesh_assert_equal_to(_local_node_map.size(), local_values.size());
574 
575  // number of local and ghost nodes
576  data->SetNumberOfValues(_local_node_map.size());
577 
578  // copy values into vtk
579  for (auto i : index_range(local_values)) {
580  data->SetValue(i, local_values[i]);
581  }
582
