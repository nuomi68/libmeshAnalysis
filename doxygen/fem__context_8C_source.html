85 FEType FEMContext::find_hardest_fe_type()
86 {
87  const System & sys = this->get_system();
88  FEType hardest_fe_type = sys.variable_type(0);
89 
90  auto check_var = [&hardest_fe_type, &sys](unsigned int v)
91  {
92  FEType fe_type = sys.variable_type(v);
93 
94  // Make sure we find a non-SCALAR FE family, even in the case
95  // where the first variable(s) weren't
96  if (hardest_fe_type.family == SCALAR)
97  {
98  hardest_fe_type.family = fe_type.family;
99  hardest_fe_type.order = fe_type.order;
100  }
101 
102  // FIXME - we don't yet handle mixed finite elements from
103  // different families which require different quadrature rules
104  // libmesh_assert_equal_to (fe_type.family, hardest_fe_type.family);
105 
106  // We need to detect SCALAR's so we can prepare FE objects for
107  // them, and so we don't mistake high order scalars as a reason
108  // to crank up the quadrature order on other types.
109  if (fe_type.family != SCALAR && fe_type.order > hardest_fe_type.order)
110  hardest_fe_type = fe_type;
111  };
112 
113  if (_active_vars)
114  for (auto v : *_active_vars)
115  check_var(v);
116  else
117  for (auto v : make_range(sys.n_vars()))
118  check_var(v);
119 
120  return hardest_fe_type;
121 }

124 void FEMContext::attach_quadrature_rules()
125 {
126  const System & sys = this->get_system();
127 
128  auto attach_rules = [this, &sys](unsigned int v)
129  {
130  for (const auto & dim : _elem_dims)
131  {
132  FEType fe_type = sys.variable_type(v);
133 
134  _element_fe[dim][fe_type]->attach_quadrature_rule(_element_qrule[dim].get());
135  if (dim)
136  _side_fe[dim][fe_type]->attach_quadrature_rule(_side_qrule[dim].get());
137  if (dim == 3)
138  _edge_fe[fe_type]->attach_quadrature_rule(_edge_qrule.get());
139  };
140  };
141 
142  if (_active_vars)
143  for (auto v : *_active_vars)
144  attach_rules(v);
145  else
146  for (auto v : make_range(sys.n_vars()))
147  attach_rules(v);

153 {
154  _extra_quadrature_order = extra_quadrature_order;
155 
156  FEType hardest_fe_type = this->find_hardest_fe_type();
157 
158  for (const auto & dim : _elem_dims)
159  {
160  // Create an adequate quadrature rule
161  _element_qrule[dim] =
162  hardest_fe_type.default_quadrature_rule(dim, _extra_quadrature_order);
163  if (dim)
164  _side_qrule[dim] =
165  hardest_fe_type.default_quadrature_rule(dim-1, _extra_quadrature_order);
166  if (dim == 3)
167  _edge_qrule = hardest_fe_type.default_quadrature_rule(1, _extra_quadrature_order);
168  }
169 
170  this->attach_quadrature_rules();
171 }

195 void FEMContext::init_internal_data(const System & sys)
196 {
197  // Reserve space for the FEAbstract and QBase objects for each
198  // element dimension possibility (0,1,2,3)
199 
200  // Note: we would simply resize() all four of these vectors, but
201  // some compilers (ICC 19, MSVC) generate a diagnostic about copying
202  // a std::unique_ptr in this case, so the following two lines are a
203  // workaround.
204  _element_fe = std::vector<std::map<FEType, std::unique_ptr<FEAbstract>>>(4);
205  _side_fe = std::vector<std::map<FEType, std::unique_ptr<FEAbstract>>>(4);
206  _element_fe_var.resize(4);
207  _side_fe_var.resize(4);
208 
209  // We need to know which of our variables has the hardest
210  // shape functions to numerically integrate.
211 
212  unsigned int nv = sys.n_vars();
213  libmesh_assert (nv);
214 
215  bool have_scalar = false;
216 
217  if (_active_vars)
218  {
219  for (auto v : *_active_vars)
220  if (sys.variable_type(v).family == SCALAR)
221  {
222  have_scalar = true;
223  break;
224  }
225  }
226  else
227  {
228  for (auto v : make_range(sys.n_vars()))
229  if (sys.variable_type(v).family == SCALAR)
230  {
231  have_scalar = true;
232  break;
233  }
234  }
235 
236  if (have_scalar)
237  // SCALAR FEs have dimension 0 by assumption
238  _elem_dims.insert(0);
239 
240  auto build_var_fe = [this, &sys](unsigned int dim,
241  unsigned int i)
242  {
243  FEType fe_type = sys.variable_type(i);
244  const auto & dof_map = sys.get_dof_map();
245  const bool add_p_level = dof_map.should_p_refine(dof_map.var_group_from_var_number(i));
246 
247  auto & element_fe = _element_fe[dim][fe_type];
248  auto & side_fe = _side_fe[dim][fe_type];
249  if (!element_fe)
250  {
251  element_fe = FEAbstract::build(dim, fe_type);
252  element_fe->add_p_level_in_reinit(add_p_level);
253  side_fe = FEAbstract::build(dim, fe_type);
254  side_fe->add_p_level_in_reinit(add_p_level);
255 
256  if (dim == 3)
257  {
258  auto & edge_fe = _edge_fe[fe_type];
259  edge_fe = FEAbstract::build(dim, fe_type);
260  edge_fe->add_p_level_in_reinit(add_p_level);
261  }
262  }
263 
264  _element_fe_var[dim][i] = element_fe.get();
265  _side_fe_var[dim][i] = side_fe.get();
266  if ((dim) == 3)
267  _edge_fe_var[i] = _edge_fe[fe_type].get();
268  };
269 
270  for (const auto & dim : _elem_dims)
271  {
272  // Create finite element objects
273  _element_fe_var[dim].resize(nv);
274  _side_fe_var[dim].resize(nv);
275  if (dim == 3)
276  _edge_fe_var.resize(nv);
277 
278  if (_active_vars)
279  for (auto v : *_active_vars)
280  build_var_fe(dim, v);
281  else
282  for (auto v : make_range(nv))
283  build_var_fe(dim, v);
284  }
285 
286  this->use_default_quadrature_rules(_extra_quadrature_order);
287 }

309 template<typename OutputType,
310  typename FEMContext::FENeeded<OutputType>::value_getter fe_getter,
311  FEMContext::diff_subsolution_getter subsolution_getter>
312 void FEMContext::some_value(unsigned int var, unsigned int qp, OutputType & u) const
313 {
314  // Get local-to-global dof index lookup
315  const unsigned int n_dofs = cast_int<unsigned int>
316  (this->get_dof_indices(var).size());
317 
318  // Get current local coefficients
319  const DenseSubVector<Number> & coef = (this->*subsolution_getter)(var);
320 
321  // Get finite element object
322  typename FENeeded<OutputType>::value_base * fe = nullptr;
323  (this->*fe_getter)( var, fe, this->get_elem_dim() );
324 
325  // Get shape function values at quadrature point
326  const std::vector<std::vector
327  <typename FENeeded<OutputType>::value_shape>> & phi = fe->get_phi();
328 
329  // Accumulate solution value
330  u = 0.;
331 
332  for (unsigned int l=0; l != n_dofs; l++)
333  u += phi[l][qp] * coef(l);
334 }

2013 template<typename OutputShape>
2014 FEGenericBase<OutputShape> *
2015 FEMContext::build_new_fe( const FEGenericBase<OutputShape>* fe,
2016  const Point & p,
2017  const Real tolerance,
2018  const int get_derivative_level) const
2019 {
2020  FEType fe_type = fe->get_fe_type();
2021 
2022  // If we don't have an Elem to evaluate on, then the only functions
2023  // we can sensibly evaluate are the scalar dofs which are the same
2024  // everywhere.
2025  libmesh_assert(this->has_elem() || fe_type.family == SCALAR);
2026 
2027 #ifdef LIBMESH_ENABLE_AMR
2028  const bool add_p_level = fe->add_p_level_in_reinit();
2029  if ((algebraic_type() == OLD) &&
2030  this->has_elem())
2031  {
2032  if (this->get_elem().p_refinement_flag() == Elem::JUST_REFINED)
2033  fe_type.order = static_cast<Order>(fe_type.order - add_p_level);
2034  else if (this->get_elem().p_refinement_flag() == Elem::JUST_COARSENED)
2035  fe_type.order = static_cast<Order>(fe_type.order + add_p_level);
2036  }
2037 #endif // LIBMESH_ENABLE_AMR
2038 
2039  const unsigned int elem_dim = this->has_elem() ? this->get_elem().dim() : 0;
2040 
2041  FEGenericBase<OutputShape>* fe_new =
2042  cached_fe<OutputShape>(elem_dim, fe_type, get_derivative_level);
2043 #ifdef LIBMESH_ENABLE_AMR
2044  fe_new->add_p_level_in_reinit(add_p_level);
2045 #endif // LIBMESH_ENABLE_AMR
2046 
2047  // Map the physical co-ordinates to the master co-ordinates using the inverse_map from fe_interface.h
2048  // Build a vector of point co-ordinates to send to reinit
2049  Point master_point = this->has_elem() ?
2050  FEMap::inverse_map (elem_dim, &this->get_elem(), p, tolerance) :
2051  Point(0);
2052 
2053  std::vector<Point> coor(1, master_point);
2054 
2055  switch (get_derivative_level)
2056  {
2057  case -1:
2058  fe_new->get_phi();
2059  fe_new->get_dphi();
2060 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
2061  fe_new->get_d2phi();
2062 #endif
2063  fe_new->get_curl_phi();
2064  break;
2065  case 0:
2066  fe_new->get_phi();
2067  break;
2068  case 1:
2069  fe_new->get_dphi();
2070  break;
2071  case 2:
2072 #ifdef LIBMESH_ENABLE_SECOND_DERIVATIVES
2073  fe_new->get_d2phi();
2074 #else
2075  // here a different configuration is required.
2076  libmesh_not_implemented();
2077 #endif
2078  break;
2079  case 3:
2080  fe_new->get_curl_phi();
2081  break;
2082  default:
2083  libmesh_error();
2084  }
2085 
2086  // Reinitialize the element and compute the shape function values at coor
2087  if (this->has_elem())
2088  fe_new->reinit (&this->get_elem(), &coor);
2089  else
2090  // If !this->has_elem(), then we assume we are dealing with a SCALAR variable
2091  fe_new->reinit (nullptr, &coor);
2092 
2093  return fe_new;
2094 }
