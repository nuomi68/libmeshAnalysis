152 void Tri3::connectivity(const unsigned int libmesh_dbg_var(sf),
153  const IOPackage iop,

155 {
156  libmesh_assert_less (sf, this->n_sub_elem());

204 std::pair<Real, Real> Tri3::min_and_max_angle() const
205 {
206  Point v10 ( this->point(1) - this->point(0) );
207  Point v20 ( this->point(2) - this->point(0) );
208  Point v21 ( this->point(2) - this->point(1) );
209 
210  const Real
211  len_10=v10.norm(),
212  len_20=v20.norm(),
213  len_21=v21.norm();
214 
215  const Real
216  theta0=std::acos(( v10*v20)/len_10/len_20),
217  theta1=std::acos((-v10*v21)/len_10/len_21),
218  theta2=libMesh::pi - theta0 - theta1
219  ;
220 
221  libmesh_assert_greater (theta0, 0.);
222  libmesh_assert_greater (theta1, 0.);
223  libmesh_assert_greater (theta2, 0.);
224 
225  return std::make_pair(std::min(theta0, std::min(theta1,theta2)),
226  std::max(theta0, std::max(theta1,theta2)));
227 }

229 bool Tri3::contains_point (const Point & p, Real tol) const
230 {
231  // See "Barycentric Technique" section at
232  // http://www.blackpawn.com/texts/pointinpoly for details.
233 
234  // Compute vectors
235  Point v0 = this->point(1) - this->point(0);
236  Point v1 = this->point(2) - this->point(0);
237  Point v2 = p - this->point(0);
238 
239  // Compute dot products
240  Real dot00 = v0 * v0;
241  Real dot01 = v0 * v1;
242  Real dot02 = v0 * v2;
243  Real dot11 = v1 * v1;
244  Real dot12 = v1 * v2;
245 
246  // Out of plane check
247  if (std::abs(triple_product(v2, v0, v1)) / std::max(dot00, dot11) > tol)
248  return false;
249 
250  // Compute barycentric coordinates
251  Real invDenom = 1 / (dot00 * dot11 - dot01 * dot01);
252  Real u = (dot11 * dot02 - dot01 * dot12) * invDenom;
253  Real v = (dot00 * dot12 - dot01 * dot02) * invDenom;
254 
255  // Check if point is in triangle
256  return (u > -tol) && (v > -tol) && (u + v < 1 + tol);
257 }
