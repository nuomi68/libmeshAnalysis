181 Real Quad::quality (const ElemQuality q) const
182 {
183  switch (q)
184  {
185  // CUBIT 15.1 User Documentation:
186  // Aspect Ratio: Maximum edge length ratios
187  case ASPECT_RATIO:
188  {
189  Real lengths[4] = {this->length(0,1), this->length(1,2), this->length(2,3), this->length(3,0)};
190  Real
191  max = *std::max_element(lengths, lengths+4),
192  min = *std::min_element(lengths, lengths+4);
193 
194  // Return 0. instead of dividing by zero.
195  if (min == 0.)
196  return 0.;
197  else
198  return max / min;
199  }
200 
201  // Compute the min/max diagonal ratio.
202  // This is modeled after the Hex element
203  case DISTORTION:
204  case DIAGONAL:
205  {
206  // Diagonal between node 0 and node 2
207  const Real d02 = this->length(0,2);
208 
209  // Diagonal between node 1 and node 3
210  const Real d13 = this->length(1,3);
211 
212  // Find the biggest and smallest diagonals
213  if ((d02 > 0.) && (d13 >0.))
214  if (d02 < d13) return d02 / d13;
215  else return d13 / d02;
216  else
217  return 0.;
218  break;
219  }
220 
221  // CUBIT 15.1 User Documentation:
222  // Stretch: Sqrt(2) * minimum edge length / maximum diagonal length
223  case STRETCH:
224  {
225  Real lengths[4] = {this->length(0,1), this->length(1,2), this->length(2,3), this->length(3,0)};
226  Real min_edge = *std::min_element(lengths, lengths+4);
227  Real d_max = std::max(this->length(0,2), this->length(1,3));
228 
229  // Return 0. instead of dividing by zero.
230  if (d_max == 0.)
231  return 0.;
232  else
233  return std::sqrt(2) * min_edge / d_max;
234  }
235 
236  case SHAPE:
237  case SKEW:
238  {
239  // From: P. Knupp, "Algebraic mesh quality metrics for
240  // unstructured initial meshes," Finite Elements in Analysis
241  // and Design 39, 2003, p. 217-241, Sections 5.2 and 5.3.
242  typedef std::array<Real, 4> Array4;
243  typedef std::array<Real, 6> Array6;
244 
245  // x, y, z node coordinates.
246  std::vector<Real>
247  x = {this->point(0)(0), this->point(1)(0), this->point(2)(0), this->point(3)(0)},
248  y = {this->point(0)(1), this->point(1)(1), this->point(2)(1), this->point(3)(1)},
249  z = {this->point(0)(2), this->point(1)(2), this->point(2)(2), this->point(3)(2)};
250 
251  // Nodal Jacobians. These are 3x2 matrices, hence we represent
252  // them by Array6.
253  std::array<Array6, 4> A;
254  for (unsigned int k=0; k<4; ++k)
255  {
256  unsigned int
257  kp1 = k+1 > 3 ? k+1-4 : k+1,
258  kp3 = k+3 > 3 ? k+3-4 : k+3;
259 
260  // To initialize std::array we need double curly braces in
261  // C++11 but not C++14 apparently.
262  A[k] = {{x[kp1] - x[k], x[kp3] - x[k],
263  y[kp1] - y[k], y[kp3] - y[k],
264  z[kp1] - z[k], z[kp3] - z[k]}};
265  }
266 
267  // Compute metric tensors, T_k = A_k^T * A_k. These are 2x2
268  // square matrices, hence we represent them by Array4.
269  std::array<Array4, 4> T;
270  for (unsigned int k=0; k<4; ++k)
271  {
272  Real
273  top_left = A[k][0]*A[k][0] + A[k][2]*A[k][2] + A[k][4]*A[k][4],
274  off_diag = A[k][0]*A[k][1] + A[k][2]*A[k][3] + A[k][4]*A[k][5],
275  bot_rigt = A[k][1]*A[k][1] + A[k][3]*A[k][3] + A[k][5]*A[k][5];
276 
277  T[k] = {{top_left, off_diag,
278  off_diag, bot_rigt}};
279  }
280 
281 
282  // Nodal areas. These are approximated as sqrt(det(A^T * A))
283  // to handle the general case of a 2D element living in 3D.
284  Array4 alpha;
285  for (unsigned int k=0; k<4; ++k)
286  alpha[k] = std::sqrt(T[k][0]*T[k][3] - T[k][1]*T[k][2]);
287 
288  // All nodal areas must be strictly positive. Return 0 (the
289  // lowest quality) otherwise. We know they can't be negative
290  // because they are the result of a sqrt, but they might be
291  // identically 0.
292  if (*std::min_element(alpha.begin(), alpha.end()) == 0.)
293  return 0.;
294 
295  // Compute and return the shape metric. These only use the
296  // diagonal entries of the T_k.
297  Real den = 0.;
298  if (q == SHAPE)
299  {
300  for (unsigned int k=0; k<4; ++k)
301  den += (T[k][0] + T[k][3]) / alpha[k];
302  return (den == 0.) ? 0 : (8. / den);
303  }
304  else
305  {
306  for (unsigned int k=0; k<4; ++k)
307  den += std::sqrt(T[k][0] * T[k][3]) / alpha[k];
308  return (den == 0.) ? 0 : (4. / den);
309  }
310  }
311 
312  // This test returns 0 if a Quad:
313  // 1) Is "twisted" (i.e. has an invalid numbering)
314  // 2) Has nearly parallel adjacent edges
315  // 3) Has a nearly zero length edge
316  // Otherwise, it returns 1.
317  case TWIST:
318  {
319  // Compute the cross product induced by each "corner" of the QUAD
320  // and check that:
321  // 1) None of the cross products are zero (within a tolernace), and
322  // 2) all of the cross products point in the same direction.
323 
324  // Corner 0
325  Point vec_01 = point(1) - point(0);
326  Point vec_03 = point(3) - point(0);
327  Point corner_0_vec = vec_01.cross(vec_03);
328 
329  // Corner 1
330  Point vec_12 = point(2) - point(1);
331  Point vec_10 = point(0) - point(1);
332  Point corner_1_vec = vec_12.cross(vec_10);
333 
334  // Corner 2
335  Point vec_23 = point(3) - point(2);
336  Point vec_21 = point(1) - point(2);
337  Point corner_2_vec = vec_23.cross(vec_21);
338 
339  // Corner 3
340  Point vec_30 = point(0) - point(3);
341  Point vec_32 = point(2) - point(3);
342  Point corner_3_vec = vec_30.cross(vec_32);
343 
344  // If any of these cross products is nearly zero, then either
345  // we have nearly parallel adjacent edges or a nearly zero
346  // length edge. We return 0 in this case.
347  if ((corner_0_vec.norm() < TOLERANCE*TOLERANCE) ||
348  (corner_1_vec.norm() < TOLERANCE*TOLERANCE) ||
349  (corner_2_vec.norm() < TOLERANCE*TOLERANCE) ||
350  (corner_3_vec.norm() < TOLERANCE*TOLERANCE))
351  {
352  return 0.;
353  }
354 
355  // Now check whether the element is twisted.
356  Real dot_01 = corner_0_vec * corner_1_vec;
357  Real dot_02 = corner_0_vec * corner_2_vec;
358  Real dot_03 = corner_0_vec * corner_3_vec;
359 
360  if ((dot_01 <= 0.) || (dot_02 <= 0.) || (dot_03 <= 0.))
361  return 0.;
362 
363  // If we made it here, then Elem is not twisted.
364  return 1.;
365  }
366 
367  default:
368  return Elem::quality(q);
369  }
