147 dof_id_type Pyramid18::key (const unsigned int s) const
148 {
149  libmesh_assert_less (s, this->n_sides());
150 
151  switch (s)
152  {
153  case 0: // triangular face 1
154  case 1: // triangular face 2
155  case 2: // triangular face 3
156  case 3: // triangular face 4
157  return this->compute_key (this->node_id(s+14));
158 
159  case 4: // the quad face at z=0
160  return this->compute_key (this->node_id(13));
161 
162  default:
163  libmesh_error_msg("Invalid side s = " << s);
164  }

196 std::unique_ptr<Elem> Pyramid18::build_side_ptr (const unsigned int i, bool proxy)
197 {
198  libmesh_assert_less (i, this->n_sides());
199 
200  std::unique_ptr<Elem> face;
201  if (proxy)
202  {
203 #ifdef LIBMESH_ENABLE_DEPRECATED
204  libmesh_deprecated();
205  switch (i)
206  {
207  case 0:
208  case 1:
209  case 2:
210  case 3:
211  {
212  face = std::make_unique<Side<Tri7,Pyramid18>>(this,i);
213  break;
214  }
215 
216  case 4:
217  {
218  face = std::make_unique<Side<Quad9,Pyramid18>>(this,i);
219  break;
220  }
221 
222  default:
223  libmesh_error_msg("Invalid side i = " << i);
224  }
225 #else
226  libmesh_error();
227 #endif // LIBMESH_ENABLE_DEPRECATED
228  }
229 
230  else
231  {
232  switch (i)
233  {
234  case 0: // triangular face 1
235  case 1: // triangular face 2
236  case 2: // triangular face 3
237  case 3: // triangular face 4
238  {
239  face = std::make_unique<Tri7>();
240  break;
241  }
242  case 4: // the quad face at z=0
243  {
244  face = std::make_unique<Quad9>();
245  break;
246  }
247  default:
248  libmesh_error_msg("Invalid side i = " << i);
249  }
250 
251  // Set the nodes
252  for (auto n : face->node_index_range())
253  face->set_node(n) = this->node_ptr(Pyramid18::side_nodes_map[i][n]);
254  }
255 
256 #ifdef LIBMESH_ENABLE_DEPRECATED
257  if (!proxy) // proxy sides used to leave parent() set
258 #endif
259  face->set_parent(nullptr);
260  face->set_interior_parent(this);
261 
262  face->subdomain_id() = this->subdomain_id();
263  face->set_mapping_type(this->mapping_type());
264 #ifdef LIBMESH_ENABLE_AMR
265  face->set_p_level(this->p_level());
266 #endif
267 
268  return face;
269 }

273 void Pyramid18::build_side_ptr (std::unique_ptr<Elem> & side,
274  const unsigned int i)
275 {
276  libmesh_assert_less (i, this->n_sides());
277 
278  switch (i)
279  {
280  case 0: // triangular face 1
281  case 1: // triangular face 2
282  case 2: // triangular face 3
283  case 3: // triangular face 4
284  {
285  if (!side.get() || side->type() != TRI7)
286  {
287  side = this->build_side_ptr(i, false);
288  return;
289  }
290  break;
291  }
292  case 4: // the quad face at z=0
293  {
294  if (!side.get() || side->type() != QUAD9)
295  {
296  side = this->build_side_ptr(i, false);
297  return;
298  }
299  break;
300  }
301  default:
302  libmesh_error_msg("Invalid side i = " << i);
303  }
304 
305  side->subdomain_id() = this->subdomain_id();
306  side->set_mapping_type(this->mapping_type());
307 
308  // Set the nodes
309  for (auto n : side->node_index_range())
310  side->set_node(n) = this->node_ptr(Pyramid18::side_nodes_map[i][n]);
311 }

387 unsigned short int Pyramid18::second_order_adjacent_vertex (const unsigned int n,
388  const unsigned int v) const
389 {
390  libmesh_assert_greater_equal (n, this->n_vertices());
391  libmesh_assert_less (n, this->n_nodes());
392 
393  switch (n)
394  {
395  case 5:
396  case 6:
397  case 7:
398  case 8:
399  case 9:
400  case 10:
401  case 11:
402  case 12:
403  {
404  libmesh_assert_less (v, 2);
405 
406  // This is the analog of the static, const arrays
407  // {Hex,Prism,Tet10}::_second_order_adjacent_vertices
408  // defined in the respective source files... possibly treat
409  // this similarly once the Pyramid13 has been added?
410  constexpr unsigned short node_list[8][2] =
411  {
412  {0,1},
413  {1,2},
414  {2,3},
415  {0,3},
416  {0,4},
417  {1,4},
418  {2,4},
419  {3,4}
420  };
421 
422  return node_list[n-5][v];
423  }
424 
425  // mid-face node on bottom
426  case 13:
427  {
428  libmesh_assert_less (v, 4);
429 
430  // The vertex nodes surrounding node 13 are 0, 1, 2, and 3.
431  // Thus, the v'th node is simply = v.
432  return cast_int<unsigned short>(v);
433  }
434 
435  // mid-face nodes on triangles
436  case 14:
437  case 15:
438  case 16:
439  case 17:
440  {
441  libmesh_assert_less (v, 3);
442 
443  constexpr unsigned short node_list[4][3] =
444  {
445  {0,1,4},
446  {1,2,4},
447  {2,3,4},
448  {0,3,4}
449  };
