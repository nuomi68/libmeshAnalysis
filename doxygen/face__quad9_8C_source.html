160 bool Quad9::has_affine_map() const
161 {
162  // make sure corners form a parallelogram
163  Point v = this->point(1) - this->point(0);
164  if (!v.relative_fuzzy_equals(this->point(2) - this->point(3), affine_tol))
165  return false;
166  // make sure "horizontal" sides are straight
167  v /= 2;
168  if (!v.relative_fuzzy_equals(this->point(4) - this->point(0), affine_tol) ||
169  !v.relative_fuzzy_equals(this->point(6) - this->point(3), affine_tol))
170  return false;
171  // make sure "vertical" sides are straight
172  // and the center node is centered
173  v = (this->point(3) - this->point(0))/2;
174  if (!v.relative_fuzzy_equals(this->point(7) - this->point(0), affine_tol) ||
175  !v.relative_fuzzy_equals(this->point(5) - this->point(1), affine_tol) ||
176  !v.relative_fuzzy_equals(this->point(8) - this->point(4), affine_tol))
177  return false;
178  return true;
179 }

257 void Quad9::connectivity(const unsigned int sf,
258  const IOPackage iop,
259  std::vector<dof_id_type> & conn) const
260 {
261  libmesh_assert_less (sf, this->n_sub_elem());
262  libmesh_assert_not_equal_to (iop, INVALID_IO_PACKAGE);
263 
264  conn.resize(4);
265 
266  switch (iop)
267  {
268  case TECPLOT:
269  {
270  switch(sf)
271  {
272  case 0:
273  // linear sub-quad 0
274  conn[0] = this->node_id(0)+1;
275  conn[1] = this->node_id(4)+1;
276  conn[2] = this->node_id(8)+1;
277  conn[3] = this->node_id(7)+1;
278  return;
279 
280  case 1:
281  // linear sub-quad 1
282  conn[0] = this->node_id(4)+1;
283  conn[1] = this->node_id(1)+1;
284  conn[2] = this->node_id(5)+1;
285  conn[3] = this->node_id(8)+1;
286  return;
287 
288  case 2:
289  // linear sub-quad 2
290  conn[0] = this->node_id(7)+1;
291  conn[1] = this->node_id(8)+1;
292  conn[2] = this->node_id(6)+1;
293  conn[3] = this->node_id(3)+1;
294  return;
295 
296  case 3:
297  // linear sub-quad 3
298  conn[0] = this->node_id(8)+1;
299  conn[1] = this->node_id(5)+1;
300  conn[2] = this->node_id(2)+1;
301  conn[3] = this->node_id(6)+1;
302  return;
303 
304  default:
305  libmesh_error_msg("Invalid sf = " << sf);
306  }
307  }
308 
309  case VTK:
310  {
311  conn.resize(Quad9::num_nodes);
312  for (auto i : index_range(conn))
313  conn[i] = this->node_id(i);
314  return;
315  }
316 
317  default:
318  libmesh_error_msg("Unsupported IO package " << iop);
319  }
320 }

324 BoundingBox Quad9::loose_bounding_box () const
325 {
326  // This might have curved edges, or might be a curved surface in
327  // 3-space, in which case the full bounding box can be larger than
328  // the bounding box of just the nodes.
329  //
330  //
331  // FIXME - I haven't yet proven the formula below to be correct for
332  // biquadratics - RHS
333  Point pmin, pmax;
334 
335  for (unsigned d=0; d<LIBMESH_DIM; ++d)
336  {
337  const Real center = this->point(8)(d);
338  Real hd = std::abs(center - this->point(0)(d));
339  for (unsigned int p=0; p != 8; ++p)
340  hd = std::max(hd, std::abs(center - this->point(p)(d)));
341 
342  pmin(d) = center - hd;
343  pmax(d) = center + hd;
344  }
345 
346  return BoundingBox(pmin, pmax);
347 }

351 Real Quad9::volume () const
352 {
353  // This specialization is good for Lagrange mappings only
354  if (this->mapping_type() != LAGRANGE_MAP)
355  return this->Elem::volume();
356 
357  // Make copies of our points. It makes the subsequent calculations a bit
358  // shorter and avoids dereferencing the same pointer multiple times.
359  Point
360  x0 = point(0), x1 = point(1), x2 = point(2),
361  x3 = point(3), x4 = point(4), x5 = point(5),
362  x6 = point(6), x7 = point(7), x8 = point(8);
363 
364  // Construct constant data vectors.
365  // \vec{x}_{\xi} = \vec{a1}*xi*eta^2 + \vec{b1}*eta**2 + \vec{c1}*xi*eta + \vec{d1}*xi + \vec{e1}*eta + \vec{f1}
366  // \vec{x}_{\eta} = \vec{a2}*xi^2*eta + \vec{b2}*xi**2 + \vec{c2}*xi*eta + \vec{d2}*xi + \vec{e2}*eta + \vec{f2}
367  // This is copy-pasted directly from the output of a Python script.
368  Point
369  a1 = x0/2 + x1/2 + x2/2 + x3/2 - x4 - x5 - x6 - x7 + 2*x8,
370  b1 = -x0/4 + x1/4 + x2/4 - x3/4 - x5/2 + x7/2,
371  c1 = -x0/2 - x1/2 + x2/2 + x3/2 + x4 - x6,
372  d1 = x5 + x7 - 2*x8,
373  e1 = x0/4 - x1/4 + x2/4 - x3/4,
374  f1 = x5/2 - x7/2,
375  a2 = a1,
376  b2 = -x0/4 - x1/4 + x2/4 + x3/4 + x4/2 - x6/2,
377  c2 = -x0/2 + x1/2 + x2/2 - x3/2 - x5 + x7,
378  d2 = x0/4 - x1/4 + x2/4 - x3/4,
379  e2 = x4 + x6 - 2*x8,
380  f2 = -x4/2 + x6/2;
381 
382  // 3x3 quadrature, exact for bi-quintics
383  const unsigned int N = 3;
384  const Real q[N] = {-std::sqrt(15)/5., 0., std::sqrt(15)/5.};
385  const Real w[N] = {5./9, 8./9, 5./9};
386 
387  Real vol=0.;
388  for (unsigned int i=0; i<N; ++i)
389  for (unsigned int j=0; j<N; ++j)
390  vol += w[i] * w[j] *
391  cross_norm(q[i]*q[j]*q[j]*a1 + q[j]*q[j]*b1 + q[j]*q[i]*c1 + q[i]*d1 + q[j]*e1 + f1,
392  q[i]*q[i]*q[j]*a2 + q[i]*q[i]*b2 + q[j]*q[i]*c2 + q[i]*d2 + q[j]*e2 + f2);
393 
394  return vol;
395 }
