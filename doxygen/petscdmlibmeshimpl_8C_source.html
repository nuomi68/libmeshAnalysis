93 PetscErrorCode DMlibMeshSetSystem_libMesh(DM dm, NonlinearImplicitSystem & sys)
94 {
95  const Parallel::Communicator & comm = sys.comm();
96 
97  PetscErrorCode ierr;
98  PetscFunctionBegin;
99  PetscValidHeaderSpecific(dm,DM_CLASSID,1);
100  PetscBool islibmesh;
101  ierr = PetscObjectTypeCompare((PetscObject)dm, DMLIBMESH,&islibmesh);
102  CHKERRQ(ierr);
103  if (!islibmesh) LIBMESH_SETERRQ2(((PetscObject)dm)->comm, PETSC_ERR_ARG_WRONG, "Got DM of type %s, not of type %s", ((PetscObject)dm)->type_name, DMLIBMESH);
104 
105  if (dm->setupcalled) SETERRQ(((PetscObject)dm)->comm, PETSC_ERR_ARG_WRONGSTATE, "Cannot reset the libMesh system after DM has been set up.");
106  DM_libMesh * dlm = (DM_libMesh *)(dm->data);
107  dlm->sys =&sys;
108  /* Initially populate the sets of active blockids and varids using all of the
109  existing blocks/variables (only variables are supported at the moment). */
110  DofMap & dofmap = dlm->sys->get_dof_map();
111  dlm->varids->clear();
112  dlm->varnames->clear();
113  for (auto v : make_range(dofmap.n_variables())) {
114  std::string vname = dofmap.variable(v).name();
115  dlm->varids->insert(std::pair<std::string,unsigned int>(vname,v));
116  dlm->varnames->insert(std::pair<unsigned int,std::string>(v,vname));
117  }
118  const MeshBase & mesh = dlm->sys->get_mesh();
119  dlm->blockids->clear();
120  dlm->blocknames->clear();
121  std::set<subdomain_id_type> blocks;
122  /* The following effectively is a verbatim copy of MeshBase::n_subdomains(). */
123  // This requires an inspection on every processor
124  libmesh_parallel_only(mesh.comm());
125  for (const auto & elem : mesh.active_element_ptr_range())
126  blocks.insert(elem->subdomain_id());
127  // Some subdomains may only live on other processors
128  comm.set_union(blocks);
129 
130  std::set<subdomain_id_type>::iterator bit = blocks.begin();
131  std::set<subdomain_id_type>::iterator bend = blocks.end();
132  if (bit == bend) SETERRQ(((PetscObject)dm)->comm, PETSC_ERR_PLIB, "No mesh blocks found.");
133 
134  for (; bit != bend; ++bit) {
135  subdomain_id_type bid = *bit;
136  std::string bname = mesh.subdomain_name(bid);
137  if (!bname.length()) {
138  /* Block names are currently implemented for Exodus II meshes
139  only, so we might have to make up our own block names and
140  maintain our own mapping of block ids to names.
141  */
142  std::ostringstream ss;
143  ss << "dm" << bid;
144  bname = ss.str();
145  }
146  dlm->blockids->insert(std::pair<std::string,unsigned int>(bname,bid));
147  dlm->blocknames->insert(std::pair<unsigned int,std::string>(bid,bname));
148  }
149  ierr = DMlibMeshSetUpName_Private(dm); CHKERRQ(ierr);
150  PetscFunctionReturn(0);
151 }

232 PetscErrorCode DMlibMeshSetUpName_Private(DM dm)
233 {
234  DM_libMesh * dlm = (DM_libMesh *)dm->data;
235  PetscErrorCode ierr;
236  PetscFunctionBegin;
237  std::string name = dlm->sys->name();
238  std::map<unsigned int, std::string> * dnames = LIBMESH_PETSC_NULLPTR,
239  * enames = LIBMESH_PETSC_NULLPTR;
240  if (dlm->decomposition_type == DMLIBMESH_FIELD_DECOMPOSITION) {
241  name += ":dec:var:";
242  dnames = dlm->varnames;
243  }
244  if (dlm->decomposition_type == DMLIBMESH_DOMAIN_DECOMPOSITION) {
245  name += ":dec:block:";
246  dnames = dlm->blocknames;
247  }
248  if (dnames) {
249  for (auto decomp : *dlm->decomposition) {
250  for (std::set<unsigned int>::iterator dit_begin = decomp.begin(),
251  dit = dit_begin,
252  dit_end = decomp.end();
253  dit != dit_end; ++dit) {
254  unsigned int id = *dit;
255  if (dit != dit_begin)
256  name += ",";
257  name += (*dnames)[id];
258  }
259  name += ";";
260  }
261  }
262  if (dlm->embedding_type == DMLIBMESH_FIELD_EMBEDDING) {
263  name += ":emb:var:";
264  enames = dlm->varnames;
265  }
266  if (dlm->embedding_type == DMLIBMESH_DOMAIN_EMBEDDING) {
267  name += ":emb:block:";
268  enames = dlm->blocknames;
269  }
270  if (enames) {
271  for (auto eit = enames->begin(),
272  eit_end = enames->end(); eit != eit_end; ++eit) {
273  std::string & ename = eit->second;
274  if (eit != enames->begin())
275  name += ",";
276  name += ename;
277  }
278  name += ";";
279  }
280  ierr = PetscObjectSetName((PetscObject)dm, name.c_str()); CHKERRQ(ierr);
281  PetscFunctionReturn(0);
282 }

287 static PetscErrorCode DMCreateFieldDecomposition_libMesh(DM dm, PetscInt * len, char *** namelist, IS ** islist, DM ** dmlist)
288 {
289  PetscFunctionBegin;
290  PetscErrorCode ierr;
291  DM_libMesh * dlm = (DM_libMesh *)(dm->data);
292  NonlinearImplicitSystem * sys = dlm->sys;
293  IS emb;
294  if (dlm->decomposition_type != DMLIBMESH_FIELD_DECOMPOSITION) PetscFunctionReturn(0);
295 
296  *len = cast_int<unsigned int>(dlm->decomposition->size());
297  if (namelist) {ierr = PetscMalloc(*len*sizeof(char *), namelist); CHKERRQ(ierr);}
298  if (islist) {ierr = PetscMalloc(*len*sizeof(IS), islist); CHKERRQ(ierr);}
299  if (dmlist) {ierr = PetscMalloc(*len*sizeof(DM), dmlist); CHKERRQ(ierr);}
300  DofMap & dofmap = dlm->sys->get_dof_map();
301  for (auto d : index_range(*dlm->decomposition)) {
302  std::set<numeric_index_type> dindices;
303  std::string dname;
304  std::map<std::string, unsigned int> dvarids;
305  std::map<unsigned int, std::string> dvarnames;
306  unsigned int dvcount = 0;
307  for (const auto & v : (*dlm->decomposition)[d]) {
308  std::string vname = (*dlm->varnames)[v];
309  dvarids.insert(std::pair<std::string, unsigned int>(vname,v));
310  dvarnames.insert(std::pair<unsigned int,std::string>(v,vname));
311  if (!dvcount) dname = vname;
312  else dname += "_" + vname;
313  ++dvcount;
314  if (!islist) continue;
315  // Iterate only over this DM's blocks.
316  for (const auto & pr : *(dlm->blockids)) {
317  const subdomain_id_type sbd_id = cast_int<subdomain_id_type>(pr.second);
318  for (const auto & elem :
319  as_range(sys->get_mesh().active_local_subdomain_elements_begin(sbd_id),
320  sys->get_mesh().active_local_subdomain_elements_end(sbd_id))) {
321  //unsigned int e_subdomain = elem->subdomain_id();
322  std::vector<numeric_index_type> evindices;
323  // Get the degree of freedom indices for the given variable off the current element.
324  dofmap.dof_indices(elem, evindices, v);
325  for (numeric_index_type dof : evindices) {
326  if (dof >= dofmap.first_dof() && dof < dofmap.end_dof()) // might want to use variable_first/last_local_dof instead
327  dindices.insert(dof);
328  }
329  }
330  }
331  }
332  if (namelist) {
333  ierr = PetscStrallocpy(dname.c_str(),(*namelist)+d); CHKERRQ(ierr);
334  }
335  if (islist) {
336  IS dis;
337  PetscInt * darray;
338  ierr = PetscMalloc(sizeof(PetscInt)*dindices.size(), &darray); CHKERRQ(ierr);
339  numeric_index_type i = 0;
340  for (const auto & id : dindices) {
341  darray[i] = id;
342  ++i;
343  }
344  ierr = ISCreateGeneral(((PetscObject)dm)->comm,
345  cast_int<PetscInt>(dindices.size()),
346  darray, PETSC_OWN_POINTER, &dis);
347  CHKERRQ(ierr);
348  if (dlm->embedding) {
349  /* Create a relative embedding into the parent's index space. */
350  ierr = ISEmbed(dis,dlm->embedding, PETSC_TRUE, &emb); CHKERRQ(ierr);
351  PetscInt elen, dlen;
352  ierr = ISGetLocalSize(emb, &elen); CHKERRQ(ierr);
353  ierr = ISGetLocalSize(dis, &dlen); CHKERRQ(ierr);
354  if (elen != dlen) LIBMESH_SETERRQ1(((PetscObject)dm)->comm, PETSC_ERR_PLIB, "Failed to embed subdomain %zu", d);
355  ierr = ISDestroy(&dis); CHKERRQ(ierr);
356  dis = emb;
357  }
358  else {
359  emb = dis;
360  }
361  (*islist)[d] = dis;
362  }
363  if (dmlist) {
364  DM ddm;
365  ierr = DMCreate(((PetscObject)dm)->comm, &ddm); CHKERRQ(ierr);
366  ierr = DMSetType(ddm, DMLIBMESH); CHKERRQ(ierr);
367  DM_libMesh * ddlm = (DM_libMesh *)(ddm->data);
368  ddlm->sys = dlm->sys;
369  /* copy over the block ids and names */
370  *ddlm->blockids = *dlm->blockids;
371  *ddlm->blocknames = *dlm->blocknames;
372  /* set the vars from the d-th part of the decomposition. */
373  *ddlm->varids = dvarids;
374  *ddlm->varnames = dvarnames;
375  ierr = PetscObjectReference((PetscObject)emb); CHKERRQ(ierr);
376  ddlm->embedding = emb;
377  ddlm->embedding_type = DMLIBMESH_FIELD_EMBEDDING;
378 
379  ierr = DMlibMeshSetUpName_Private(ddm); CHKERRQ(ierr);
380  ierr = DMSetFromOptions(ddm); CHKERRQ(ierr);
381  (*dmlist)[d] = ddm;
382  }
383  }
384  PetscFunctionReturn(0);
385 }

617 {
