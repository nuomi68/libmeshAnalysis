134 std::unique_ptr<Elem> Pyramid5::build_side_ptr (const unsigned int i,
135  bool proxy)
136 {
137  libmesh_assert_less (i, this->n_sides());
138 
139  std::unique_ptr<Elem> face;
140  if (proxy)
141  {
142 #ifdef LIBMESH_ENABLE_DEPRECATED
143  libmesh_deprecated();
144  switch (i)
145  {
146  case 0:
147  case 1:
148  case 2:
149  case 3:
150  {
151  face = std::make_unique<Side<Tri3,Pyramid5>>(this,i);
152  break;
153  }
154 
155  case 4:
156  {
157  face = std::make_unique<Side<Quad4,Pyramid5>>(this,i);
158  break;
159  }
160 
161  default:
162  libmesh_error_msg("Invalid side i = " << i);
163  }
164 #else
165  libmesh_error();
166 #endif // LIBMESH_ENABLE_DEPRECATED
167  }
168  else
169  {
170  switch (i)
171  {
172  case 0: // triangular face 1
173  case 1: // triangular face 2
174  case 2: // triangular face 3
175  case 3: // triangular face 4
176  {
177  face = std::make_unique<Tri3>();
178  break;
179  }
180  case 4: // the quad face at z=0
181  {
182  face = std::make_unique<Quad4>();
183  break;
184  }
185  default:
186  libmesh_error_msg("Invalid side i = " << i);
187  }
188 
189  // Set the nodes
190  for (auto n : face->node_index_range())
191  face->set_node(n) = this->node_ptr(Pyramid5::side_nodes_map[i][n]);
192  }
193 
194 #ifdef LIBMESH_ENABLE_DEPRECATED
195  if (!proxy) // proxy sides used to leave parent() set
196 #endif
197  face->set_parent(nullptr);
198  face->set_interior_parent(this);
199 
200  face->subdomain_id() = this->subdomain_id();
201  face->set_mapping_type(this->mapping_type());
202 #ifdef LIBMESH_ENABLE_AMR
203  face->set_p_level(this->p_level());
204 #endif
205 
206  return face;
207 }

233 void Pyramid5::connectivity(const unsigned int libmesh_dbg_var(sc),
234  const IOPackage iop,
235  std::vector<dof_id_type> & conn) const
236 {
237  libmesh_assert(_nodes);
238  libmesh_assert_less (sc, this->n_sub_elem());
239  libmesh_assert_not_equal_to (iop, INVALID_IO_PACKAGE);
240 
241  switch (iop)
242  {
243  case TECPLOT:
244  {
245  conn.resize(8);
246  conn[0] = this->node_id(0)+1;
247  conn[1] = this->node_id(1)+1;
248  conn[2] = this->node_id(2)+1;
249  conn[3] = this->node_id(3)+1;
250  conn[4] = this->node_id(4)+1;
251  conn[5] = this->node_id(4)+1;
252  conn[6] = this->node_id(4)+1;
253  conn[7] = this->node_id(4)+1;
254  return;
255  }
256 
257  case VTK:
258  {
259  conn.resize(5);
260  conn[0] = this->node_id(3);
261  conn[1] = this->node_id(2);
262  conn[2] = this->node_id(1);
263  conn[3] = this->node_id(0);
264  conn[4] = this->node_id(4);
265  return;
266  }
267 
268  default:
269  libmesh_error_msg("Unsupported IO package " << iop);
270  }
271 }

274 Point Pyramid5::true_centroid () const
275 {
276  // Call Hex8 static helper function, passing 4 copies of the final
277  // vertex point, effectively treating the Pyramid as a degenerate
278  // hexahedron. In my testing, this still seems to give correct
279  // results.
280  return Hex8::centroid_from_points(
281  point(0), point(1), point(2), point(3),
282  point(4), point(4), point(4), point(4));
283 }

287 Real Pyramid5::volume () const
288 {
289  // The pyramid with a bilinear base has volume given by the
290  // formula in: "Calculation of the Volume of a General Hexahedron
291  // for Flow Predictions", AIAA Journal v.23, no.6, 1984, p.954-
292  Point
293  x0 = point(0), x1 = point(1), x2 = point(2),
294  x3 = point(3), x4 = point(4);
295 
296  // Construct various edge and diagonal vectors.
297  Point v40 = x0 - x4;
298  Point v13 = x3 - x1;
299  Point v02 = x2 - x0;
300  Point v03 = x3 - x0;
301  Point v01 = x1 - x0;
302 
303  // Finally, ready to return the volume!
304  return
305  triple_product(v40, v13, v02) / 6. +
306  triple_product(v02, v01, v03) / 12.;
307 }
