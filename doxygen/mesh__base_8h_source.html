1537  /*
1538  * element_iterator accessors
1539  *
1540  * The basic elements_begin() and elements_end() iterators iterate
1541  * over all elements in a mesh, returning element pointers or const
1542  * element pointers when dereferenced (depending on whether the mesh
1543  * reference was const). range-for loops can be written using
1544  * element_ptr_range()
1545  *
1546  * Filtered versions of these iterators, which skip over all
1547  * elements not matching some predicate, are also available, by
1548  * adding a prefix to the methods above. E.g. local_ (in a form
1549  * like local_elements_begin() or local_element_ptr_range()) will
1550  * iterate only over elements whose processor_id() is the current
1551  * processor, or active_ will iterate only over active elements even
1552  * if the mesh is refined, or active_local_ will iterate over
1553  * elements that are both active and local. Negation forms such as
1554  * not_local_ also exist.
1555  *
1556  * For some iterator prefixes, such as type_, an argument is needed
1557  * for the filter; e.g. the ElemType to select for in that case.
1558  *
1559  * All valid prefixes and their corresponding arguments can be found
1560  * in the macro invocations below.
1561  */
1562  ABSTRACT_ELEM_ITERATORS(,) // elements_begin(), element_ptr_range(): all elements
1563  ABSTRACT_ELEM_ITERATORS(active_,) // Elem::active() == true
1564  ABSTRACT_ELEM_ITERATORS(ancestor_,) // Elem::ancestor() == true
1565  ABSTRACT_ELEM_ITERATORS(subactive_,) // Elem::subactive() == true
1566  ABSTRACT_ELEM_ITERATORS(local_,) // Elem::processor_id() == this processor
1567  ABSTRACT_ELEM_ITERATORS(unpartitioned_,) // Elem::processor_id() == invalid_processor_id
1568  ABSTRACT_ELEM_ITERATORS(facelocal_,) // is on or has a neighbor on this processor
1569  ABSTRACT_ELEM_ITERATORS(level_,unsigned int level) // Elem::level() == level
1570  ABSTRACT_ELEM_ITERATORS(pid_,processor_id_type pid) // Elem::processor_id() == pid
1571  ABSTRACT_ELEM_ITERATORS(type_,ElemType type) // Elem::type() == type
